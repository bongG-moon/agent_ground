137def _safe_text(value: Any, maximum: int = 2_000) -> str:
138 """사용자/모델 문자열에서 실행 가능한 마크업을 제거한다."""
139
140 text = html.unescape(str(value or "")).replace("\x00", " ")
141 text = re.sub(r"<script\b[^>]*>[\s\S]*?</script\s*>", "", text, flags=re.I)
142 text = re.sub(r"<style\b[^>]*>[\s\S]*?</style\s*>", "", text, flags=re.I)
143 text = re.sub(r"<!--[\s\S]*?-->", "", text)
144 text = re.sub(r"<[^>]+>", "", text)
145 text = re.sub(r"(?:javascript|vbscript)\s*:", "", text, flags=re.I)
146 return text.strip()[:maximum]

215def _referenced_fields(value: Any) -> set[str]:
216 """encoding에서 데이터 컬럼을 의미하는 문자열을 재귀적으로 모은다."""
217
218 result: set[str] = set()
219 if isinstance(value, str) and value.strip():
220 result.add(value.strip())
221 elif isinstance(value, list):
222 for item in value:
223 result.update(_referenced_fields(item))
224 elif isinstance(value, dict):
225 for key, item in value.items():
226 if key in {"unit", "format", "aggregation", "sort", "direction", "label"}:
227 continue
228 result.update(_referenced_fields(item))

232def _normalize_encoding(value: Any, available: set[str], warnings: list[str], context: str) -> dict[str, Any] | None:
233 if not isinstance(value, dict):
234 return {}
235 clean: dict[str, Any] = {}
236 for key in ("x", "y", "series", "value", "columns", "category", "size", "color", "unit", "format"):
237 item = value.get(key)
238 if key in {"unit", "format"}:
239 if item is not None:
240 clean[key] = _safe_text(item, 100)
241 continue
242 if isinstance(item, str):
243 clean[key] = _safe_text(item, 200)
244 elif isinstance(item, list):
245 clean[key] = [_safe_text(entry, 200) for entry in item[:20] if isinstance(entry, str) and _safe_text(entry, 200)]
246 fields = _referenced_fields(clean)
247 missing = sorted(field for field in fields if field not in available)
248 if missing:
249 warnings.append(f"{context}: 데이터셋에 없는 컬럼 참조를 제거했습니다: {', '.join(missing[:8])}")
250 return None
251 return clean

633 views_by_id,
634 datasets,
635 dataset_columns,

643 element_id = f"s{index:02d}-title"
644 used_ids.add(element_id)
645 elements = [

647 "element_id": element_id,
648 "element_type": "text",

671 warnings.append("마지막 슬라이드 레이아웃을 conclusion으로 보정했습니다.")
672 slides[-1]["layout"] = "conclusion"
673 if not slides:
674 errors.append({"code": "empty_plan", "message": "정규화 후 표시할 슬라이드가 없습니다."})

676 brief = request.get("brief") if isinstance(request.get("brief"), dict) else {}

678 normalized_plan = {
679 "plan_version": "ppt-reference-plan-v1",
680 "title": _safe_text(plan.get("title") or brief.get("title") or "제목 미정 프레젠테이션", 300),
681 "language": _safe_text(plan.get("language") or brief.get("language") or "ko", 20),
682 "audience": _safe_text(plan.get("audience") or brief.get("audience"), 300),
683 "purpose": _safe_text(plan.get("purpose") or brief.get("purpose"), 1_000),

685 "design_system": _normalize_design(design_input),
686 "storyline": [_safe_text(item, 800) for item in plan.get("storyline", [])[:maximum] if _safe_text(item, 800)]
687 if isinstance(plan.get("storyline"), list)
688 else [slide["key_message"] for slide in slides],
689 "data_views": views,
690 "slides": slides,
691 "normalization_warnings": warnings,
692 "warnings": [],
693 "errors": errors,
694 }
695 clean_datasets = [_safe_dataset_copy(item) for item in datasets.values()]
