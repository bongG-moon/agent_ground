37def _payload(value: Any) -> dict[str, Any]:
38 if isinstance(value, dict):
39 return deepcopy(value)
40 data = getattr(value, "data", None)
41 if isinstance(data, dict):
42 return deepcopy(data)
43 text = getattr(value, "text", None) or getattr(value, "content", None)
44 if isinstance(text, str) and text.strip():
45 try:
46 parsed = json.loads(text)
47 return deepcopy(parsed) if isinstance(parsed, dict) else {}
48 except Exception:
49 return {}
50 return {}

90 def handle_starttag(self, tag: str, attrs: list[tuple[str, str | None]]) -> None:
91 tag = tag.lower()
92 self.tags[tag] = self.tags.get(tag, 0) + 1
93 attributes = {str(key).lower(): str(value or "") for key, value in attrs}
94 if tag in _PROHIBITED_TAGS:
95 self.prohibited_tags.append(tag)
96 if tag == "html":
97 self.html_lang = attributes.get("lang", "").strip()
98 if tag == "title":
99 self.has_title = True
100 if tag == "main":
101 self.has_main = True
102 if tag in {"section", "article"}:
103 classes = set(attributes.get("class", "").split())
104 if "slide" in classes or "presentation-slide" in classes or "data-slide" in attributes:
105 self.slide_sections += 1
106 if tag == "img":
107 self.images += 1
108 if not attributes.get("alt", "").strip():
109 self.images_without_alt += 1
110 if tag == "table":
111 self.tables += 1
112 self._table_has_th.append(False)
113 if tag == "th" and self._table_has_th:
114 self._table_has_th[-1] = True
115 for key, value in attributes.items():
116 if key.startswith("on"):
117 self.event_attributes.append(key)
118 if key in {"href", "src", "action", "poster", "data"} and value:
119 if _EXTERNAL_URL_RE.search(value):
120 self.external_urls.append(value[:200])
121 if tag == "script" and key == "src" and value:
122 self.external_script_or_link = True
123 if tag == "link" and key == "href" and value:
124 self.external_script_or_link = True
125
126 def handle_endtag(self, tag: str) -> None:
127 if tag.lower() == "table" and self._table_has_th:
128 if not self._table_has_th.pop():
129 self.tables_without_header += 1

149def _motion_duration_ms(source: str) -> list[float]:
150 values: list[float] = []
151 for declaration in _MOTION_DECLARATION_RE.findall(source):
152 for number, unit in re.findall(r"(\d+(?:\.\d+)?)\s*(ms|s)\b", declaration, re.I):
153 value = float(number) * (1000 if unit.lower() == "s" else 1)
154 values.append(value)
155 return values
