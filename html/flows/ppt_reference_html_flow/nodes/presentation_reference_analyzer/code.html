105def parse_json_object(value: Any) -> dict[str, Any]:
106 """코드 펜스나 앞뒤 설명이 섞인 모델 응답에서 첫 JSON object를 읽는다."""
107
108 if isinstance(value, dict):
109 return deepcopy(value)
110 text = _extract_text(value).strip()
111 if not text:
112 raise ValueError("모델 응답이 비어 있습니다.")
113 fenced = re.search(r"```(?:json)?\s*(\{[\s\S]*?\})\s*```", text, re.I)
114 candidates = [fenced.group(1)] if fenced else []
115 candidates.append(text)
116 decoder = json.JSONDecoder()
117 for candidate in candidates:
118 try:
119 parsed = json.loads(candidate)
120 if isinstance(parsed, dict):
121 return parsed
122 except Exception:
123 pass
124 for index, char in enumerate(candidate):
125 if char != "{":
126 continue
127 try:
128 parsed, _ = decoder.raw_decode(candidate[index:])
129 except Exception:
130 continue
131 if isinstance(parsed, dict):
132 return parsed
133 raise ValueError("모델 응답에서 JSON object를 찾지 못했습니다.")

136def _safe_text(value: Any, maximum: int = 500) -> str:
137 text = str(value or "").replace("\x00", " ")
138 text = re.sub(r"<\s*/?\s*(?:script|style)[^>]*>", "", text, flags=re.I)
139 text = re.sub(r"[\r\n\t]+", " ", text)
140 return re.sub(r"\s{2,}", " ", text).strip()[:maximum]

269def _reference_list(request: dict[str, Any], maximum: int) -> list[dict[str, Any]]:
270 reference_images = request.get("reference_images")
271 if not isinstance(reference_images, dict):
272 return []
273 result: list[dict[str, Any]] = []
274 for role in ("cover", "body"):
275 items = reference_images.get(role)
276 if not isinstance(items, list):
277 continue
278 for item in items:
279 if not isinstance(item, dict) or len(result) >= maximum:
280 break
281 data_url = str(item.get("data_url") or "")
282 if not _DATA_URL_RE.fullmatch(data_url):
283 continue
284 result.append(
285 {
286 "reference_id": str(item.get("reference_id") or f"{role}_{len(result) + 1}"),
287 "role": role,
288 "filename": str(item.get("filename") or "")[:200],
289 "mime_type": str(item.get("mime_type") or ""),
290 "data_url": data_url,
291 }
292 )
293 return result
