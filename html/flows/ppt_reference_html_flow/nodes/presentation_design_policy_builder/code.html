30def _payload(value: Any) -> dict[str, Any]:
31 if isinstance(value, dict):
32 return deepcopy(value)
33 data = getattr(value, "data", None)
34 if isinstance(data, dict):
35 return deepcopy(data)
36 text = getattr(value, "text", None) or getattr(value, "content", None)
37 if isinstance(text, str) and text.strip():
38 try:
39 parsed = json.loads(text)
40 return deepcopy(parsed) if isinstance(parsed, dict) else {}
41 except Exception:
