84def parse_json_object(value: Any) -> dict[str, Any]:
85 if isinstance(value, dict):
86 return deepcopy(value)
87 text = _extract_text(value).strip()
88 if not text:
89 raise ValueError("계획 모델 응답이 비어 있습니다.")
90 decoder = json.JSONDecoder()
91 for candidate in (re.sub(r"^```(?:json)?\s*|\s*```$", "", text, flags=re.I), text):
92 try:
93 parsed = json.loads(candidate)
94 if isinstance(parsed, dict):
95 return parsed
96 except Exception:
97 pass
98 for index, char in enumerate(candidate):
99 if char != "{":
100 continue
101 try:
102 parsed, _ = decoder.raw_decode(candidate[index:])
103 except Exception:
104 continue
105 if isinstance(parsed, dict):
106 return parsed
107 raise ValueError("계획 모델 응답에서 JSON object를 찾지 못했습니다.")

110def _safe_text(value: Any, maximum: int = 1_000) -> str:
111 text = str(value or "").replace("\x00", " ")
112 text = re.sub(r"<script\b[^>]*>[\s\S]*?</script\s*>", "", text, flags=re.I)
113 text = re.sub(r"<style\b[^>]*>[\s\S]*?</style\s*>", "", text, flags=re.I)
114 text = re.sub(r"[\r\t]+", " ", text)
115 return text.strip()[:maximum]

181def _content_points(content: str) -> list[str]:
182 points: list[str] = []
183 for part in re.split(r"(?:\r?\n)+|(?<=[.!?。])\s+", str(content or "")):
184 text = re.sub(r"^[\-•*\d.)\s]+", "", part).strip()
185 if text and text not in points:
186 points.append(text[:240])
187 return points[:40]
