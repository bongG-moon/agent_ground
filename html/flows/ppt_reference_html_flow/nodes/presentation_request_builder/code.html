175def _decode_image_value(value: str, mime_hint: str) -> tuple[str, str, int]:
176 """Base64 또는 Data URL을 검증하고 Data URL로 통일한다."""
177
178 text = str(value or "").strip()
179 match = _DATA_URL_RE.fullmatch(text)
180 if match:
181 mime = match.group(1).lower()
182 encoded = re.sub(r"\s+", "", match.group(2))
183 else:
184 mime = mime_hint.lower()
185 encoded = re.sub(r"\s+", "", text)
186 if mime not in _ALLOWED_IMAGE_MIME:
187 raise ValueError(f"허용되지 않은 이미지 형식입니다: {mime or '알 수 없음'}")
188 try:
189 raw = base64.b64decode(encoded, validate=True)
190 except Exception as exc:
191 raise ValueError("이미지 Base64 값이 올바르지 않습니다.") from exc
192 if not raw:
193 raise ValueError("빈 이미지입니다.")
194 return f"data:{mime};base64,{encoded}", mime, len(raw)

210def normalize_reference_images(
211 cover_value: Any,
212 body_value: Any,
213 *,
214 max_images: int = 12,
215 max_total_bytes: int = 16 * 1024 * 1024,
216) -> tuple[dict[str, list[dict[str, Any]]], list[dict[str, str]], list[str]]:
217 """Encoder 결과를 역할별 이미지 참조로 정규화한다.
218
219 오류에는 파일명과 이유만 기록하며 Base64 본문은 절대 포함하지 않는다.
220 """
221
222 result = {"cover": [], "body": []}
223 errors: list[dict[str, str]] = []
224 warnings: list[str] = []
225 total_bytes = 0
226 position = 0
227 for role, value in (("cover", cover_value), ("body", body_value)):
228 for item in _image_items(value):
229 position += 1
230 if position > max_images:
231 errors.append({"code": "too_many_images", "message": f"이미지는 최대 {max_images}개까지 허용됩니다."})
232 break
233 filename = Path(str(item.get("filename") or item.get("name") or f"{role}_{position}")).name[:200]
234 raw_value = str(item.get("value") or item.get("data_url") or item.get("base64") or "")
235 mime_hint = str(item.get("mime_type") or _MIME_FROM_EXTENSION.get(Path(filename).suffix.lower(), ""))
236 try:
237 data_url, mime, byte_size = _decode_image_value(raw_value, mime_hint)
238 except ValueError as exc:
239 errors.append({"code": "invalid_reference_image", "message": f"{filename}: {exc}"})
240 continue
241 total_bytes += byte_size
242 if total_bytes > max_total_bytes:
243 errors.append({"code": "reference_images_too_large", "message": "참고 이미지 전체 용량 제한을 초과했습니다."})
244 break
245 result[role].append(
246 {
247 "reference_id": f"{role}_{len(result[role]) + 1:02d}",
248 "role": role,
249 "position": len(result[role]) + 1,
250 "filename": filename,
251 "mime_type": mime,
252 "byte_size": byte_size,
253 "encoding": "data_url",
254 "value": data_url,
255 "data_url": data_url,
256 "sha256": str(item.get("sha256") or "")[:128],
257 }
258 )
259 if not result["cover"]:
260 warnings.append("유효한 표지 참고 이미지가 없습니다. 기본 표지 스타일을 사용합니다.")
261 if not result["body"]:
262 warnings.append("유효한 본문 참고 이미지가 없습니다. 기본 본문 스타일을 사용합니다.")
263 return result, errors, warnings

286def _semantic_type(name: str, values: list[Any], declared: str = "") -> str:
287 declared_key = str(declared or "").strip().lower()
288 aliases = {
289 "number": "quantitative",
290 "numeric": "quantitative",
291 "quantitative": "quantitative",
292 "date": "temporal",
293 "datetime": "temporal",
294 "time": "temporal",
295 "temporal": "temporal",
296 "category": "nominal",
297 "categorical": "nominal",
298 "dimension": "nominal",
299 "nominal": "nominal",
300 "text": "text",
301 "ordinal": "ordinal",
302 "identifier": "identifier",
303 }
304 if declared_key in aliases:
305 return aliases[declared_key]
306 sample = _non_empty(values)[:100]
307 if not sample:
308 return "text"
309 if sum(_is_number(value) for value in sample) / len(sample) >= 0.9:
310 return "quantitative"
311 temporal_count = sum(
312 isinstance(value, (date, datetime)) or bool(_TEMPORAL_RE.fullmatch(str(value).strip()))
313 for value in sample
314 )
315 if temporal_count / len(sample) >= 0.8:
316 return "temporal"
317 unique_ratio = len({str(value) for value in sample}) / len(sample)
318 average_length = sum(len(str(value)) for value in sample) / len(sample)
319 normalized_name = re.sub(r"[^a-z0-9]", "", str(name).lower())
320 if unique_ratio >= 0.9 and any(token in normalized_name for token in ("id", "identifier", "code", "번호", "코드")):
321 return "identifier"
322 if unique_ratio <= 0.6 and average_length <= 80:
323 return "nominal"
324 return "text"

327def _recommend_visual(columns: list[dict[str, Any]], rows: list[dict[str, Any]], preferred: str) -> list[str]:
328 allowed = {"auto", "table", "kpi", "bar", "line", "scatter", "histogram", "stacked_bar"}
329 requested = preferred if preferred in allowed else "auto"
330 if requested != "auto":
331 return [requested, "table"] if requested != "table" else ["table"]
332 numeric = [column["name"] for column in columns if column["semantic_type"] == "quantitative"]
333 temporal = [column["name"] for column in columns if column["semantic_type"] == "temporal"]
334 dimensions = [column["name"] for column in columns if column["semantic_type"] in {"nominal", "ordinal"}]
335 if temporal and numeric:
336 return ["line", "table"]
337 if dimensions and numeric:
338 categories = len({str(row.get(dimensions[0])) for row in rows})
339 if 1 < categories <= 12:
340 return ["bar", "table"]
341 if len(numeric) >= 2 and len(rows) >= 8:
342 return ["scatter", "table"]
343 if len(numeric) == 1 and len(rows) == 1:
344 return ["kpi", "table"]
345 return ["table"]

439def _datasets_from_files(files: Any, errors: list[dict[str, str]]) -> list[dict[str, Any]]:
440 items = files if isinstance(files, (list, tuple)) else ([] if not files else [files])
441 candidates: list[dict[str, Any]] = []
442 for item in items:
443 try:
444 filename, content = _read_file_bytes(item)
445 if len(content) > _MAX_DATASET_FILE_BYTES:
446 raise ValueError("파일 크기가 5MB를 초과합니다.")
447 extension = Path(filename).suffix.lower()
448 text = content.decode("utf-8-sig")
449 if extension == ".csv":
450 rows = list(csv.DictReader(io.StringIO(text)))
451 candidates.append({"dataset_id": Path(filename).stem, "title": Path(filename).stem, "rows": rows})
452 elif extension == ".json":
453 parsed = json.loads(text)
454 candidates.extend(_dataset_candidates(parsed))
455 else:
456 raise ValueError("CSV 또는 JSON 파일만 사용할 수 있습니다.")
457 except Exception as exc:
458 safe_name = Path(str(getattr(item, "name", None) or item or "dataset")).name[:200]
459 errors.append({"code": "invalid_dataset_file", "message": f"{safe_name}: {exc}"})
460 return candidates

463def normalize_datasets(value: Any, files: Any = None) -> tuple[list[dict[str, Any]], list[dict[str, str]], list[str]]:
464 """JSON/CSV 데이터셋을 행·컬럼 계약과 시각화 권고로 정리한다."""
465
466 errors: list[dict[str, str]] = []
467 warnings: list[str] = []
468 candidates = _dataset_candidates(value) + _datasets_from_files(files, errors)
469 if len(candidates) > _MAX_DATASETS:
470 warnings.append(f"데이터셋은 최대 {_MAX_DATASETS}개까지만 처리했습니다.")
471 datasets: list[dict[str, Any]] = []
472 seen_ids: set[str] = set()
473 for index, raw in enumerate(candidates[:_MAX_DATASETS], 1):
474 dataset = _dataset_from_rows(raw, index, warnings)
475 if dataset is None:
476 continue
477 base_id = dataset["dataset_id"]
478 suffix = 2
479 while dataset["dataset_id"] in seen_ids:
480 dataset["dataset_id"] = f"{base_id}_{suffix}"
481 suffix += 1
482 seen_ids.add(dataset["dataset_id"])
483 datasets.append(dataset)
484 return datasets, errors, warnings

487def build_presentation_request(
488 presentation_request: Any,
489 cover_images: Any,
490 body_images: Any,
491 *,
492 brief: str = "",
493 content: str = "",
494 datasets_json: str = "",
495 dataset_files: Any = None,
496 target_slide_count: Any = 10,
497 user_request: str = "",
498) -> dict[str, Any]:
499 """모든 입력을 후속 Node가 공유하는 `presentation_request`로 합친다."""
500
501 source = _payload(presentation_request)
502 source_dict = source if isinstance(source, dict) else {}
503 brief_source: Any = source_dict.get("brief") if isinstance(source_dict.get("brief"), dict) else brief or source
504 count = _safe_int(target_slide_count, 10, 3, 30)
505 normalized_brief, brief_warnings = _normalize_brief(
506 brief_source,
507 content or str(source_dict.get("content") or ""),
508 count,
509 user_request or str(source_dict.get("user_request") or ""),
510 )
511 dataset_source = source_dict.get("datasets") if "datasets" in source_dict else datasets_json
512 datasets, dataset_errors, dataset_warnings = normalize_datasets(dataset_source, dataset_files)
513 references, image_errors, image_warnings = normalize_reference_images(cover_images, body_images)
514 errors = image_errors + dataset_errors
515 warnings = brief_warnings + dataset_warnings + image_warnings
516 request_contract = {
517 "brief": normalized_brief,
518 "user_request": normalized_brief.get("user_request", ""),
519 "template_mode": str(source_dict.get("template_mode") or "analyze_and_rebuild"),
520 "datasets": datasets,
521 "reference_images": references,
