49def _normalize_triple_quoted_json(text: str) -> str:
50 """triple quote로 감싼 Oracle TNS 값을 JSON 문자열로 바꿉니다."""
51 def replace(match: re.Match[str]) -> str:
52 # match.group(2)는 따옴표 안쪽 실제 문자열입니다.
53 # json.dumps를 사용하면 줄바꿈/따옴표가 JSON에서 안전한 형태로 escape됩니다.

57 return re.sub(r'("""|\'\'\')(.*?)(\1)', replace, str(text or ""), flags=re.DOTALL)

66def _parse_named_tns_blocks(text: str) -> Dict[str, Any]:
67 """Text Input의 'DB_KEY: TNS' 블록을 Oracle 설정 dict로 바꿉니다."""
68 # 예:
69 # PKG_RPT:
70 # (DESCRIPTION=...)
71 # PKG_PLAN:
72 # (DESCRIPTION=...)
73 configs: Dict[str, Any] = {}
74 current_key = ""
75 current_lines: list[str] = []
76
77 def save_current() -> None:
78 nonlocal current_key, current_lines
79 tns = "\n".join(current_lines).strip()
80 if current_key and _looks_like_tns(tns):
81 configs[current_key] = {"tns": tns}
82 current_key = ""
83 current_lines = []
84
85 for raw_line in str(text or "").splitlines():
86 line = raw_line.strip()
87 # PKG_RPT:, PKG_PLAN:처럼 TNS 블록 앞의 DB key 라인을 찾습니다.
88 key_match = re.match(r"^([A-Za-z][A-Za-z0-9_-]*)\s*:\s*(.*)$", line)
89 if key_match and not line.startswith("("):
90 possible_tns = key_match.group(2).strip()
91 # TNS 안쪽 줄에는 ADDRESS= 같은 key가 나오지만, DB_KEY: 형태는 보통 맨 앞 식별자입니다.
92 save_current()
93 current_key = key_match.group(1).strip()
94 if possible_tns:
95 current_lines.append(possible_tns)
96 continue
97 if current_key:
98 current_lines.append(raw_line)
99
100 save_current()
101 return configs

104def parse_jsonish(value: Any) -> tuple[Any, list[str]]:
105 """JSON 설정이나 Text Input의 raw TNS 문자열을 Oracle 설정 dict로 변환합니다."""
106 # 1) 이미 dict/list로 들어온 값은 Langflow Data 연결이므로 그대로 복사해서 사용합니다.
107 if value is None:
108 return {}, []
109 if isinstance(value, (dict, list)):
110 return deepcopy(value), []
111
112 # 2) 일반 JSON 문자열과 Python literal 문자열을 차례로 시도합니다.
113 text = str(value or "").strip()
114 if not text:
115 return {}, []
116 errors: list[str] = []
117 for parser in (json.loads, ast.literal_eval):
118 try:
119 return parser(text), []
120 except Exception as exc:
121 errors.append(str(exc))
122
123 # 3) 기존 JSON 설정이 여러 줄 triple quote를 포함할 수 있어 JSON 문자열로 치환한 뒤 다시 시도합니다.
124 normalized = _normalize_triple_quoted_json(text)
125 if normalized != text:
126 for parser in (json.loads, ast.literal_eval):
127 try:
128 return parser(normalized), []
129 except Exception as exc:
130 errors.append(str(exc))
131
132 named_tns = _parse_named_tns_blocks(text)
133 if named_tns:
134 return named_tns, []
135
136 if _looks_like_tns(text):
137 # Text Input에는 JSON wrapper 없이 TNS 문자열만 넣을 수 있게 합니다.
138 # 실제 접속 시 db_key가 다르게 들어오면 OracleConnector가 단일 설정 fallback으로 이 값을 사용합니다.
139 return {DEFAULT_ORACLE_DB_KEY: {"tns": text}}, []
140 return {}, errors

279def _sql_literal(value: Any) -> str:
280 """Python 값을 SQL에 직접 넣을 수 있는 literal 문자열로 변환합니다."""
281 if value is None:
282 return "NULL"
283 if isinstance(value, (datetime, date)):
284 # 날짜 객체는 현장 쿼리에서 자주 쓰는 YYYYMMDD 문자열로 감쌉니다.
285 return f"'{value.strftime('%Y%m%d')}'"
286 if isinstance(value, Decimal):
287 return str(value)
288 if isinstance(value, (int, float)) and not isinstance(value, bool):
289 # 숫자는 따옴표 없이 그대로 넣습니다.
290 return str(value)
291 # 문자열은 작은따옴표를 escape해서 SQL literal로 만듭니다.
292 text = str(value).replace("'", "''")
293 return f"'{text}'"

296def _render_sql_template(template: str, params: Dict[str, Any]) -> tuple[str, list[str]]:
297 """{DATE} 같은 템플릿 변수를 SQL literal로 치환하고 누락 변수를 함께 반환합니다."""
298 missing: list[str] = []
299
300 def replace(match: re.Match[str]) -> str:
301 # {DATE}에서 DATE 부분만 꺼내 params에서 값을 찾습니다.
302 key = match.group(1).strip()
303 value = _param_value(params, key)
304 if value in (None, "", []):
305 # 값이 없으면 원래 placeholder를 남겨두고 missing 목록에 기록합니다.
306 missing.append(key)
307 return match.group(0)
308 # 값이 있으면 SQL literal로 바꿔 안전하게 삽입합니다.
309 return _sql_literal(value)
310
311 # 중괄호 placeholder를 모두 찾아 replace 함수로 치환합니다.
312 return re.sub(r"\{([^{}]+)\}", replace, str(template or "")), missing

423 def get_connection(self, db_key: str) -> Any:

428 resolved = next(iter(self.config))
429 if not resolved:
430 raise ValueError(f"Unknown Oracle config key: {db_key}")

432 # 2) 다양한 현장 표기(user/username/id, dsn/tns 등)를 받아 하나의 접속값으로 정리합니다.
433 config = self.config[resolved]

441 # TNS만 입력한 환경에서는 외부 인증/지갑 설정을 사용하는 경우가 있어 dsn만 전달합니다.

501def retrieve_simple_oracle_data(data_request_value: Any, oracle_config_value: Any = "", fetch_limit_value: Any = "5000") -> Dict[str, Any]:
502 """전체 data_request 중 source_type이 oracle인 요청만 골라 실행합니다."""
503 # 1) 이전 노드가 넘긴 data_request와 fetch_limit 입력을 실행 가능한 값으로 정리합니다.
504 request_body = _request_body(data_request_value)
505 try:
506 fetch_limit = max(1, int(fetch_limit_value or 5000))
507 except Exception:
508 fetch_limit = 5000
509
510 # 2) multi request일 수 있으므로 oracle 요청만 필터링하고 다른 source 요청은 건너뜁니다.
511 oracle_config, oracle_errors = parse_jsonish(oracle_config_value)
512 oracle_config = oracle_config if isinstance(oracle_config, dict) else {}
513 source_requests = []
514 for item in _request_items(request_body):
515 source_type = _source_type(item.get("source_type") or item.get("source") or _source_config(item).get("source_type"))
516 if source_type == "oracle":
517 source_requests.append(item)
518 if not source_requests:
519 return {"skipped": True, "source_type": "oracle", "skip_reason": "No oracle request.", "items": []}
520
521 # 3) 설정 파싱 실패가 있으면 각 oracle 요청을 동일한 실패 결과로 반환합니다.

525 items.append(_error_result(item, "Oracle config parse failed: " + "; ".join(oracle_errors), "config_parse_failed"))
526 else:
527 items = []
528 for item in source_requests:
529 items.append(_run_oracle(item, oracle_config, fetch_limit))
530 return {"source_type": "oracle", "items": items}
