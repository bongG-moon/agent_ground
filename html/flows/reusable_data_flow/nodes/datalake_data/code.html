62def _payload_from_value(value: Any) -> Dict[str, Any]:
63 """Data/Message/Text/JSON 입력을 내부 처리용 dict로 맞춥니다."""
64 # 입력이 비어 있으면 뒤쪽 필터링 단계에서 skipped 처리되도록 빈 dict를 반환합니다.
65 if value is None:
66 return {}
67
68 # 이미 dict로 들어온 payload는 원본 변경을 피하기 위해 복사합니다.
69 if isinstance(value, dict):
70 return deepcopy(value)
71
72 # Langflow Data 객체는 .data에 실제 payload가 들어옵니다.
73 data = getattr(value, "data", None)
74 if isinstance(data, dict):
75 return deepcopy(data)
76
77 # Message/Text 객체는 text/content를 JSON으로 파싱합니다.
78 text = getattr(value, "text", None) or getattr(value, "content", None)
79 if isinstance(text, str):
80 try:
81 parsed = json.loads(text)
82 return parsed if isinstance(parsed, dict) else {"text": text}
83 except Exception:
84 # JSON이 아니면 원문을 보존해 진단할 수 있게 합니다.
85 return {"text": text}
86 return {}

193def _sql_literal(value: Any) -> str:
194 """Python 값을 SQL에 직접 넣을 수 있는 literal 문자열로 변환합니다."""
195 if value is None:
196 return "NULL"
197 if isinstance(value, (datetime, date)):
198 # 날짜 객체는 YYYYMMDD 문자열로 감싸 현장 SQL에서 바로 쓰기 쉽게 합니다.
199 return f"'{value.strftime('%Y%m%d')}'"
200 if isinstance(value, Decimal):
201 return str(value)
202 if isinstance(value, (int, float)) and not isinstance(value, bool):
203 return str(value)
204 # 문자열은 작은따옴표를 escape해서 SQL literal로 만듭니다.
205 text = str(value).replace("'", "''")
206 return f"'{text}'"

209def _render_template(template: str, params: Dict[str, Any]) -> tuple[str, list[str]]:
210 """{YM} 같은 템플릿 변수를 SQL literal로 치환하고 누락 변수를 반환합니다."""
211 missing: list[str] = []
212
213 def replace(match: re.Match[str]) -> str:
214 # {YM}에서 YM만 꺼내 params에서 값을 찾습니다.
215 key = match.group(1).strip()
216 value = _param_value(params, key)
217 if value in (None, "", []):
218 # 값이 없으면 placeholder는 유지하고 missing 목록에 기록합니다.
219 missing.append(key)
220 return match.group(0)
221 # 값이 있으면 SQL literal로 바꿔 템플릿에 넣습니다.
222 return _sql_literal(value)
223
224 # 중괄호 placeholder 전체를 replace 함수로 치환합니다.
225 return re.sub(r"\{([^{}]+)\}", replace, str(template or "")), missing

228def _json_ready(value: Any) -> Any:
229 """Datalake 결과에 섞인 날짜/Decimal/NaN 등을 JSON 안전값으로 바꿉니다."""
230 if value is None or isinstance(value, (str, int, float, bool)):
231 return value
232 if isinstance(value, (datetime, date)):
233 return value.isoformat()
234 if isinstance(value, Decimal):
235 return float(value)
236 if isinstance(value, dict):
237 # dict key는 문자열로 맞추고 value는 재귀적으로 정리합니다.
238 return {str(key): _json_ready(item) for key, item in value.items()}
239 if isinstance(value, (list, tuple, set)):
240 return [_json_ready(item) for item in value]
241 try:
242 # pandas/numpy NaN은 자기 자신과 같지 않으므로 None으로 바꿉니다.
243 if value != value:
244 return None
245 except Exception:
246 pass
247 return str(value)

250def _frame_to_rows(frame: Any) -> list[Dict[str, Any]]:
251 """Spark/Pandas/list 결과를 JSON으로 직렬화 가능한 row 목록으로 바꿉니다."""
252 # 예전 LakeHouse 경로는 Spark DataFrame, 새 SmallData 경로는 pandas DataFrame으로 올 수 있습니다.
253 # 테스트나 다른 구현에서는 이미 pandas/list로 들어올 수 있어 가능한 형태를 순서대로 처리합니다.
254 if hasattr(frame, "toPandas"):
255 frame = frame.toPandas()
256 if hasattr(frame, "to_dict"):
257 try:
258 # pandas DataFrame은 orient="records"가 가장 명확합니다.
259 rows = frame.to_dict(orient="records")
260 except TypeError:
261 # 일부 DataFrame 유사 객체는 positional 인자만 받는 경우가 있어 보조로 처리합니다.
262 rows = frame.to_dict("records")
263 elif isinstance(frame, list):
264 # 이미 list[dict]로 반환하는 테스트 구현이나 API 래퍼도 허용합니다.
265 rows = frame
266 else:
267 rows = []
268 # datetime/Decimal/NaN 등 JSON 직렬화에 걸릴 수 있는 값을 안전한 값으로 바꿉니다.
269 return [_json_ready(dict(row)) for row in rows if isinstance(row, dict)]

308async def _get_running_cluster_endpoint(
309 user_id: str,
310 token: str,
311 api_base_url: str = DATALAKE_API_BASE_URL,
312 cluster_type: str = DATALAKE_CLUSTER_TYPE,
313 max_attempts: int = DATALAKE_CLUSTER_MAX_ATTEMPTS,
314 wait_seconds: int = DATALAKE_CLUSTER_WAIT_SECONDS,
315) -> tuple[str, int]:
316 """Datalake API에서 실행 중인 StarRocks cluster의 jdbc endpoint를 가져옵니다."""
317 aiohttp, _, _ = _load_small_data_runtime_modules()
318 headers = {
319 "accept": "application/json;charset=UTF-8",
320 "Authorization": f"Bearer {token}",
321 "user_id": str(user_id or ""),
322 }
323 target_url = f"runtime/cluster/{cluster_type}/running"
324
325 async with aiohttp.ClientSession(base_url=api_base_url) as session:
326 last_error = ""
327 for attempt in range(1, max_attempts + 1):
328 try:
329 async with session.get(target_url, headers=headers) as response:
330 payload = await response.json()
331 if payload.get("status") == "RUNNING":
332 endpoints = payload.get("endpoints") if isinstance(payload.get("endpoints"), dict) else {}
333 return _endpoint_host_port(endpoints.get("jdbc-external"))
334 last_error = f"cluster status={payload.get('status')}"
335 logger.info("Datalake cluster is not RUNNING yet. attempt=%s status=%s", attempt, payload.get("status"))
336 except aiohttp.ClientError as exc:
337 last_error = str(exc)
338 logger.warning("Datalake cluster status request failed. attempt=%s error=%s", attempt, exc)
339 if attempt < max_attempts:
340 await asyncio.sleep(wait_seconds)
341 raise RuntimeError(f"Datalake cluster is not RUNNING after {max_attempts} attempts. last_error={last_error}")

460def _run_datalake(request: Dict[str, Any], credentials: Dict[str, str], fetch_limit: int) -> Dict[str, Any]:
461 """단일 Datalake 요청을 검증하고, 더미 또는 실제 SmallData 결과를 표준 결과로 바꿉니다."""
462 # 1) SQL 템플릿에 들어갈 params를 모으고 required_params 누락을 먼저 확인합니다.
463 # 필수값이 없으면 Datalake를 호출하기 전에 명확한 오류를 돌려줍니다.
464 params = _params(request)
465 missing = _missing_required_params(params, _as_list(request.get("required_params")))
466 if missing:
467 return _error_result(request, f"Missing required parameter(s): {', '.join(missing)}", "missing_required_params")
468
469 # 2) source_config에서 query_template을 가져오고 {YM} 같은 placeholder를 params 값으로 치환합니다.
470 # 치환 후 SQL은 source_results에 기록해 실행 내용을 확인하기 쉽게 둡니다.
471 config = _source_config(request)
472 query_template = str(config.get("query_template") or "").strip()
473 if not query_template:
474 return _error_result(request, "Datalake source_config must include query_template.", "missing_query_template")

478

482 rows = dummy_builder(request, params, sql)[:fetch_limit]

485 # 4) 실제 Datalake 실행부입니다. 더미 블록을 주석 처리한 뒤 아래 블록을 사용합니다.
486 missing_creds = []

489 missing_creds.append(key)
490 if missing_creds:

493 rows = _run_small_data_sql(sql, credentials, config, fetch_limit)
494 return _result(request, rows, {"executed_query": sql, "cluster_type": str(config.get("cluster_type") or DATALAKE_CLUSTER_TYPE)})
