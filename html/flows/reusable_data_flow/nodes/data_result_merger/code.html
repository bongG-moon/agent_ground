20def _payload_from_value(value: Any) -> Dict[str, Any]:
21 """Data node 출력이 Data/JSON/Text 중 무엇이든 내부 처리용 dict로 맞춥니다."""
22 # 연결이 비어 있으면 처리할 source가 없는 것으로 봅니다.
23 if value is None:
24 return {}
25
26 # 이미 dict면 원본을 건드리지 않도록 복사합니다.
27 if isinstance(value, dict):
28 return deepcopy(value)
29
30 # Langflow Data 객체는 .data에 payload를 담습니다.
31 data = getattr(value, "data", None)
32 if isinstance(data, dict):
33 return deepcopy(data)
34
35 # Message/Text로 들어온 JSON도 최대한 dict로 해석합니다.
36 text = getattr(value, "text", None) or getattr(value, "content", None)
37 if isinstance(text, str):
38 try:
39 parsed = json.loads(text)
40 return parsed if isinstance(parsed, dict) else {"text": text}
41 except Exception:
42 # JSON이 아니면 원문만 남기고 실제 병합에서는 payload 없음처럼 처리됩니다.
43 return {"text": text}
44 return {}

47def _source_payload(value: Any) -> Dict[str, Any]:
48 """source_result wrapper를 벗겨 data node가 만든 payload만 반환합니다."""
49 # 각 data node 출력이 source_result로 한 번 감싸져 있어도 같은 방식으로 읽을 수 있게 합니다.
50 payload = _payload_from_value(value)
51 if isinstance(payload.get("source_result"), dict):
52 return deepcopy(payload["source_result"])
53 return payload

56def _columns_from_rows(rows: list[Dict[str, Any]]) -> list[str]:
57 """여러 source row를 합친 뒤 표시할 컬럼 순서를 만듭니다."""
58 # row마다 컬럼이 다를 수 있으므로 처음 등장한 key 순서를 유지해 전체 컬럼 목록을 만듭니다.
59 columns: list[str] = []
60 for row in rows:
61 for key in row:
62 text = str(key)
63 if text not in columns:
64 columns.append(text)
65 return columns

68def _source_group(item: Dict[str, Any]) -> Dict[str, Any]:
69 """같은 source_type 요청이 여러 개일 때도 요청별 결과 묶음을 보존합니다."""
70 # merger 내부 data_result는 기존 연결 호환을 위해 전체 row를 합쳐 두고,
71 # 최종 API 출력은 Data Output Builder가 source_results 기준으로 다시 나눕니다.
72 source_rows = item.get("data_result")
73 if not isinstance(source_rows, list):
74 source_rows = item.get("rows", [])
75 rows = [deepcopy(row) for row in source_rows if isinstance(row, dict)]
76
77 group = {
78 "name": item.get("name", ""),
79 "source_type": item.get("source_type", ""),
80 "success": bool(item.get("success")),
81 "row_count": int(item.get("row_count", len(rows)) or 0),
82 "columns": item.get("columns") if isinstance(item.get("columns"), list) else _columns_from_rows(rows),
83 "data_result": rows,
84 "error_message": item.get("error_message", ""),
85 }
86 if item.get("failure_type") not in (None, "", [], {}):
87 group["failure_type"] = item.get("failure_type")
88
89 # data node가 request_params/request_label을 제공하는 경우 그대로 노출합니다.
90 # 같은 Oracle source를 날짜/공정만 바꿔 여러 번 조회할 때 구분용으로 유용합니다.
91 if isinstance(item.get("request_params"), dict):
92 group["request_params"] = deepcopy(item["request_params"])
93 if item.get("request_label") not in (None, "", [], {}):
94 group["request_label"] = item.get("request_label")
95
96 # 실제 데이터 row에는 넣지 않지만 source별 실행 확인에 필요한 값은 source_results에 보존합니다.
97 for key in ("db_key", "executed_query", "request_body", "doc_id"):
98 if item.get(key) not in (None, "", [], {}):
99 group[key] = deepcopy(item[key])
100 return group

103def merge_simple_data_results(
104 oracle_result_value: Any = None,
105 h_api_result_value: Any = None,
106 datalake_result_value: Any = None,
107 goodocs_result_value: Any = None,
108) -> Dict[str, Any]:
109 """Oracle/H-API/Datalake/Goodocs 결과를 한 개의 data_result로 합칩니다."""
110 # 각 전용 data node는 같은 data_request를 받아도 자기 source_type이 없으면 skipped=True를 냅니다.
111 # merger는 네 data node 출력을 한 번에 받아 실제 처리된 source만 최종 결과에 포함합니다.
112 source_values = [oracle_result_value, h_api_result_value, datalake_result_value, goodocs_result_value]
113 source_items: list[Dict[str, Any]] = []
114 source_groups: list[Dict[str, Any]] = []
115 combined_rows: list[Dict[str, Any]] = []
116
117 for value in source_values:
118 # Langflow Data wrapper 또는 source_result wrapper를 벗겨 data node payload만 남깁니다.
119 payload = _source_payload(value)
120 if not payload:
121 continue
122 if payload.get("skipped"):
123 # 다른 타입 data node가 "내 요청이 아님"이라고 알린 경우라서 최종 payload에는 싣지 않습니다.
124 continue
125
126 # 현재 data node 표준은 items이지만, 이전 테스트 payload와의 호환을 위해 results도 허용합니다.
127 items = payload.get("items")
128 if not isinstance(items, list):
129 items = payload.get("results") if isinstance(payload.get("results"), list) else []
130 for item in items:
131 if not isinstance(item, dict):
132 continue
133 # 전체 성공 여부를 판단하기 위해 data node item을 보관합니다.
134 source_items.append(deepcopy(item))
135 # source_results에는 요청별 상세 결과를 보존합니다.
136 source_groups.append(_source_group(item))
137 if item.get("success"):
138 # 성공한 source의 data_result만 합쳐 최종 표 데이터로 만듭니다.
139 # 과거 rows 필드로 오는 payload도 data_result로 취급합니다.
140 source_rows = item.get("data_result")
141 if not isinstance(source_rows, list):
142 source_rows = item.get("rows", [])
143 for row in source_rows:
144 if isinstance(row, dict):
145 combined_rows.append(deepcopy(row))
146
147 # 하나라도 실패 source가 있으면 전체 success는 false로 둡니다.
148 # source가 한 개면 single, 여러 개면 multi로 표시해 Chat Output에서 흐름을 빠르게 볼 수 있게 합니다.
149 has_failure = any(not bool(item.get("success")) for item in source_items)
150 success = bool(source_items) and not has_failure and any(bool(item.get("success")) for item in source_items)
151 data_result = {
152 "success": success,
153 "mode": "single" if len(source_items) == 1 else "multi",
154 "data_result": combined_rows,
155 "source_results": source_groups,
156 }
157 if not source_items:
158 # 어떤 data node도 처리하지 않았다면 flow 연결 또는 source_type 매칭 문제일 가능성이 큽니다.
159 data_result["mode"] = "empty"
160 data_result["source_results"] = [
161 {
162 "name": "",
163 "source_type": "",
164 "success": False,
165 "row_count": 0,
166 "columns": [],
167 "data_result": [],
168 "error_message": "No source data node produced data.",
169 "failure_type": "no_source_result",
170 }
171 ]
172 return {"data_result": data_result}

190 def build_data_result(self):
191 """각 source_result를 병합하고 상태 표시용 row/source 개수를 기록합니다."""
192 # 네 data node 출력 포트를 모두 받아 하나의 병합 함수에 넘깁니다.
193 payload = merge_simple_data_results(
194 getattr(self, "oracle_result", None),
195 getattr(self, "h_api_result", None),
196 getattr(self, "datalake_result", None),
197 getattr(self, "goodocs_result", None),
198 )
199 result = payload.get("data_result", {})
200
201 # Langflow 노드 status에는 병합 성공 여부와 규모만 작게 표시합니다.
202 self.status = {
203 "success": bool(result.get("success")),
204 "mode": result.get("mode", ""),
205 "source_count": len(result.get("source_results", [])) if isinstance(result.get("source_results"), list) else 0,
206 "row_count": len(result.get("data_result", [])) if isinstance(result.get("data_result"), list) else 0,
207 }
