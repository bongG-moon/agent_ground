19def _text_from_value(value: Any) -> str:
20 """Prompt Template, Message, Data, dict 어디에서 오든 실제 prompt 문자열을 꺼냅니다."""
21 if value is None:
22 return ""
23 if isinstance(value, str):
24 return value
25 if isinstance(value, dict):
26 for key in ("prompt", "text", "content", "message", "output", "result", "value", "output_text"):
27 item = value.get(key)
28 if isinstance(item, str) and item.strip():
29 return item
30 if isinstance(item, dict):
31 nested = _text_from_value(item)
32 if nested:
33 return nested
34 for attr in ("text", "content", "message"):
35 item = getattr(value, attr, None)
36 if isinstance(item, str) and item.strip():
37 return item
38 data = getattr(value, "data", None)
39 if isinstance(data, dict):
40 return _text_from_value(data)
41 return ""

57def call_reusable_llm(prompt_value: Any, llm_api_key: str = "", model_name: str = "", temperature: Any = "0") -> Dict[str, Any]:
58 """Prompt Template 결과를 LLM에 전달하고 Normalizer들이 읽기 쉬운 llm_result로 반환합니다."""
59 prompt = _text_from_value(prompt_value)
60 errors: list[str] = []
61 llm_text = ""
62
63 if str(llm_api_key or "").strip():
64 try:
65 selected_model = str(model_name or "").strip()
66 if not selected_model:
67 errors.append("Model Name is required when LLM API Key is set.")
68 else:
69 temp = float(temperature or 0)
70 llm = _load_llm(str(llm_api_key).strip(), selected_model, temp)
71 response = llm.invoke(prompt)
72 llm_text = str(getattr(response, "content", response))
73 except Exception as exc:
74 errors.append(str(exc))
75
76 return {
77 "llm_result": {
78 "llm_text": llm_text,
79 "errors": errors,
80 "prompt": prompt,
81 "model_name": str(model_name or "").strip(),
82 }
83 }

102 def build_result(self):
103 """입력 prompt를 LLM에 호출하고 결과 payload를 Data로 내보냅니다."""
104 payload = call_reusable_llm(
105 getattr(self, "prompt", None),
106 getattr(self, "llm_api_key", ""),
107 getattr(self, "model_name", ""),
108 getattr(self, "temperature", "0"),

110 llm_result = payload.get("llm_result", {})
111 self.status = {
