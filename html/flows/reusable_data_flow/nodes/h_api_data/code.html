22def _payload_from_value(value: Any) -> Dict[str, Any]:
23 """Data/Message/Text/JSON 입력을 내부 처리용 dict로 맞춥니다."""
24 # 입력이 비어 있으면 뒤쪽 로직이 안전하게 skipped 처리하도록 빈 dict를 반환합니다.
25 if value is None:
26 return {}
27
28 # dict는 이미 payload이므로 원본을 변경하지 않게 복사합니다.

30 return deepcopy(value)

32 # Langflow Data 객체는 .data에 실제 payload가 들어오는 경우가 많습니다.

38 text = getattr(value, "text", None) or getattr(value, "content", None)

40 try:
41 parsed = json.loads(text)

43 except Exception:

174def _extract_path(payload: Any, path: str) -> Any:
175 """점(.)으로 구분한 response_path를 따라 API 응답 내부 값을 꺼냅니다."""
176 current = payload
177 for token in [part.strip() for part in str(path or "").split(".") if part.strip()]:
178 if isinstance(current, dict):
179 # dict에서는 token 이름의 key를 따라갑니다.
180 current = current.get(token)
181 elif isinstance(current, list) and token.isdigit():
182 # list에서는 숫자 token만 index로 허용합니다.
183 index = int(token)
184 current = current[index] if 0 <= index < len(current) else None
185 else:
186 # 중간 값이 dict/list가 아니면 더 이상 경로를 따라갈 수 없습니다.
187 return None
188 return current

191def _rows_from_api_payload(payload: Any, response_path: str = "") -> list[Dict[str, Any]]:
192 """API 응답 JSON에서 response_path 또는 흔한 data/rows/items 필드를 row 목록으로 변환합니다."""
193 # response_path가 지정되어 있으면 먼저 그 경로를 따라 실제 데이터 배열을 찾아봅니다.
194 # 예: response_path="data.row"이면 payload["data"]["row"]를 우선 사용합니다.
195 original_payload = payload
196 if response_path:
197 payload = _extract_path(payload, response_path)
198 if payload in (None, [], {}):
199 # 경로가 잘못되었거나 비어 있으면 전체 응답에서 흔한 배열 필드를 다시 찾습니다.
200 # 테스트 중 response_path를 잘못 넣어도 응답 전체를 잃지 않기 위한 방어 코드입니다.
201 payload = original_payload
202 if isinstance(payload, list):
203 # API가 곧바로 list를 반환하는 경우는 이 list 자체를 row 후보로 봅니다.
204 rows = payload
205 elif isinstance(payload, dict):
206 # 대부분의 API는 data/rows/items/result/results 중 하나에 실제 row 배열을 담습니다.
207 # 사용자가 response_path를 생략해도 이 흔한 이름들을 자동으로 확인합니다.
208 rows = payload.get("data") or payload.get("rows") or payload.get("items") or payload.get("result") or payload.get("results") or []
209 if isinstance(rows, dict):
210 # row 한 건만 object로 온 경우도 표 형태로 다루기 위해 list로 감쌉니다.
211 rows = [rows]
212 elif not rows:
213 # 위 표준 key에 없더라도 dict 내부 값 중 list[dict] 모양이 있으면 그것을 데이터로 사용합니다.
214 for value in payload.values():
215 if isinstance(value, list) and any(isinstance(item, dict) for item in value):
216 rows = value
217 break
218 if not rows:
219 # 어떤 데이터 배열도 못 찾았지만 응답 자체가 dict이면, 응답 전체를 한 row로 보여줍니다.
220 # 이렇게 해야 "응답은 있는데 빈 list"로 사라지는 문제를 줄일 수 있습니다.
221 rows = [payload]
222 else:
223 rows = []
224 result: list[Dict[str, Any]] = []
225 for row in rows:
226 if isinstance(row, dict):
227 # 표준 결과는 list[dict]입니다.
228 result.append(dict(row))
229 elif row not in (None, ""):
230 # scalar 값도 버리지 않고 value 컬럼 하나짜리 row로 바꿉니다.
231 result.append({"value": row})
232 return result

246def _json_ready(value: Any) -> Any:
247 """API 응답 값을 JSON 직렬화 가능한 기본 타입으로 정리합니다."""
248 if value is None or isinstance(value, (str, int, float, bool)):
249 return value
250 if isinstance(value, dict):
251 # dict key는 문자열로 맞추고 value는 재귀적으로 정리합니다.
252 return {str(key): _json_ready(item) for key, item in value.items()}
253 if isinstance(value, (list, tuple, set)):
254 return [_json_ready(item) for item in value]

369def retrieve_simple_h_api_data(data_request_value: Any, h_api_token: str = "", fetch_limit_value: Any = "5000") -> Dict[str, Any]:
370 """전체 data_request 중 source_type이 h_api인 요청만 골라 실행합니다."""
371 # 1) 앞 노드가 넘긴 data_request wrapper를 벗기고 fetch_limit을 숫자로 정리합니다.
372 request_body = _request_body(data_request_value)
373 try:
374 fetch_limit = max(1, int(fetch_limit_value or 5000))
375 except Exception:
376 fetch_limit = 5000
377
378 # 2) multi request 구조에서는 oracle/datalake/goodocs 요청도 함께 들어올 수 있습니다.
379 # 이 data node는 h_api 요청만 처리하고 나머지는 그대로 건너뜁니다.
380 source_requests = []
381 for item in _request_items(request_body):
382 source_type = _source_type(item.get("source_type") or item.get("source") or _source_config(item).get("source_type"))
383 if source_type == "h_api":
384 source_requests.append(item)
385 if not source_requests:
386 return {"skipped": True, "source_type": "h_api", "skip_reason": "No h_api request.", "items": []}
387
388 # 3) 같은 flow 안에서 여러 H-API source를 요청할 수 있으므로 item별 결과를 list로 유지합니다.
389 items = []
390 for item in source_requests:
391 items.append(_run_h_api(item, h_api_token, fetch_limit))
392 return {"source_type": "h_api", "items": items}
