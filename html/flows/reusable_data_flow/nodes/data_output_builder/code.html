36def _payload_from_value(value: Any) -> Dict[str, Any]:
37 """Merger 출력이 Data/JSON/Text 중 무엇이든 내부 처리용 dict로 맞춥니다."""
38 # 연결이 비어 있으면 출력할 body가 없는 상태입니다.
39 if value is None:
40 return {}
41
42 # dict payload는 원본 변경을 피하기 위해 복사합니다.
43 if isinstance(value, dict):
44 return deepcopy(value)
45
46 # Langflow Data 객체는 .data 안에 merger 결과를 담습니다.
47 data = getattr(value, "data", None)
48 if isinstance(data, dict):
49 return deepcopy(data)
50
51 # Message/Text로 JSON이 들어온 경우도 허용합니다.
52 text = getattr(value, "text", None) or getattr(value, "content", None)
53 if isinstance(text, str):
54 try:
55 parsed = json.loads(text)
56 return parsed if isinstance(parsed, dict) else {"text": text}
57 except Exception:
58 # JSON이 아니면 원문만 남기고 body 변환 단계에서 빈 결과로 처리됩니다.
59 return {"text": text}
60 return {}

156def _columns_from_body(body: Dict[str, Any], rows: list[Dict[str, Any]]) -> list[str]:
157 """body.columns와 실제 row key를 합쳐 표시 컬럼 순서를 만듭니다."""
158 columns: list[str] = []
159
160 # merger가 넘긴 columns를 우선 사용합니다.
161 for column in body.get("columns", []):
162 text = str(column or "").strip()
163 if text and text not in columns:
164 columns.append(text)
165
166 # columns에 없는 key가 row에 있으면 뒤에 추가해 데이터가 누락되어 보이지 않게 합니다.
167 for row in rows:
168 for key in row:
169 text = str(key)
170 if text not in columns:
171 columns.append(text)
172 return columns

175def _cell_text(value: Any, max_cell_chars: int = DEFAULT_MAX_CELL_CHARS) -> str:
176 """Markdown 표 한 칸에 넣을 값을 짧고 안전한 문자열로 바꿉니다."""
177 if value is None:
178 return ""
179 if isinstance(value, (dict, list, tuple, set)):
180 # 복잡한 값은 JSON 문자열로 바꿔 표 안에서 구조를 볼 수 있게 합니다.
181 text = json.dumps(value, ensure_ascii=False, default=str)
182 else:
183 text = str(value)
184
185 # Markdown 표 문법을 깨는 줄바꿈과 파이프 문자를 정리합니다.
186 text = text.replace("\r", " ").replace("\n", " ").replace("|", "\\|").strip()
187 if len(text) > max_cell_chars:
188 # 너무 긴 셀은 Chat Output 가독성을 위해 잘라냅니다.
189 return text[: max_cell_chars - 1] + "..."
190 return text

193def _markdown_table(rows: list[Dict[str, Any]], columns: list[str], max_message_rows: int = DEFAULT_MAX_MESSAGE_ROWS, max_cell_chars: int = DEFAULT_MAX_CELL_CHARS) -> str:
194 """테스트용 Message 출력에서 데이터를 빠르게 확인할 Markdown 표를 만듭니다."""
195 # 조회 결과가 없을 때도 Chat Output이 빈 화면처럼 보이지 않도록 안내 문구를 반환합니다.
196 if not rows:
197 return "_표시할 데이터가 없습니다._"
198
199 # columns가 비어 있으면 row의 key 순서를 모아 표 헤더를 자동 생성합니다.
200 if not columns:
201 columns = _columns_from_body({}, rows)
202
203 # Markdown 표는 헤더, 구분선, 데이터 행 순서로 구성합니다.
204 header = "| " + " | ".join(columns) + " |"
205 separator = "| " + " | ".join(["---"] * len(columns)) + " |"
206 lines = [header, separator]
207
208 # Chat Output에서 너무 긴 표가 화면을 덮지 않도록 설정된 행 수만 표시합니다.
209 for row in rows[:max_message_rows]:
210 values = [_cell_text(row.get(column), max_cell_chars) for column in columns]
211 lines.append("| " + " | ".join(values) + " |")
212 if len(rows) > max_message_rows:
213 # 전체 row 수는 남겨 사용자가 결과가 잘린 것을 알 수 있게 합니다.
214 lines.append(f"\n_총 {len(rows)}행 중 {max_message_rows}행만 표시했습니다._")
215 return "\n".join(lines)

229def build_simple_data_message(data_result_value: Any, max_message_rows_value: Any = DEFAULT_MAX_MESSAGE_ROWS, max_cell_chars_value: Any = DEFAULT_MAX_CELL_CHARS) -> str:
230 """Chat Output에 연결해 사람이 확인할 수 있는 요약 메시지를 만듭니다."""
231 max_message_rows = _positive_int(max_message_rows_value, DEFAULT_MAX_MESSAGE_ROWS)
232 max_cell_chars = _positive_int(max_cell_chars_value, DEFAULT_MAX_CELL_CHARS)
233 # 1) merger 결과에서 실제 body와 rows/columns를 꺼냅니다.
234 # data_result 필드를 우선 사용하고, 과거 rows 필드는 보조로만 읽습니다.
235 body = _body_from_value(data_result_value)
236 rows = _message_rows(body)
237 columns = _columns_from_body(body, rows)
238 row_count = len(rows)
239 mode = str(body.get("mode") or "-")
240 success = bool(body.get("success", False))
241 status = "성공" if success else "실패"
242
243 # 2) 사람이 먼저 봐야 하는 실행 상태를 간단한 bullet로 만듭니다.
244 lines = [
245 "### 데이터 조회 결과",
246 f"- 상태: {status}",
247 f"- 모드: {mode}",
248 f"- 행 수: {row_count}",
249 ]
250
251 # 3) 여러 source가 함께 실행된 경우 source_results를 기준으로 어떤 source가 몇 행을 냈는지 요약합니다.
252 source_results = body.get("source_results") if isinstance(body.get("source_results"), list) else []
253 if source_results:
254 source_names = []
255 for source in source_results:
256 if not isinstance(source, dict):
257 continue
258 name = str(source.get("name") or source.get("source_type") or "").strip()
259 source_type = str(source.get("source_type") or "").strip()
260 source_rows = _as_rows(source.get("data_result"))
261 row_text = str(source.get("row_count", len(source_rows))).strip()
262 label = name
263 if source_type:
264 label = f"{label}({source_type})" if label else source_type
265 if row_text:
266 label = f"{label}: {row_text}행"
267 if label:
268 source_names.append(label)
269 if source_names:
270 lines.append("- 소스: " + ", ".join(source_names))
271
272 # 4) 실패 source가 있으면 source_results 안의 error_message를 표 위에 먼저 보여줍니다.
273 errors = []
274 for source in source_results:
275 if not isinstance(source, dict):
276 continue
277 if bool(source.get("success")):
278 continue
279 error_message = source.get("error_message")
280 if error_message:
281 errors.append(
282 {
283 "name": source.get("name") or source.get("source_type") or "source",
284 "error_message": error_message,
285 }
286 )
287 if errors:
288 lines.append("")
289 lines.append("#### 오류")
290 for error in errors:
291 if isinstance(error, dict):
292 lines.append(f"- {error.get('name', 'source')}: {error.get('error_message', error)}")
293 else:
294 lines.append(f"- {error}")
295
296 # 5) 같은 source_type/source_name이 여러 번 실행된 경우에는 결과를 요청별로 나눠 보여줍니다.
297 # API용 data_json.data_result는 조회 순서별 실제 데이터 목록이고, source_results에는 상세 정보를 남깁니다.
298 if len(source_results) > 1:
299 lines.append("")
300 lines.append("#### 요청별 결과")
301 for index, source in enumerate(source_results, start=1):
302 if not isinstance(source, dict):
303 continue
304 name = str(source.get("name") or source.get("source_type") or f"source_{index}")
305 source_type = str(source.get("source_type") or "")
306 title = f"{index}. {name}"
307 if source_type:
308 title += f" ({source_type})"
309 lines.append("")
310 lines.append(f"##### {title}")
311 params_text = _request_params_text(source, max_cell_chars)
312 if params_text:
313 lines.append(f"- params: {params_text}")
314 # bullet 바로 다음 줄에 표가 오면 일부 Chat Output 렌더러가 표를 일반 텍스트로 처리합니다.
315 # 빈 줄을 하나 넣어 표를 독립 Markdown 블록으로 분리합니다.
316 lines.append("")
317 source_rows = _as_rows(source.get("data_result"))
318 source_columns = _columns_from_body(source, source_rows)
319 lines.append(_markdown_table(source_rows, source_columns, max_message_rows, max_cell_chars))
320 return "\n".join(lines)
321
322 # 6) 단일 source이거나 source_results가 없는 예전 payload는 기존처럼 한 표로 보여줍니다.
323 lines.append("")
324 lines.append(_markdown_table(rows, columns, max_message_rows, max_cell_chars))
325 return "\n".join(lines)

340 "test_message": test_message,
