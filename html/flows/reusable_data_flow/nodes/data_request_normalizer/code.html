47def _text_from_value(value: Any) -> str:
48 """Message/Data/dict 안에서 LLM이 만든 텍스트를 찾아 꺼냅니다."""
49 # None은 연결이 비어 있는 상태이므로 빈 문자열로 통일합니다.
50 if value is None:
51 return ""
52
53 # Text Input이나 MessageTextInput에서 바로 온 값은 문자열 자체입니다.
54 if isinstance(value, str):
55 return value.strip()
56
57 # dict payload에서는 Langflow/LLM/Text Input wrapper마다 다른 key를 쓸 수 있어 흔한 key를 순서대로 확인합니다.
58 if isinstance(value, dict):
59 for key in ("llm_result", "llm_text", "text", "content", "message", "response", "output", "result", "value", "output_text", "output_value"):
60 item = value.get(key)
61 if isinstance(item, str) and item.strip():
62 return item.strip()
63 if isinstance(item, dict):
64 # 값이 한 번 더 dict로 감싸진 경우 재귀적으로 안쪽 텍스트를 찾습니다.
65 nested = _text_from_value(item)
66 if nested:
67 return nested
68 if isinstance(value.get("data"), dict):
69 # 일부 Data preview/copy 결과는 {"data": {...}} 모양으로 한 번 더 감싸질 수 있습니다.
70 nested = _text_from_value(value["data"])
71 if nested:
72 return nested
73

78 return item.strip()

80 # Langflow Data 객체 안의 .data가 dict이면 위 dict 처리 로직을 재사용합니다.
81 data = getattr(value, "data", None)

87def _extract_json_text(text: str) -> str:
88 """마크다운 코드블록이나 설명문에 섞인 JSON object/list 부분만 잘라냅니다."""
89 # 1) LLM이 ```json ... ``` 형태로 감싸는 경우가 많아서 먼저 코드블록 껍데기를 제거합니다.
90 # 아래 정규식은 ```json 으로 시작해서 다음 ``` 전까지의 본문만 찾습니다.
91 cleaned = str(text or "").strip()
92 fence = re.search(r"```(?:json)?\s*(.*?)```", cleaned, flags=re.DOTALL | re.IGNORECASE)
93 if fence:
94 cleaned = fence.group(1).strip()
95
96 # 2) 응답이 설명문 + JSON일 수도 있으므로, object({})와 list([]) 후보를 모두 찾습니다.
97 candidates: list[tuple[int, int, str]] = []
98 first_object = cleaned.find("{")
99 last_object = cleaned.rfind("}")
100 if 0 <= first_object < last_object:
101 candidates.append((first_object, last_object, cleaned[first_object : last_object + 1]))
102 first_list = cleaned.find("[")
103 last_list = cleaned.rfind("]")
104 if 0 <= first_list < last_list:
105 candidates.append((first_list, last_list, cleaned[first_list : last_list + 1]))
106
107 # 3) 가장 먼저 시작하는 JSON 후보를 선택합니다.
108 # 같은 위치라면 더 긴 후보를 고르면 중첩 JSON을 덜 잘라내게 됩니다.
109 if candidates:
110 candidates.sort(key=lambda item: (item[0], -(item[1] - item[0])))
111 return candidates[0][2]
112 return cleaned

115def _escape_newlines_inside_json_strings(text: str) -> str:
116 """JSON 문자열 값 안에 실제 줄바꿈이 섞인 경우 \\n 문자로 보정합니다."""
117 # catalog_data preview를 복사하면 query_template 값 안의 \n이 실제 줄바꿈으로 풀려
118 # JSON 파서가 실패할 수 있습니다. 따옴표 안쪽 줄바꿈만 escape해서 다시 JSON으로 읽게 합니다.
119 result: list[str] = []
120 in_string = False
121 escaped = False
122 for char in str(text or ""):
123 if escaped:
124 result.append(char)
125 escaped = False
126 continue
127 if char == "\\" and in_string:
128 result.append(char)
129 escaped = True
130 continue
131 if char == '"':
132 in_string = not in_string
133 result.append(char)
134 continue
135 if in_string and char == "\n":
136 result.append("\\n")
137 continue
138 if in_string and char == "\r":
139 result.append("\\r")
140 continue
141 result.append(char)
142 return "".join(result)

145def _parse_json_or_python(text: str) -> Any:
146 """JSON을 우선 파싱하고, 작은따옴표 dict 같은 Python literal도 보조로 허용합니다."""
147 # LLM 응답이 코드블록/설명문을 포함할 수 있어 JSON으로 보이는 부분만 먼저 잘라냅니다.
148 cleaned = _extract_json_text(text)
149
150 # 표준 JSON을 먼저 시도하고, 작은따옴표 dict 같은 Python literal을 보조로 허용합니다.
151 for parser in (json.loads, ast.literal_eval):
152 try:
153 return parser(cleaned)
154 except Exception:
155 pass
156
157 # Text Input에 붙여넣은 catalog_data가 query_template 안에 실제 줄바꿈을 포함하면
158 # JSON 문법상 invalid가 되므로, 따옴표 안의 줄바꿈만 \n으로 바꿔 한 번 더 시도합니다.
159 repaired = _escape_newlines_inside_json_strings(cleaned)
160 if repaired != cleaned:
161 try:
162 return json.loads(repaired)
163 except Exception:
164 pass
165
166 # 둘 다 실패하면 호출부가 오류 처리로 넘어갈 수 있도록 None을 반환합니다.
167 return None

190def _string_list(value: Any) -> list[str]:
191 """쉼표, 세미콜론, 줄바꿈으로 적은 값을 깨끗한 문자열 목록으로 바꿉니다."""
192 result: list[str] = []
193
194 # 사람이 Text Input에 "DATE, YM"처럼 적은 경우를 지원하기 위해 문자열은 구분자로 분해합니다.
195 # 정규식 [,;\n]은 쉼표, 세미콜론, 줄바꿈 중 하나를 구분자로 본다는 뜻입니다.
196 if isinstance(value, str):
197 items = re.split(r"[,;\n]", value)
198 else:
199 # list/tuple/set/단일 값은 공통 list 형태로 맞춘 뒤 같은 루프에서 정리합니다.
200 items = _as_list(value)
201
202 # 빈 값은 버리고 앞뒤 공백을 제거해 catalog/request 비교가 안정적으로 되게 합니다.
203 for item in items:
204 text = str(item or "").strip()
205 if text:
206 result.append(text)
207 return result

210def _string_dict(value: Any) -> Dict[str, str]:
211 """param_formats처럼 key/value로 된 설정을 깨끗한 dict로 맞춥니다."""
212 result: Dict[str, str] = {}
213 if isinstance(value, dict):
214 for key, item in value.items():
215 name = str(key or "").strip()
216 text = str(item or "").strip()
217 if name and text:
218 result[name] = text
219 return result
220
221 # 사람이 Text Input에 "DATE=YYYYMMDD, YM=YYYYMM"처럼 적은 경우를 지원합니다.
222 for part in re.split(r"[,;\n]", str(value or "")):
223 match = re.match(r"\s*([A-Za-z][A-Za-z0-9_]*)\s*(?:=|:)\s*(.+?)\s*$", part)
224 if match:
225 result[match.group(1)] = match.group(2).strip()
226 return result

246def _catalog_sources(source_catalog_value: Any) -> list[Dict[str, Any]]:
247 """Text Input 또는 Data로 들어온 source_catalog를 source dict 목록으로 정규화합니다."""
248 # 1) Text Input은 문자열로, 이전 노드 Data 출력은 dict로 들어올 수 있으므로 둘 다 준비합니다.
249 text = _text_from_value(source_catalog_value)
250 payload = _payload_from_value(source_catalog_value)
251
252 # 2) 문자열만 있는 경우 JSON/Python literal로 해석합니다.
253 # 줄글 source 설명은 Catalog Normalizer에서 먼저 JSON catalog로 바꾸는 것을 표준으로 둡니다.
254 if text and not any(key in payload for key in ("source_catalog", "catalog", "sources", "data")):
255 parsed = _parse_json_or_python(text)
256 if isinstance(parsed, dict):
257 payload = parsed
258 elif isinstance(parsed, list):
259 payload = {"sources": parsed}
260
261 # 3) catalog가 한 번 더 감싸져 있으면 실제 sources가 들어있는 안쪽 dict로 이동합니다.
262 if isinstance(payload.get("data"), dict):
263 data_payload = payload["data"]
264 if any(key in data_payload for key in ("source_catalog", "catalog", "sources")):
265 payload = deepcopy(data_payload)
266 if isinstance(payload.get("source_catalog"), dict):
267 payload = deepcopy(payload["source_catalog"])
268 if isinstance(payload.get("catalog"), dict):
269 payload = deepcopy(payload["catalog"])
270
271 # 4) sources는 dict/list/단일 source object 세 가지 모양을 허용합니다.
272 raw_sources: list[Any] = []
273 if isinstance(payload.get("sources"), dict):
274 for name, item in payload["sources"].items():
275 if isinstance(item, dict):
276 copied = deepcopy(item)
277 copied.setdefault("name", str(name))
278 raw_sources.append(copied)
279 elif isinstance(payload.get("sources"), list):
280 raw_sources = payload["sources"]
281 elif payload:
282 raw_sources = [payload]
283
284 # 5) data node가 기대하는 표준 필드와 source_config 별칭을 한 번에 정리합니다.
285 sources: list[Dict[str, Any]] = []
286 for item in raw_sources:
287 if not isinstance(item, dict):
288 continue
289 source = deepcopy(item)
290 source["name"] = str(source.get("name") or source.get("source") or source.get("dataset_key") or "").strip()
291 source["source_type"] = _source_type(source.get("source_type") or source.get("source"))
292 source["required_params"] = _string_list(source.get("required_params"))
293 source["param_order"] = _string_list(source.get("param_order")) or list(source["required_params"])
294 source["param_formats"] = _string_dict(source.get("param_formats") or source.get("value_formats"))
295 source["keywords"] = _string_list(source.get("keywords"))
296 source["aliases"] = _string_list(source.get("aliases"))
297 source["example_questions"] = _string_list(source.get("example_questions"))
298 config = deepcopy(source.get("source_config")) if isinstance(source.get("source_config"), dict) else {}
299 for key in ("db_key", "query_template", "sql_template", "sql", "query", "api_url", "url", "timeout", "doc_id", "document_id", "response_path"):
300 if source.get(key) not in (None, "", [], {}) and key not in config:
301 config[key] = deepcopy(source[key])
302 if config.get("url") and not config.get("api_url"):
303 config["api_url"] = config["url"]
304 if config.get("document_id") and not config.get("doc_id"):
305 config["doc_id"] = config["document_id"]
306 for alias in ("sql_template", "sql", "query"):
307 if config.get(alias) and not config.get("query_template"):
308 config["query_template"] = config[alias]
309 source["source_config"] = config
310 sources.append(source)
311 return sources

314def _find_catalog_source(request: Dict[str, Any], sources: list[Dict[str, Any]]) -> Dict[str, Any]:
315 """LLM 요청의 name/source_type과 source_catalog 항목을 연결합니다."""
316 # LLM은 name/source_name/dataset_key/tool_name 중 아무 이름으로 source를 가리킬 수 있습니다.
317 wanted_names = _string_list([request.get("name"), request.get("source_name"), request.get("dataset_key"), request.get("tool_name")])
318 wanted_source_type = _source_type(request.get("source_type") or request.get("source"))
319 lower_names = {name.lower() for name in wanted_names if name}
320
321 # catalog source의 name뿐 아니라 aliases/keywords까지 비교해 자연어 기반 매칭을 조금 더 유연하게 합니다.
322 for source in sources:
323 names = [source.get("name")] + _as_list(source.get("aliases")) + _as_list(source.get("keywords"))
324 if any(str(name or "").strip().lower() in lower_names for name in names):
325 return source
326
327 # 이름으로 못 찾았고 같은 source_type이 딱 하나라면 그 source를 안전하게 선택할 수 있습니다.
328 same_type = [source for source in sources if source.get("source_type") == wanted_source_type and wanted_source_type]
329 return same_type[0] if len(same_type) == 1 else {}

332def _normalize_request(request: Dict[str, Any], sources: list[Dict[str, Any]]) -> tuple[Dict[str, Any], list[str]]:
333 """LLM이 만든 짧은 요청에 catalog의 실행 설정을 채워 실행 가능한 request로 만듭니다."""
334 warnings: list[str] = []
335
336 # 1) LLM은 보통 name/source_type/params 정도만 반환합니다.
337 # source_catalog에서 같은 source를 찾아 query/api/doc 설정을 기본값으로 깔아 둡니다.
338 catalog = _find_catalog_source(request, sources)
339 normalized = deepcopy(catalog) if catalog else {}
340
341 # 2) 사용자 질문에서 추출된 params/variables는 실제 실행값이므로 catalog 값보다 우선합니다.
342 if request:
343 normalized["params"] = deepcopy(request.get("params")) if isinstance(request.get("params"), dict) else {}
344 if isinstance(request.get("variables"), dict):
345 for key, value in request["variables"].items():
346 normalized["params"].setdefault(key, deepcopy(value))
347
348 # 3) LLM이 직접 넘긴 source_config가 있으면 catalog 설정 위에 덮어씁니다.
349 # 다만 정상 흐름에서는 SQL/API URL을 LLM이 매번 복사하지 않도록 catalog에서 채우는 편이 좋습니다.
350 normalized["name"] = str(request.get("name") or normalized.get("name") or request.get("dataset_key") or request.get("tool_name") or "").strip()
351 normalized["source_type"] = _source_type(request.get("source_type") or normalized.get("source_type") or request.get("source"))
352 if isinstance(request.get("source_config"), dict):
353 config = deepcopy(normalized.get("source_config")) if isinstance(normalized.get("source_config"), dict) else {}
354 config.update(deepcopy(request["source_config"]))
355 normalized["source_config"] = config
356 for key in ("required_params", "param_order"):
357 if request.get(key):
358 normalized[key] = _string_list(request.get(key))
359 if request.get("param_formats") or request.get("value_formats"):
360 # LLM이 source_catalog의 param_formats를 다시 보내거나 보정한 경우도 보존합니다.
361 param_formats = _string_dict(normalized.get("param_formats"))
362 param_formats.update(_string_dict(request.get("param_formats") or request.get("value_formats")))
363 normalized["param_formats"] = param_formats
364
365 # 4) 뒤쪽 data node가 key 유무를 매번 방어하지 않아도 되도록 빈 기본값을 보장합니다.
366 if not normalized.get("source_type"):
367 warnings.append("Request is missing source_type.")
368 if not normalized.get("name"):
369 normalized["name"] = str(normalized.get("source_type") or "source")
370 normalized["params"] = deepcopy(normalized.get("params")) if isinstance(normalized.get("params"), dict) else {}
371 normalized["required_params"] = _string_list(normalized.get("required_params"))
372 normalized["param_order"] = _string_list(normalized.get("param_order")) or list(normalized["required_params"])
373 normalized["param_formats"] = _string_dict(normalized.get("param_formats"))
374 normalized["source_config"] = deepcopy(normalized.get("source_config")) if isinstance(normalized.get("source_config"), dict) else {}
375 return normalized, warnings

402 # 2) LLM 응답은 Message 텍스트일 수도 있고 Data dict일 수도 있어 텍스트 파싱을 먼저 시도합니다.
403 parsed: Any = _parse_json_or_python(text) if text else None
404 if parsed is None:
405 parsed = _payload_from_value(llm_result_value)
406 if not isinstance(parsed, dict):
407 errors.append("LLM output must be a JSON object.")
408 parsed = {}
409
410 # 3) data_request/body wrapper를 벗긴 뒤 실제 request 후보만 처리합니다.
411 candidate = _unwrap_candidate(parsed)
412
413 if candidate.get("needs_more_info"):
414 # 4-A) LLM이 정보 부족을 선언한 경우에는 data node로 보내지 않고 질문 목록만 유지합니다.
415 questions = _string_list(candidate.get("questions"))
416 data_request = {"needs_more_info": True, "questions": questions, "requests": []}
417 elif isinstance(candidate.get("requests"), list):
418 # 4-B) 여러 source를 한 번에 조회하는 요청은 각 item별로 catalog 설정을 보강합니다.
419 requests = []
420 for item in candidate["requests"]:
421 if isinstance(item, dict):
422 normalized, item_warnings = _normalize_request(item, sources)
423 requests.append(normalized)
424 warnings.extend(item_warnings)
425 else:
426 warnings.append("Ignored a non-object request item.")
427 data_request = {"requests": requests}
428 elif candidate:
429 # 4-C) 단일 source 조회는 data_request 자체가 하나의 실행 요청이 됩니다.
430 data_request, item_warnings = _normalize_request(candidate, sources)
431 warnings.extend(item_warnings)
432 else:
433 data_request = {}
434
435 return {"data_request": data_request, "parse_errors": errors, "warnings": warnings}
