97def _extract_json_text(text: str) -> str:
98 """마크다운 코드블록이나 설명문에 섞인 JSON object/list 부분만 잘라냅니다."""
99 # LLM은 JSON만 반환하라고 해도 ```json ... ``` 코드블록으로 감싸는 경우가 있습니다.
100 # 먼저 앞뒤 공백을 제거하고, 코드블록이 있으면 블록 내부만 파싱 대상으로 좁힙니다.
101 cleaned = str(text or "").strip()
102 # 정규식은 ```json 으로 시작해서 다음 ``` 전까지의 본문만 찾습니다.
103 fence = re.search(r"```(?:json)?\s*(.*?)```", cleaned, flags=re.DOTALL | re.IGNORECASE)
104 if fence:
105 cleaned = fence.group(1).strip()
106
107 # 응답 앞뒤에 "아래 JSON입니다" 같은 설명문이 붙을 수 있으므로
108 # JSON object({ ... }) 후보와 JSON list([ ... ]) 후보를 둘 다 찾아둡니다.
109 candidates: list[tuple[int, int, str]] = []
110 first_object = cleaned.find("{")
111 last_object = cleaned.rfind("}")
112 if 0 <= first_object < last_object:
113 candidates.append((first_object, last_object, cleaned[first_object : last_object + 1]))
114
115 # source_catalog는 object로 오는 경우가 많지만, LLM이 source 배열만 반환할 수도 있습니다.
116 # 그래서 list 후보도 별도로 잘라두고 아래에서 더 적절한 후보를 고릅니다.
117 first_list = cleaned.find("[")
118 last_list = cleaned.rfind("]")
119 if 0 <= first_list < last_list:
120 candidates.append((first_list, last_list, cleaned[first_list : last_list + 1]))
121
122 if candidates:
123 # 설명문 중간에 JSON이 있을 때는 가장 먼저 시작하는 후보가 실제 JSON일 가능성이 큽니다.
124 # 시작 위치가 같다면 더 긴 후보를 고르면 중첩 JSON을 중간에서 잘라먹는 일을 줄일 수 있습니다.
125 candidates.sort(key=lambda item: (item[0], -(item[1] - item[0])))
126 return candidates[0][2]
127
128 # JSON 괄호를 찾지 못한 경우에는 원문 전체를 그대로 반환합니다.
129 # 이후 단계의 json.loads / ast.literal_eval이 실패하면 줄글 파서로 넘어갑니다.
130 return cleaned

133def _parse_json_or_python(text: str) -> Any:
134 """JSON을 우선 파싱하고, 작은따옴표 dict 같은 Python literal도 보조로 허용합니다."""
135 # 코드블록/설명문을 제거한 JSON 후보만 파서에 넣습니다.
136 cleaned = _extract_json_text(text)
137
138 # 표준 JSON을 먼저 시도하고, Python literal은 보조로만 허용합니다.
139 for parser in (json.loads, ast.literal_eval):
140 try:
141 parsed = parser(cleaned)
142 if isinstance(parsed, str) and parsed.strip() and parsed.strip() != cleaned:
143 # LLM이 JSON 문자열 안에 다시 JSON을 문자열로 넣는 경우가 있어 한 번 더 파싱합니다.
144 nested = _parse_json_or_python(parsed)
145 if nested is not None:
146 return nested
147 return parsed
148 except Exception:
149 pass
150
151 # 모두 실패하면 줄글 parser가 처리할 수 있도록 None을 반환합니다.
152 return None

197def _param_formats_from_value(value: Any) -> Dict[str, str]:
198 """dict 또는 DATE=YYYYMMDD 문자열을 param_formats dict로 변환합니다."""
199 result: Dict[str, str] = {}
200 if isinstance(value, dict):

216def _param_formats_from_sentence(line: str) -> Dict[str, str]:
217 """'DATE 형식은 YYYYMMDD' 같은 줄글에서 변수 형식을 추출합니다."""
218 result: Dict[str, str] = {}
219 text = str(line or "")
220 if "형식" not in text and "포맷" not in text and "format" not in text.lower():
221 return result
222
223 # "DATE 형식은 YYYYMMDD" 또는 "DATE format: YYYY-MM-DD" 형태를 찾습니다.
224 pattern_after_label = rf"([A-Za-z][A-Za-z0-9_]*)\s*(?:의\s*)?(?:값\s*)?(?:형식|포맷|format)\s*(?:은|는|:|=)?\s*({PARAM_FORMAT_PATTERN})"
225 for match in re.finditer(pattern_after_label, text, flags=re.IGNORECASE):
226 result[match.group(1)] = _clean_param_format(match.group(2))
227
228 # "FROM_YM, TO_YM 형식은 YYYYMM"처럼 여러 변수를 한꺼번에 적은 문장도 처리합니다.
229 group_pattern = rf"([A-Za-z][A-Za-z0-9_,\s]*?)\s*(?:형식|포맷|format)\s*(?:은|는|:|=)\s*({PARAM_FORMAT_PATTERN})"
230 for match in re.finditer(group_pattern, text, flags=re.IGNORECASE):
231 for name in _identifier_list(match.group(1)):
232 result[name] = _clean_param_format(match.group(2))
233
234 # "DATE는 YYYYMMDD 형식"처럼 형식값이 먼저 나오는 문장도 허용합니다.
235 pattern_before_label = rf"([A-Za-z][A-Za-z0-9_]*)\s*(?:은|는|:|=)\s*({PARAM_FORMAT_PATTERN})\s*(?:형식|포맷|format)"
236 for match in re.finditer(pattern_before_label, text, flags=re.IGNORECASE):
237 result[match.group(1)] = _clean_param_format(match.group(2))
238 return result

379def _simple_catalog_blocks(text: str) -> list[Dict[str, Any]]:
380 """LLM 없이 줄글로 쓴 source 설명을 source별 dict 목록으로 나눕니다."""
381 # 사용자가 여러 source를 한 번에 적을 때는 --- 구분선을 쓰도록 안내했습니다.
382 # 여기서는 각 블록을 하나의 source 설명으로 보고 독립적으로 파싱합니다.
383 # 줄 전체가 --- 인 부분을 source 구분선으로 보고 여러 source 설명을 나눕니다.
384 blocks = re.split(r"(?m)^\s*---+\s*$", str(text or "").strip())
385 sources: list[Dict[str, Any]] = []
386
387 for block in blocks:
388 # source에는 표준 필드(name, source_type 등)를, source_config에는 실행 설정만 모읍니다.
389 # 이렇게 분리해야 뒤쪽 data node가 source_config만 보고 실행할 수 있습니다.
390 source: Dict[str, Any] = {}
391 source_config: Dict[str, Any] = {}
392 description_lines: list[str] = []
393 query_lines: list[str] = []
394 lines = block.splitlines()
395 index = 0
396
397 while index < len(lines):
398 line = lines[index].strip()
399 index += 1
400 if not line:
401 continue
402
403 if "쿼리는 아래와 같다" in line or line.lower().startswith(("query:", "sql:", "query_template:")):
404 # SQL은 줄바꿈이 의미 있는 경우가 많으므로, "쿼리는 아래와 같다" 이후의 모든 줄을
405 # query_template로 보존합니다. 사용자가 복사한 SQL을 최대한 그대로 유지하기 위함입니다.
406 if ":" in line and line.split(":", 1)[1].strip():
407 query_lines.append(line.split(":", 1)[1].strip())
408 while index < len(lines):
409 query_line = lines[index].rstrip()
410 index += 1
411 if query_line.strip() or query_lines:
412 query_lines.append(query_line)
413 continue
414
415 # source: production, db_key: PKG_RPT 같은 key:value 한 줄을 찾습니다.
416 key_match = re.match(r"^([A-Za-z_][A-Za-z0-9_]*)\s*:\s*(.*)$", line)
417 if key_match:
418 # source: production 처럼 명시적인 key:value 형식은 가장 확실한 신호입니다.
419 # 표준 필드는 source에 넣고, URL/쿼리/문서번호 같은 실행값은 source_config에 넣습니다.
420 key = key_match.group(1).strip()
421 value = key_match.group(2).strip()
422 if key in ("source", "name", "source_name"):
423 source["name"] = value
424 elif key == "source_type":
425 source["source_type"] = _source_type(value)
426 elif key in ("keywords", "aliases", "example_questions", "required_params", "param_order"):
427 source[key] = _string_list(value)
428 elif key in ("param_formats", "param_format", "value_formats", "value_format"):
429 source["param_formats"] = _param_formats_from_value(value)
430 elif key in ("doc_id", "document_id", "db_key", "api_url", "url", "query_template"):
431 source_config[key] = value
432 else:
433 source[key] = value
434 continue
435
436 param_formats = _param_formats_from_sentence(line)
437 if param_formats:
438 # DATE=YYYYMMDD 같은 값 형식은 LLM이 params를 만들 때 참고해야 하므로 source 필드로 보존합니다.
439 current_formats = source.get("param_formats") if isinstance(source.get("param_formats"), dict) else {}
440 current_formats.update(param_formats)
441 source["param_formats"] = current_formats
442 continue
443
444 if "필수 파라미터" in line:
445 # "DATE가 필수 파라미터다" 같은 자연어 문장에서 필요한 변수명을 뽑습니다.
446 # param_order가 같이 적혀 있으면 H-API bindParams 순서에도 사용합니다.
447 required_params = source.get("required_params") if isinstance(source.get("required_params"), list) else []
448 for param in _required_params_from_sentence(line):
449 _add_unique_text(required_params, param)
450 source["required_params"] = required_params
451
452 param_order = source.get("param_order") if isinstance(source.get("param_order"), list) else []
453 for param in _param_order_from_sentence(line):
454 _add_unique_text(param_order, param)
455 if param_order:
456 source["param_order"] = param_order
457 continue
458
459 lower_line = line.lower()
460 if "예시 질문" in line or re.match(r"^(example|examples|example_questions)\b", lower_line):
461 # 예시 질문은 나중에 LLM이 어떤 source를 골라야 하는지 판단하는 힌트입니다.
462 # 따옴표 안에 있는 문장만 뽑아 example_questions에 저장합니다.
463 questions = source.get("example_questions") if isinstance(source.get("example_questions"), list) else []
464 for question in _quoted_values(line):
465 _add_unique_text(questions, question)
466 source["example_questions"] = questions
467 continue
468
469 response_path = _response_path_from_sentence(line)
470 if response_path:
471 # H-API 응답에서 실제 row 배열이 들어있는 위치를 source_config에 저장합니다.
472 # 예: "data 필드 안에 row 배열" -> data.row
473 source_config["response_path"] = response_path
474 continue
475
476 api_url = _api_url_from_sentence(line)
477 if api_url:
478 # URL은 문장 어디에 있어도 API 호출에 필요한 실행값이므로 source_config로 이동합니다.
479 source_config["api_url"] = api_url
480 continue
481
482 if "문서번호" in line or "doc_id" in line.lower() or "document_id" in line.lower():
483 # Goodocs는 doc_id가 핵심 실행값입니다.
484 # "문서번호는 XXX이다" 같은 문장을 사람이 읽는 조사/마침표 없이 값만 남깁니다.
485 value = line
486 if ":" in value:
487 value = value.split(":", 1)[1]
488 elif "는" in value:
489 value = value.split("는", 1)[1]
490 source_config["doc_id"] = _clean_sentence_value(value)
491 continue
492
493 if "같은 말이 나오면" in line or "키워드" in line or "keyword" in line.lower():
494 # "생산, output 같은 말이 나오면..." 문장에서 검색 키워드를 추출합니다.
495 # 이 값은 이후 자연어 요청을 어떤 source로 보낼지 판단할 때 사용됩니다.
496 keywords = source.get("keywords") if isinstance(source.get("keywords"), list) else []
497 for keyword in _keywords_from_sentence(line):
498 if keyword not in keywords:
499 keywords.append(keyword)
500 source["keywords"] = keywords
501 continue
502
503 source_type = _infer_source_type_from_text(line)
504 if source_type:
505 # "Oracle에서 조회한다", "H-API로 조회한다"처럼 source_type이 문장에 섞인 경우입니다.
506 # Oracle 문장에는 db_key가 같이 붙는 경우가 많아 함께 추출합니다.
507 if not source.get("source_type"):
508 source["source_type"] = source_type
509 db_key = _db_key_from_oracle_sentence(line)
510 if db_key and not source_config.get("db_key"):
511 source_config["db_key"] = db_key
512 description_lines.append(line)
513 continue
514
515 description_lines.append(line)
516
517 if source or source_config or description_lines:
518 # 명시 description이 없으면, 파싱 중 실행 설정으로 분류되지 않은 문장들을 설명으로 묶습니다.
519 # query_lines가 있으면 줄바꿈을 보존해서 source_config.query_template에 저장합니다.
520 if description_lines and not source.get("description"):
521 source["description"] = " ".join(description_lines).strip()
522 if query_lines:
523 source_config["query_template"] = "\n".join(query_lines).strip()
524 if source_config:
525 source["source_config"] = source_config
526 sources.append(source)
527
528 return sources

531def _copy_runtime_config(source: Dict[str, Any]) -> Dict[str, Any]:
532 """query_template, api_url, doc_id처럼 실행에 필요한 설정만 source_config에 모읍니다."""
533 # source_config가 이미 있으면 그것을 기본으로 두고, 상위에 흩어진 실행 key를 추가로 모읍니다.
534 config = deepcopy(source.get("source_config")) if isinstance(source.get("source_config"), dict) else {}
535
536 # 아래 key들은 source 설명이 아니라 실제 data node 실행에 쓰이는 설정값입니다.
537 runtime_keys = [
538 "db_key",
539 "query_template",
540 "sql_template",
541 "sql",
542 "query",
543 "api_url",
544 "url",
545 "timeout",
546 "response_path",
547 "doc_id",
548 "document_id",
549 ]
550 for key in runtime_keys:
551 if source.get(key) not in (None, "", [], {}) and key not in config:
552 config[key] = deepcopy(source[key])
553
554 # url/document_id/sql 같은 별칭은 data node가 기대하는 표준 key로 통일합니다.
555 if config.get("url") and not config.get("api_url"):
556 config["api_url"] = config["url"]
557 if config.get("document_id") and not config.get("doc_id"):
558 config["doc_id"] = config["document_id"]
559 for alias in ("sql_template", "sql", "query"):
560 if config.get(alias) and not config.get("query_template"):
561 config["query_template"] = config[alias]
562 return config

565def _source_name(source: Dict[str, Any], index: int) -> str:
566 """source 이름이 없을 때도 source_1 같은 안전한 이름을 보장합니다."""
567 # LLM이나 줄글 parser가 source 이름을 여러 key 중 하나로 만들 수 있어 순서대로 확인합니다.
568 for key in ("name", "source", "source_name", "dataset_key", "tool_name"):
569 text = str(source.get(key) or "").strip()
570 if text:
571 if "\n" in text:
572 # 잘못 파싱되어 여러 줄이 name에 들어온 경우 첫 줄만 사용합니다.
573 text = text.splitlines()[0].strip()
574 if text.lower().startswith("source:"):
575 # "source: production"처럼 값까지 들어온 경우 production만 남깁니다.
576 text = text.split(":", 1)[1].strip()
577 if not text:
578 continue
579 return text
580
581 # 이름을 전혀 못 찾으면 source_1, source_2처럼 안전한 fallback 이름을 만듭니다.
582 return f"source_{index}"

585def _normalize_one_source(raw_source: Dict[str, Any], index: int) -> tuple[str, Dict[str, Any], list[str], list[str]]:
586 """source 한 개를 표준 필드로 정리하고 누락 정보 질문을 만듭니다."""
587 # 원본 source는 그대로 보존하기 위해 복사본에서 정규화합니다.
588 # warnings는 동작은 가능하지만 주의가 필요한 정보, questions는 실행에 필요한 누락 정보입니다.
589 warnings: list[str] = []
590 questions: list[str] = []
591 source = deepcopy(raw_source)
592 name = _source_name(source, index)
593 source_type = _source_type(source.get("source_type") or source.get("type") or source.get("source"))
594 config = _copy_runtime_config(source)
595
596 # required_params가 빠져 있어도 SQL 템플릿에 {DATE}, {YM} 같은 placeholder가 있으면
597 # 실행에 필요한 변수로 볼 수 있으므로 자동으로 보강합니다.
598 required_params = _string_list(source.get("required_params"))
599 inferred_params = _placeholder_params(str(config.get("query_template") or ""))
600 for param in inferred_params:
601 if param not in required_params:
602 required_params.append(param)
603
604 # H-API bindParams처럼 순서가 필요한 source도 있으므로 param_order를 보장합니다.
605 # 별도 순서가 없으면 required_params 순서를 그대로 사용합니다.
606 param_order = _string_list(source.get("param_order"))
607 if not param_order:
608 param_order = list(required_params)
609
610 # downstream 노드들이 공통으로 기대하는 catalog 모양입니다.
611 # 실행 설정은 source_config 하나에만 모아두어 중복 필드를 줄입니다.
612 normalized = {
613 "source_type": source_type,
614 "description": str(source.get("description") or source.get("info") or "").strip(),
615 "keywords": _string_list(source.get("keywords")),
616 "aliases": _string_list(source.get("aliases")),
617 "example_questions": _string_list(source.get("example_questions")),
618 "required_params": required_params,
619 "param_order": param_order,
620 "param_formats": _param_formats_from_value(source.get("param_formats") or source.get("value_formats")),
621 "source_config": config,
622 }
623
624 if not source_type:
625 questions.append(f"{name} source_type을 알려주세요. 가능한 값: oracle, h_api, datalake, goodocs.")
626 elif source_type not in SUPPORTED_SOURCE_TYPES:
627 warnings.append(f"{name} has unsupported source_type: {source_type}")
628
629 # source_type별로 실제 조회에 반드시 필요한 값을 확인합니다.
630 # 누락된 값은 즉시 실패시키지 않고 questions로 돌려 사용자가 보완할 수 있게 합니다.
631 if source_type == "oracle":
632 if not str(config.get("db_key") or "").strip():
633 questions.append(f"{name} Oracle 조회에 사용할 db_key를 알려주세요.")
634 if not str(config.get("query_template") or "").strip():
635 questions.append(f"{name} Oracle query_template을 알려주세요.")
636 elif source_type == "h_api":
637 if not str(config.get("api_url") or "").strip():
638 questions.append(f"{name} H-API api_url을 알려주세요.")
639 if not normalized["param_order"] and normalized["required_params"]:
640 normalized["param_order"] = list(normalized["required_params"])
641 elif source_type == "datalake":
642 if not str(config.get("query_template") or "").strip():
643 questions.append(f"{name} Datalake query_template을 알려주세요.")
644 elif source_type == "goodocs":
645 if not str(config.get("doc_id") or "").strip():
646 questions.append(f"{name} Goodocs doc_id를 알려주세요.")
647
648 return name, normalized, warnings, questions

651def _raw_sources_from_payload(payload: Any) -> tuple[list[Dict[str, Any]], bool, list[str]]:
652 """LLM 결과가 sources dict/list/source_catalog 중 어떤 모양이어도 source 목록으로 풉니다."""
653 # LLM이 정보 부족을 반환한 경우에는 source 정규화를 진행하지 않고 질문만 전달합니다.
654 if isinstance(payload, dict) and payload.get("needs_more_info"):
655 return [], True, _string_list(payload.get("questions"))
656
657 # source_catalog/catalog wrapper가 있으면 실제 catalog 안쪽으로 들어갑니다.
658 if isinstance(payload, dict) and isinstance(payload.get("source_catalog"), dict):
659 payload = payload["source_catalog"]
660 if isinstance(payload, dict) and isinstance(payload.get("catalog"), dict):
661 payload = payload["catalog"]
662
663 raw_sources: list[Dict[str, Any]] = []
664 if isinstance(payload, dict) and isinstance(payload.get("sources"), dict):
665 # {"sources": {"production": {...}}} 형태에서는 dict key를 source name으로 보강합니다.
666 for name, value in payload["sources"].items():
667 if isinstance(value, dict):
668 copied = deepcopy(value)
669 copied.setdefault("name", str(name))
670 raw_sources.append(copied)
671 elif isinstance(payload, dict) and isinstance(payload.get("sources"), list):
672 # {"sources": [{...}, {...}]} 형태입니다.
673 for item in payload["sources"]:
674 if isinstance(item, dict):
675 raw_sources.append(deepcopy(item))
676 elif isinstance(payload, list):
677 # LLM이 sources wrapper 없이 배열만 반환한 경우입니다.
678 for item in payload:
679 if isinstance(item, dict):
680 raw_sources.append(deepcopy(item))
681 elif isinstance(payload, dict) and payload:
682 # 단일 source object로 들어온 경우입니다.
683 raw_sources.append(deepcopy(payload))
684 return raw_sources, False, []

687def normalize_source_catalog_llm_result(llm_result_value: Any) -> Dict[str, Any]:
688 """LLM 결과 또는 줄글 source 설명을 조회 flow에서 쓸 source_catalog JSON으로 정규화합니다."""
689 # 1) 입력이 이미 JSON/Data인 경우를 먼저 처리합니다.
690 # Text Input이나 Chat Output Message로 들어온 경우를 위해 원문 텍스트도 함께 확보합니다.
691 parsed = _payload_from_value(llm_result_value)
692 text = _text_from_value(llm_result_value)
693
694 # 2) dict처럼 보이지만 sources/source_catalog가 없는 경우는
695 # "source: production ..." 같은 줄글을 JSON으로 잘못 감싼 상황일 수 있어 텍스트를 재파싱합니다.
696 if isinstance(parsed, dict) and "sources" not in parsed and "source_catalog" not in parsed and text:
697 text_payload = _parse_json_or_python(text)
698 if text_payload is not None:
699 parsed = text_payload
700 else:
701 # JSON 파싱이 안 되면 사람이 작성한 source 설명으로 보고 block parser를 사용합니다.
702 text_sources = _simple_catalog_blocks(text)
703 if text_sources:
704 parsed = {"sources": text_sources}
705 elif not parsed and text:
706 # JSON 후보 자체가 없을 때도 줄글 parser로 마지막 변환을 시도합니다.
707 text_sources = _simple_catalog_blocks(text)
708 if text_sources:
709 parsed = {"sources": text_sources}
710 if not parsed:
711 return {
712 "source_catalog": {"sources": {}, "needs_more_info": True, "questions": ["소스 설명을 JSON으로 변환하지 못했습니다. 입력 내용을 다시 확인해주세요."]},
713 "warnings": [],
714 "questions": ["소스 설명을 JSON으로 변환하지 못했습니다. 입력 내용을 다시 확인해주세요."],
715 }
716
717 # 3) LLM이 반환한 sources dict/list, 단일 source object 등을 모두 raw source 목록으로 통일합니다.
718 raw_sources, needs_more_info, initial_questions = _raw_sources_from_payload(parsed)
719 sources: Dict[str, Any] = {}
720 warnings: list[str] = []
721 questions: list[str] = list(initial_questions)
722
723 if not needs_more_info:
724 # 4) source별로 표준 필드 정리와 누락 정보 검사를 수행합니다.
725 # 결과 dict의 key는 source 이름이므로 뒤쪽 request normalizer가 이름으로 찾을 수 있습니다.
726 for index, raw_source in enumerate(raw_sources, start=1):
727 name, normalized, item_warnings, item_questions = _normalize_one_source(raw_source, index)
728 sources[name] = normalized
729 warnings.extend(item_warnings)
730 questions.extend(item_questions)
731
732 # 5) 정상적으로 파싱했지만 source가 하나도 없으면 사용자가 바로 수정할 수 있는 질문을 남깁니다.
733 if not sources and not questions:
734 questions.append("변환된 source가 없습니다. 최소 하나의 source 설명을 입력해주세요.")
735
736 # 6) Text Input에 연결하기 좋은 실제 catalog는 source_catalog 아래에 모읍니다.
737 # warnings/questions는 Langflow status나 디버깅에서 바로 볼 수 있도록 상위에도 유지합니다.
738 source_catalog = {
739 "sources": sources,
740 "needs_more_info": bool(questions),
741 "questions": questions,
742 "warnings": warnings,
743 }
744 return {
745 "source_catalog": source_catalog,
746 "warnings": warnings,
747 "questions": questions,
748 }
