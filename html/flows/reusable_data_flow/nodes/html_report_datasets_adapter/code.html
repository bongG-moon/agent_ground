42def _payload_from_value(value: Any) -> Dict[str, Any]:
43 """Data/Message/dict/JSON 문자열 중 무엇이 들어와도 dict로 맞춥니다."""
44
45 if value is None:
46 return {}
47 if isinstance(value, dict):
48 return deepcopy(value)
49 data = getattr(value, "data", None)
50 if isinstance(data, dict):
51 return deepcopy(data)
52 text = getattr(value, "text", None) or getattr(value, "content", None)
53 if isinstance(text, str) and text.strip():
54 try:
55 parsed = json.loads(text)
56 except Exception:
57 return {"text": text}
58 return deepcopy(parsed) if isinstance(parsed, dict) else {"text": text}

70def _result_groups(payload: Dict[str, Any]) -> list[list[Dict[str, Any]]]:
71 """06번 data_json에서 조회 단위별 row 묶음을 꺼냅니다."""
72
73 data_result = payload.get("data_result")
74 if isinstance(data_result, list):
75 if all(isinstance(item, dict) for item in data_result):
76 return [_rows(data_result)]
77 groups = []
78 for item in data_result:
79 rows = _rows(item)
80 if rows:
81 groups.append(rows)
82 if groups:
83 return groups

85 source_results = payload.get("source_results")
86 if isinstance(source_results, list):
87 groups = []
88 for source in source_results:
89 if isinstance(source, dict):
90 rows = _rows(source.get("data_result")) or _rows(source.get("rows"))
91 if rows:
92 groups.append(rows)
93 if groups:
94 return groups

96 rows = _rows(payload.get("rows"))
97 return [rows] if rows else []

123def build_html_report_datasets(data_json_value: Any) -> Dict[str, Any]:
124 """06번 data_json을 HTML flow의 00번 입력용 datasets JSON으로 변환합니다."""
125
126 payload = _payload_from_value(data_json_value)
127 groups = _result_groups(payload)
128 sources = _source_results(payload)
129 datasets = []
130
131 for index, rows in enumerate(groups, start=1):
132 source = sources[index - 1] if index - 1 < len(sources) else {}
133 dataset_id = _dataset_id(source, index)
134 datasets.append(
135 {
136 "dataset_id": dataset_id,
137 "label": _dataset_label(source, dataset_id),
138 "rows": rows,
139 }
140 )
141
142 return {
143 "datasets": datasets,
144 "source": "reusable_data_flow.data_json",
145 "success": bool(payload.get("success", bool(datasets))),
146 "mode": "single" if len(datasets) == 1 else "multi" if len(datasets) > 1 else "empty",
147 }
