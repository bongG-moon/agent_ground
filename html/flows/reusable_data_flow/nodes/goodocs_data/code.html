40def _payload_from_value(value: Any) -> Dict[str, Any]:
41 """Data/Message/Text/JSON 입력을 내부 처리용 dict로 맞춥니다."""
42 # 입력이 비어 있으면 뒤쪽 필터링 단계에서 skipped 처리되도록 빈 dict를 반환합니다.
43 if value is None:
44 return {}
45
46 # 이미 dict로 들어온 payload는 원본 변경을 피하기 위해 복사합니다.
47 if isinstance(value, dict):
48 return deepcopy(value)
49
50 # Langflow Data 객체는 .data에 실제 payload가 들어옵니다.
51 data = getattr(value, "data", None)
52 if isinstance(data, dict):
53 return deepcopy(data)
54
55 # Message/Text 객체는 text/content를 JSON으로 파싱합니다.
56 text = getattr(value, "text", None) or getattr(value, "content", None)
57 if isinstance(text, str):
58 try:
59 parsed = json.loads(text)
60 return parsed if isinstance(parsed, dict) else {"text": text}
61 except Exception:
62 # JSON이 아니면 원문을 보존해 진단할 수 있게 합니다.
63 return {"text": text}
64 return {}

122def _json_ready(value: Any) -> Any:
123 """Goodocs 결과에 섞인 날짜/Decimal/NaN 등을 JSON 안전값으로 바꿉니다."""
124 if value is None or isinstance(value, (str, int, float, bool)):
125 return value
126 if isinstance(value, (datetime, date)):
127 return value.isoformat()
128 if isinstance(value, Decimal):
129 return float(value)
130 if isinstance(value, dict):
131 # dict key는 문자열로 맞추고 value는 재귀적으로 정리합니다.
132 return {str(key): _json_ready(item) for key, item in value.items()}
133 if isinstance(value, (list, tuple, set)):
134 return [_json_ready(item) for item in value]
135 try:
136 # pandas/numpy NaN은 자기 자신과 같지 않으므로 None으로 바꿉니다.
137 if value != value:
138 return None
139 except Exception:
140 pass
141 return str(value)

187def _frame_to_rows(frame: Any) -> list[Dict[str, Any]]:
188 """Goodocs/Pandas/list 결과를 JSON으로 직렬화 가능한 row 목록으로 바꿉니다."""
189 # Goodocs read_all() 결과는 pandas DataFrame인 경우가 많으므로 index를 먼저 정리합니다.
190 # reset_index가 실패해도 원본 데이터로 계속 변환할 수 있게 예외는 흡수합니다.
191 if hasattr(frame, "reset_index"):
192 try:
193 frame = frame.reset_index(drop=True)
194 except Exception:
195 pass
196
197 # DataFrame 단계에서 제거 가능한 시스템 컬럼은 먼저 drop합니다.
198 # list[dict] 입력도 아래 _drop_system_columns_from_rows에서 다시 한 번 정제됩니다.
199 if hasattr(frame, "drop"):
200 try:
201 drop_columns = []
202 for column in getattr(frame, "columns", []):
203 if _is_goodocs_system_column(column):
204 drop_columns.append(column)
205 if drop_columns:
206 frame = frame.drop(columns=drop_columns)
207 except Exception:
208 pass
209 if hasattr(frame, "to_dict"):
210 try:
211 # pandas DataFrame은 records 방향이 reusable flow의 list[dict] 결과와 가장 잘 맞습니다.
212 rows = frame.to_dict(orient="records")
213 except TypeError:
214 # DataFrame 유사 객체가 orient 키워드를 지원하지 않는 경우를 보조로 처리합니다.
215 rows = frame.to_dict("records")
216 elif isinstance(frame, list):
217 # 테스트 코드나 외부 Goodocs 래퍼가 이미 list[dict]로 반환하는 경우도 허용합니다.
218 rows = frame
219 else:
220 rows = []
221
222 # datetime/Decimal/NaN 등을 JSON 안전값으로 변환한 뒤 빈 행과 시스템 컬럼을 최종 제거합니다.
223 ready_rows: list[Dict[str, Any]] = []
224 for row in rows:
225 if isinstance(row, dict):
226 ready_rows.append(_json_ready(dict(row)))

342def retrieve_simple_goodocs_data(
343 data_request_value: Any,
344 goodocs_user_id: str = "",
345 goodocs_token_source: str = "",
346 goodocs_token_key: str = "",
347 fetch_limit_value: Any = "5000",
348) -> Dict[str, Any]:
349 """전체 data_request 중 source_type이 goodocs인 요청만 골라 실행합니다."""
350 # 1) 앞 노드의 wrapper를 벗기고 fetch_limit을 숫자로 정리합니다.
351 request_body = _request_body(data_request_value)
352 try:
353 fetch_limit = max(1, int(fetch_limit_value or 5000))
354 except Exception:
355 fetch_limit = 5000
356
357 # 2) 여러 source 요청 중 Goodocs 요청만 처리합니다.
358 source_requests = []
359 for item in _request_items(request_body):
360 source_type = _source_type(item.get("source_type") or item.get("source") or _source_config(item).get("source_type"))
361 if source_type == "goodocs":
362 source_requests.append(item)
363 if not source_requests:
364 return {"skipped": True, "source_type": "goodocs", "skip_reason": "No goodocs request.", "items": []}
365
366 # 3) 같은 flow에서 여러 Goodocs 문서를 조회할 수 있으므로 item별 결과를 list로 유지합니다.
367 items = []
368 for item in source_requests:
369 items.append(_run_goodocs(item, goodocs_user_id, goodocs_token_source, goodocs_token_key, fetch_limit))
370 return {"source_type": "goodocs", "items": items}
