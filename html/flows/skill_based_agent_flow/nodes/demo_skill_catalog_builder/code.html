96def _value_payload(value: Any) -> Any:
97 if value is None or value == "":
98 return deepcopy(DEFAULT_SKILL_CATALOG)
99 if isinstance(value, (dict, list)):
100 return deepcopy(value)
101 data = getattr(value, "data", None)
102 if isinstance(data, (dict, list)):
103 return deepcopy(data)
104 text = str(getattr(value, "text", value) or "").strip()
105 if not text:
106 return deepcopy(DEFAULT_SKILL_CATALOG)
107 try:
108 return json.loads(text)
109 except json.JSONDecodeError as exc:
110 raise ValueError(f"Skill 카탈로그 JSON을 해석할 수 없습니다: {exc.msg}") from exc

113def _clean_text(value: Any, field_name: str, *, max_length: int = 2000) -> str:
114 text = str(value or "").strip()
115 if not text:
116 raise ValueError(f"{field_name} 값이 비어 있습니다.")
117 if len(text) > max_length:
118 raise ValueError(f"{field_name} 값은 {max_length}자 이하여야 합니다.")
119 if re.search(r"[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]", text):
120 raise ValueError(f"{field_name} 값에 사용할 수 없는 제어 문자가 있습니다.")
121 return text

124def _clean_string_list(value: Any, field_name: str, *, required: bool = True) -> list[str]:
125 if not isinstance(value, list):
126 raise ValueError(f"{field_name} 값은 문자열 목록이어야 합니다.")
127 result: list[str] = []
128 for item in value:
129 text = _clean_text(item, field_name, max_length=1000)
130 if text not in result:
131 result.append(text)
132 if required and not result:
133 raise ValueError(f"{field_name} 목록은 한 개 이상 필요합니다.")
134 return result

137def _walk_text(value: Any):
138 if isinstance(value, str):
139 yield value
140 elif isinstance(value, dict):
141 for nested in value.values():
142 yield from _walk_text(nested)
143 elif isinstance(value, list):
144 for nested in value:
145 yield from _walk_text(nested)

148def _reject_instruction_override_markers(payload: dict[str, Any]) -> None:
149 for text in _walk_text(payload):

189def build_skill_catalog(catalog_value: Any = None) -> dict[str, Any]:
190 """입력 JSON을 검증하고 세 개의 데모 Skill만 포함한 카탈로그를 반환합니다."""
191
192 payload = _value_payload(catalog_value)
193 if not isinstance(payload, dict):
194 raise ValueError("Skill 카탈로그의 최상위 값은 JSON 객체여야 합니다.")
195 _reject_instruction_override_markers(payload)
196 if payload.get("contract") != CATALOG_CONTRACT:
197 raise ValueError(f"Skill 카탈로그 contract는 '{CATALOG_CONTRACT}'여야 합니다.")
198
199 skills = payload.get("skills")
200 if not isinstance(skills, list) or len(skills) != len(EXPECTED_TOOLS):
201 raise ValueError("Skill 카탈로그에는 데모 Skill 세 개가 정확히 필요합니다.")
202
203 normalized_skills: list[dict[str, Any]] = []
204 seen_ids: set[str] = set()
205 seen_tools: set[str] = set()
206 for index, raw_entry in enumerate(skills, start=1):
207 if not isinstance(raw_entry, dict):
208 raise ValueError(f"{index}번째 Skill 항목은 JSON 객체여야 합니다.")
209 skill_id = _clean_text(raw_entry.get("skill_id"), f"{index}번째 skill_id", max_length=64)
210 if ACTION_NAME_PATTERN.fullmatch(skill_id) is None:
211 raise ValueError(f"skill_id는 영문 소문자, 숫자, 밑줄만 사용할 수 있습니다: {skill_id}")
212 tool_name = _clean_text(raw_entry.get("tool_name"), f"{index}번째 tool_name", max_length=64)
213 if TOOL_NAME_PATTERN.fullmatch(tool_name) is None:
214 raise ValueError(f"Tool 이름은 영문자, 숫자, 밑줄로만 구성해야 합니다: {tool_name}")
215 if skill_id in seen_ids:
216 raise ValueError(f"중복된 skill_id가 있습니다: {skill_id}")
217 if tool_name in seen_tools:
218 raise ValueError(f"중복된 tool_name이 있습니다: {tool_name}")
219 if tool_name not in EXPECTED_TOOLS:
220 raise ValueError(f"허용되지 않은 Skill Tool이 있습니다: {tool_name}")
221 seen_ids.add(skill_id)
222 seen_tools.add(tool_name)
223
224 if not isinstance(raw_entry.get("enabled"), bool):
225 raise ValueError(f"{tool_name}의 enabled 값은 true 또는 false여야 합니다.")
226 allowed = _validate_actions(raw_entry.get("allowed_actions"), f"{tool_name}.allowed_actions")
227 forbidden = _validate_actions(raw_entry.get("forbidden_actions"), f"{tool_name}.forbidden_actions")
228 overlap = sorted(set(allowed) & set(forbidden))
229 if overlap:
230 raise ValueError(f"{tool_name}에서 허용과 금지 action이 겹칩니다: {', '.join(overlap)}")
231 required_action = EXPECTED_TOOLS[tool_name]
232 if required_action not in allowed:
233 raise ValueError(f"{tool_name}에는 필수 허용 action '{required_action}'이 필요합니다.")
234
235 normalized_skills.append(
236 {
237 "skill_id": skill_id,
238 "tool_name": tool_name,
239 "name": _clean_text(raw_entry.get("name"), f"{tool_name}.name", max_length=120),
240 "description": _clean_text(
241 raw_entry.get("description"), f"{tool_name}.description", max_length=1000
242 ),
243 "enabled": raw_entry["enabled"],
244 "triggers": _clean_string_list(raw_entry.get("triggers"), f"{tool_name}.triggers"),
245 "instructions": _clean_string_list(raw_entry.get("instructions"), f"{tool_name}.instructions"),
246 "allowed_actions": allowed,
247 "forbidden_actions": forbidden,
248 "rules": _validate_rules(tool_name, raw_entry.get("rules")),
249 }
250 )
251
252 missing = sorted(set(EXPECTED_TOOLS) - seen_tools)
253 if missing:
254 raise ValueError(f"필수 Skill Tool이 누락되었습니다: {', '.join(missing)}")
255 normalized_skills.sort(key=lambda item: list(EXPECTED_TOOLS).index(item["tool_name"]))
256 return {
257 "contract": CATALOG_CONTRACT,
258 "catalog_id": _clean_text(payload.get("catalog_id"), "catalog_id", max_length=64),
259 "version": _clean_text(payload.get("version"), "version", max_length=32),
260 "skills": normalized_skills,
261 "governance": {
262 "catalog_validated": True,
263 "tool_allowlist_enforced": True,
264 "instruction_override_checked": True,
265 "external_side_effects_allowed": False,
266 },
267 }
