69def _safe_scalar(value: Any, limit: int = 128) -> str:
70 text = str(value or "").strip()
71 text = re.sub(r"[\x00-\x1f\x7f]", "", text)
72 return text[:limit]

75def _safe_list(value: Any, limit: int = 32) -> list[str]:
76 if value is None:
77 return []
78 if isinstance(value, str):
79 items = re.split(r"[,;\n]", value)
80 elif isinstance(value, (list, tuple, set)):
81 items = list(value)
82 else:
83 items = [value]
84 result: list[str] = []
85 for item in items:
86 normalized = _safe_scalar(item).lower()
87 if normalized and normalized not in result:
88 result.append(normalized)
89 if len(result) >= limit:
90 break
91 return result

99def normalize_request(
100 question_value: Any,
101 trusted_value: Any,
102 *,
103 use_demo_identity: bool = True,
104 demo_identity: dict[str, Any] | None = None,
105) -> dict[str, Any]:
106 trusted = _payload(trusted_value)
107 trusted_present = bool(trusted)
108 actor_raw = (
109 _dict(trusted.get("actor_context"))
110 or _dict(trusted.get("actor"))
111 or _dict(_dict(trusted.get("request")).get("actor_context"))
112 or _dict(_dict(trusted.get("identity")).get("actor_context"))
113 )
114 identity_trust = "verified_context"
115 if not trusted_present and _bool(use_demo_identity):
116 actor_raw = {
117 **_dict(demo_identity),
118 "identity_verified": True,
119 }
120 identity_trust = "demo"
121
122 question = _text(question_value)[:8000]
123 identity_verified = _bool(
124 actor_raw.get("identity_verified", actor_raw.get("verified", trusted.get("identity_verified")))
125 )
126 user_id = _safe_scalar(actor_raw.get("user_id") or actor_raw.get("subject") or actor_raw.get("employee_id"))
127 tenant_id = _safe_scalar(actor_raw.get("tenant_id") or actor_raw.get("organization_id"))
128 roles = _safe_list(actor_raw.get("roles") or actor_raw.get("role"))
129 groups = _safe_list(actor_raw.get("groups") or actor_raw.get("group_ids") or actor_raw.get("departments"))
130 clearance = _classification(actor_raw.get("clearance") or actor_raw.get("max_classification"))
131
132 # Fail closed. A question is never allowed to supply or repair identity
133 # fields, and a partially populated trusted context does not become valid.
134 ready = bool(question and identity_verified and user_id and tenant_id)
135 supplied_request_id = _safe_scalar(
136 trusted.get("request_id") or _dict(trusted.get("request")).get("request_id"),
137 limit=96,
138 )
139 request_id = supplied_request_id or str(uuid.uuid4())

144 if not ready and question:

147 errors.append("Verified request context is unavailable.")
