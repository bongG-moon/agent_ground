18def _payload(value: Any) -> dict[str, Any]:
19 if value is None:
20 return {}
21 if isinstance(value, dict):
22 return value
23 data = getattr(value, "data", None)
24 if isinstance(data, dict):
25 return data
26 text = getattr(value, "text", None) or getattr(value, "content", None)
27 if isinstance(text, str):
28 try:
29 parsed = json.loads(text)
30 return parsed if isinstance(parsed, dict) else {}
31 except Exception:
32 return {}
33 return {}

36def _safe(value: Any, limit: int = 240) -> str:
37 text = unicodedata.normalize("NFKC", str(value or ""))
38 text = re.sub(r"[\x00-\x1f\x7f]", " ", text)
39 return " ".join(text.split())[:limit]

42def build_citation_message(answer_value: Any) -> str:
43 answer = _payload(answer_value)
44 if answer.get("supported") is not True:
45 return ABSTENTION_MESSAGE
46 answer_text = str(answer.get("answer_text") or "").strip()
47 if not answer_text:
48 return ABSTENTION_MESSAGE
49
50 evidence_by_id: dict[str, dict[str, Any]] = {}
51 for item in answer.get("evidence", []):
52 if not isinstance(item, dict):
53 continue
54 evidence_id = str(item.get("evidence_id") or "").strip()
55 source = item.get("source") if isinstance(item.get("source"), dict) else {}
56 title = _safe(source.get("title"))
57 page = _safe(source.get("page"), 40)
58 locator = _safe(source.get("locator"), 180)
59 if re.fullmatch(r"E\d+", evidence_id) and title and (page or locator):
60 evidence_by_id[evidence_id] = item
61
62 used_ids: list[str] = []
63 raw_ids = answer.get("used_evidence_ids")
64 if isinstance(raw_ids, list):
65 for value in raw_ids:
66 evidence_id = str(value or "").strip()
67 if evidence_id in evidence_by_id and evidence_id not in used_ids:
68 used_ids.append(evidence_id)
69
70 # A supported answer without at least one server-verifiable citation is
71 # unsafe. Do not return the unsupported answer text.
72 if not used_ids:
73 return ABSTENTION_MESSAGE
74
75 citation_number = {evidence_id: index for index, evidence_id in enumerate(used_ids, start=1)}
76 for evidence_id, number in citation_number.items():
77 answer_text = re.sub(rf"\[\s*{re.escape(evidence_id)}\s*\]", f"[{number}]", answer_text)
78 # Unknown model-created evidence tokens are removed. They can never create
79 # a citation or expose a source not carried by the authorized answer Data.
80 answer_text = re.sub(r"\[\s*E\d+\s*\]", "", answer_text)
81 answer_text = re.sub(r"[ \t]+\n", "\n", answer_text).strip()
82
83 if used_ids and not any(f"[{number}]" in answer_text for number in citation_number.values()):
84 answer_text += " " + " ".join(f"[{citation_number[item]}]" for item in used_ids)
85
86 citation_lines: list[str] = []
87 for evidence_id in used_ids:
88 item = evidence_by_id[evidence_id]
89 source = item.get("source") if isinstance(item.get("source"), dict) else {}
90 title = _safe(source.get("title") or "사내 문서")
91 version = _safe(source.get("version"), 80)
92 page = _safe(source.get("page"), 40)
93 locator = _safe(source.get("locator"), 180)
94 details = [title]
95 if version:
96 details.append(f"version {version}")
97 if page:
98 details.append(f"p.{page}")
99 if locator and locator.lower() not in {f"page:{page}".lower(), f"p.{page}".lower()}:
100 details.append(locator)
101 citation_lines.append(f"[{citation_number[evidence_id]}] " + " · ".join(details))
102
103 if citation_lines:
104 return answer_text + "\n\n근거\n" + "\n".join(citation_lines)
105 return answer_text

117 def build_message(self) -> Message:
118 answer = _payload(getattr(self, "answer", None))
119 text = build_citation_message(answer)
120 self.status = "citation response ready" if answer.get("used_evidence_ids") else "safe abstention ready"
121 return Message(text=text)
