180def _target_block_count(text: str, requested_blocks: list[str], complexity: str) -> int:
181 """요청 문장과 선택된 요소를 보고 목표 블록 수를 정합니다."""
182
183 explicit = _explicit_count(text)
184 if explicit:
185 # 사용자가 KPI/차트/표 같은 콘텐츠 요소를 함께 적은 경우에는
186 # 보고서 헤더와 데이터 범위 요약이 앞에서 잘라먹지 않도록 여유를 둡니다.
187 content_floor = len(requested_blocks) + 2 if requested_blocks else explicit
188 return max(3, min(max(explicit, content_floor), 10))
189 if complexity == "simple":
190 return 5
191 if complexity == "detailed":
192 return 10
193 base_count = 2 + len(requested_blocks)
194 if "insight_bullets" not in requested_blocks and _contains(text, ["\ud574\uc11d", "\uc694\uc57d", "\ub9ac\ud3ec\ud2b8", "report"]):
195 base_count += 1
196 return max(5, min(base_count, 10))

233def _extract_text(value: Any) -> str:
234 """Langflow의 다양한 데이터 객체에서 가능한 텍스트 내용을 꺼냅니다.
235
236 File Read 컴포넌트는 Langflow 버전/파일 유형에 따라 Data, Message, DataFrame 등
237 서로 다른 모양으로 값을 넘길 수 있습니다. 그래서 흔히 쓰이는 속성과 key를
238 차례로 확인하면서 CSV/JSON으로 변환 가능한 값을 찾습니다.
239 """
240
241 if value is None:
242 return ""
243 if isinstance(value, str):
244 return value
245 if hasattr(value, "to_csv"):
246 try:
247 # pandas DataFrame처럼 `to_csv`가 있으면 CSV 문자열로 바꿉니다.
248 return value.to_csv(index=False)
249 except Exception:
250 pass
251 if hasattr(value, "to_dict"):
252 try:
253 # pandas DataFrame처럼 `to_dict`가 있으면 row dict 목록으로 바꿉니다.
254 rows = value.to_dict(orient="records")
255 if isinstance(rows, list):
256 return json.dumps(rows, ensure_ascii=False, default=str)
257 except Exception:
258 pass
259 data = getattr(value, "data", None)
260 if isinstance(data, dict):
261 for key in ("text", "content", "data", "raw", "file_content", "file_contents"):
262 if isinstance(data.get(key), str):
263 return data[key]
264 for key in ("dataframe", "structured_content", "structuredContent"):
265 if data.get(key) is not None:
266 extracted = _extract_text(data.get(key))
267 if extracted:
268 return extracted
269 if isinstance(data.get("rows"), list):
270 return json.dumps(data, ensure_ascii=False, default=str)
271 if isinstance(data.get("records"), list):
272 return json.dumps(data["records"], ensure_ascii=False, default=str)
273 if isinstance(data.get("columns"), list) and isinstance(data.get("data"), list):
274 return json.dumps(data, ensure_ascii=False, default=str)
275 if isinstance(data, list):
276 return json.dumps(data, ensure_ascii=False, default=str)
277 for attr in ("text", "content"):
278 text = getattr(value, attr, None)
279 if isinstance(text, str):
280 return text
281 for attr in ("dataframe", "structured_content", "structuredContent", "value"):
282 nested = getattr(value, attr, None)
283 if nested is not None and nested is not value:
284 extracted = _extract_text(nested)
285 if extracted:
286 return extracted
287 if isinstance(value, dict):
288 return json.dumps(value, ensure_ascii=False, default=str)
289 if isinstance(value, list):
290 return json.dumps(value, ensure_ascii=False, default=str)
291 return str(value)

295 """입력 문자열을 데이터셋 목록으로 파싱합니다.

367def _try_jsonl(text: str) -> list[dict[str, Any]] | None:
368 """한 줄에 JSON 객체 하나씩 있는 JSONL 형식을 읽어 봅니다."""
369
370 rows = []

372 line = line.strip()

374 continue

378 return None
379 if not isinstance(parsed, dict):
380 return None

382 return rows if rows else None

385def _parse_csv(text: str) -> list[dict[str, Any]]:
386 """CSV 문자열을 dict row 목록으로 변환합니다."""
387
388 sample = text[:2048]
389 try:
390 # 구분자가 쉼표인지 탭인지 등을 csv.Sniffer가 샘플을 보고 추정합니다.
391 dialect = csv.Sniffer().sniff(sample)
392 except Exception:
393 dialect = csv.excel
394 reader = csv.DictReader(io.StringIO(text), dialect=dialect)
395 return [_clean_row(row) for row in reader if isinstance(row, dict)]

404def _coerce(value: Any) -> Any:
405 """문자열 값을 가능한 경우 int/float 숫자로 바꿉니다."""
406

438def _build_data_views(
439 datasets: list[dict[str, Any]],
440 question: str = "",
441 view_request: str = "",
442) -> tuple[list[dict[str, Any]], list[dict[str, Any]], dict[str, Any]]:
443 """여러 dataset을 렌더러가 사용할 수 있는 data view 목록으로 변환합니다.
444
445 - 단일 dataset이면 기존과 동일하게 해당 rows를 active view로 사용합니다.
446 - 여러 dataset이면 개별 view를 유지하면서, 공통 key가 있으면 자동 join view를 추가합니다.
447 - 같은 구조의 dataset 묶음이면 union view도 만들 수 있습니다.
448 """
449
450 normalized = [deepcopy(item) for item in datasets if isinstance(item, dict)]
451 if not normalized:
452 normalized = [_dataset("demo_dataset", "입력 데이터", [], [])]
453
454 views = [_view_from_dataset(item) for item in normalized]
455 candidates = _relationship_candidates(normalized)
456 intent_text = f"{question} {view_request}".lower()
457 union_view = _union_view(normalized)
458 join_view = _joined_view(normalized, candidates)
459
460 if union_view:
461 views.append(union_view)
462 if join_view:
463 views.append(join_view)
464
465 active = views[0]
466 if len(normalized) > 1:
467 if _contains_text(intent_text, ["union", "append", "concat", "세로", "합쳐", "누적"]) and union_view:
468 active = union_view
469 elif join_view:
470 active = join_view
471 elif union_view:
472 active = union_view
473 return views, candidates, active

495def _relationship_candidates(datasets: list[dict[str, Any]]) -> list[dict[str, Any]]:
496 """dataset 쌍 사이의 공통 key 후보를 찾습니다."""
497
498 candidates = []
499 for left_index, left in enumerate(datasets):
500 for right in datasets[left_index + 1 :]:
501 keys = _common_join_keys(left, right)
502 candidates.append(
503 {
504 "left_dataset": str(left.get("dataset_id") or ""),
505 "right_dataset": str(right.get("dataset_id") or ""),
506 "join_keys": keys,
507 "confidence": "high" if keys else "none",
508 "reason": "공통 key 후보가 있어 자동 결합할 수 있습니다." if keys else "공통 key 후보가 부족해 자동 결합하지 않습니다.",
509 }
510 )
511 return candidates

514def _common_join_keys(left: dict[str, Any], right: dict[str, Any]) -> list[str]:
515 """두 dataset 사이에서 join key로 쓰기 좋은 공통 컬럼을 고릅니다."""
516
517 left_rows = _rows(left.get("rows"))
518 right_rows = _rows(right.get("rows"))
519 common = [
520 column
521 for column in _strings(left.get("columns")) or _columns_from_rows(left_rows)
522 if column in set(_strings(right.get("columns")) or _columns_from_rows(right_rows))
523 ]
524 preferred = []
525 for column in common:
526 if _looks_like_join_key(column, left_rows, right_rows):
527 preferred.append(column)
528 if preferred:
529 return preferred[:4]
530 # 컬럼명이 같고 값이 숫자 metric처럼 보이지 않는 경우를 마지막 후보로 사용합니다.
531 fallback = [column for column in common if not _looks_numeric_metric(column, left_rows + right_rows)]
532 return fallback[:3]

535def _looks_like_join_key(column: str, left_rows: list[dict[str, Any]], right_rows: list[dict[str, Any]]) -> bool:
536 """컬럼명과 값 분포를 보고 join key 가능성을 판단합니다."""
537
538 text = str(column or "").lower()
539 if any(token in text for token in ["date", "dt", "day", "month", "week", "year", "날짜", "일자", "기준일"]):
540 return True
541 if any(token in text for token in ["process", "oper", "line", "product", "item", "category", "shift", "region", "warehouse", "공정", "라인", "제품", "구분"]):
542 return True
543 if any(token in text for token in ["id", "code", "key", "no", "seq", "코드", "번호"]):
544 return True
545 return not _looks_numeric_metric(column, left_rows + right_rows)

548def _looks_numeric_metric(column: str, rows: list[dict[str, Any]]) -> bool:
549 """값 대부분이 숫자이고 key 이름이 아니면 metric 컬럼으로 봅니다."""
550
551 values = [row.get(column) for row in rows if row.get(column) not in (None, "")]
552 if not values:
553 return False
554 if any(token in str(column).lower() for token in ["id", "code", "no", "seq", "date", "dt"]):
555 return False
556 numeric_count = sum(1 for value in values if isinstance(value, (int, float)) and not isinstance(value, bool))
557 return numeric_count / max(len(values), 1) >= 0.8

560def _joined_view(datasets: list[dict[str, Any]], candidates: list[dict[str, Any]]) -> dict[str, Any] | None:
561 """여러 dataset을 공통 key 기준으로 outer join한 view를 만듭니다."""
562
563 if len(datasets) < 2:
564 return None
565 common_keys = set(_strings(candidates[0].get("join_keys")) if candidates else [])
566 for candidate in candidates[1:]:
567 common_keys &= set(_strings(candidate.get("join_keys")))
568 join_keys = [column for column in (_strings(datasets[0].get("columns")) or _columns_from_rows(_rows(datasets[0].get("rows")))) if column in common_keys]
569 if not join_keys:
570 return None
571
572 joined_by_key: dict[tuple[str, ...], dict[str, Any]] = {}
573 output_columns = list(join_keys)
574 source_ids = [str(item.get("dataset_id") or f"dataset_{index}") for index, item in enumerate(datasets, start=1)]
575 non_key_counts: dict[str, int] = {}
576 for dataset in datasets:
577 for column in _strings(dataset.get("columns")) or _columns_from_rows(_rows(dataset.get("rows"))):
578 if column not in join_keys:
579 non_key_counts[column] = non_key_counts.get(column, 0) + 1
580
581 for dataset in datasets:
582 dataset_id = str(dataset.get("dataset_id") or "")
583 rows = _rows(dataset.get("rows"))
584 for row in rows:
585 key = tuple(str(row.get(column, "")) for column in join_keys)
586 target = joined_by_key.setdefault(key, {column: row.get(column, "") for column in join_keys})
587 for column, value in row.items():
588 if column in join_keys:
589 continue
590 output_name = _merged_column_name(column, dataset_id, non_key_counts.get(column, 0) > 1, output_columns)
591 target[output_name] = value
592 if output_name not in output_columns:
593 output_columns.append(output_name)
594
595 rows = list(joined_by_key.values())
596 label = " + ".join(str(item.get("label") or item.get("dataset_id") or "") for item in datasets[:3])
597 return {
598 "data_view_id": "joined_auto",
599 "label": f"자동 결합 데이터 ({label})",
600 "strategy": "join",
601 "source_dataset_ids": source_ids,
602 "join_keys": join_keys,
603 "columns": output_columns,
604 "rows": rows,
605 "row_count": len(rows),
606 "data_ref": {},
607 }
608

610def _merged_column_name(column: str, dataset_id: str, duplicated: bool, output_columns: list[str]) -> str:
611 """join 시 같은 이름의 metric이 충돌하면 dataset_id suffix를 붙입니다."""
612
613 if not duplicated:
614 return column
615 suffix = "".join(ch if ch.isalnum() else "_" for ch in str(dataset_id or "dataset")).strip("_") or "dataset"
616 return f"{column}__{suffix}"

619def _union_view(datasets: list[dict[str, Any]]) -> dict[str, Any] | None:
620 """같은 구조의 여러 dataset을 세로로 붙인 union view를 만듭니다."""
621
622 if len(datasets) < 2:
623 return None
624 column_sets = [set(_strings(item.get("columns")) or _columns_from_rows(_rows(item.get("rows")))) for item in datasets]
625 if not column_sets or len({tuple(sorted(cols)) for cols in column_sets}) != 1:
626 return None
627 columns = list(_strings(datasets[0].get("columns")) or _columns_from_rows(_rows(datasets[0].get("rows"))))
628 output_columns = ["__dataset_id", "__dataset_label", *columns]
629 rows = []
630 for dataset in datasets:
631 dataset_id = str(dataset.get("dataset_id") or "")
632 label = str(dataset.get("label") or dataset_id)
633 for row in _rows(dataset.get("rows")):
634 rows.append({"__dataset_id": dataset_id, "__dataset_label": label, **{column: row.get(column, "") for column in columns}})
635 return {
636 "data_view_id": "union_auto",
637 "label": "자동 통합 데이터",
638 "strategy": "union",
639 "source_dataset_ids": [str(item.get("dataset_id") or "") for item in datasets],
640 "join_keys": [],
641 "columns": output_columns,
642 "rows": rows,
643 "row_count": len(rows),
644 "data_ref": {},
645 }
