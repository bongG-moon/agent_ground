78 """Validate the LLM output while preserving the user's interpreted intent."""
79

82 # but this also accepts {"request_interpretation": ..., "report_plan": {...}}.

276def _normalize_request_interpretation(raw: dict[str, Any]) -> dict[str, Any]:
277 """Keep the LLM's request understanding in a compact, renderer-safe form."""
278
279 if not raw:
280 return {}
281 result: dict[str, Any] = {}
282 for key, limit in (
283 ("user_goal", 240),
284 ("data_focus", 200),
285 ("layout_intent", 240),
286 ("style_intent", 200),
287 ):
288 value = str(raw.get(key) or "").strip()
289 if value:
290 result[key] = value[:limit]
291 for key in ("requested_visuals", "requested_blocks", "requested_order", "requested_columns", "data_binding_plan"):
292 values = _normalize_string_list(raw.get(key), 12, 80)
293 if values:
294 result[key] = values
295 value_conditions = []
296 for item in _list(raw.get("requested_value_conditions")):
297 if isinstance(item, dict):
298 column = str(item.get("column") or "").strip()
299 operator = str(item.get("operator") or "").strip()
300 value = item.get("value")
301 purpose = str(item.get("purpose") or "").strip()
302 compact = {key: val for key, val in {"column": column, "operator": operator, "value": value, "purpose": purpose}.items() if val not in ("", None, [], {})}
303 if compact:
304 value_conditions.append(compact)
305 else:
306 text = str(item or "").strip()
307 if text:
308 value_conditions.append({"text": text[:160]})
309 if len(value_conditions) >= 12:
310 break
311 if value_conditions:
312 result["requested_value_conditions"] = value_conditions
313 unmet = []
314 for item in _list(raw.get("unmet_requests")):
315 if isinstance(item, dict):
316 request = str(item.get("request") or item.get("item") or item.get("label") or "").strip()
317 reason = str(item.get("reason") or item.get("why") or item.get("value") or "").strip()
318 text = ": ".join(part for part in (request, reason) if part)
319 else:
320 text = str(item or "").strip()
321 if text and text not in unmet:
322 unmet.append(text[:220])
323 if len(unmet) >= 8:
324 break
325 if unmet:
326 result["unmet_requests"] = unmet
327 try:
328 target = int(raw.get("target_block_count") or 0)
329 except Exception:
330 target = 0
331 if target:
332 result["target_block_count"] = max(1, min(target, 12))
333 return result

630def _extract_json_object(text: str) -> dict[str, Any]:
631 """LLM 응답 텍스트에서 JSON 객체를 찾아 파싱합니다.
632
633 LLM이 ```json 코드블록으로 감싸거나 앞뒤 설명을 붙이는 경우도 있어,
634 여러 후보 문자열을 순서대로 시도합니다.
635 """
636
637 raw = str(text or "").strip()
638 if not raw:
639 return {}
640 fenced = re.search(r"```(?:json)?\s*(\{.*?\})\s*```", raw, flags=re.S | re.I)
641 candidates = [fenced.group(1)] if fenced else []
642 candidates.append(raw)
643 start = raw.find("{")
644 end = raw.rfind("}")
645 if start >= 0 and end > start:
646 candidates.append(raw[start : end + 1])
647 for candidate in candidates:
648 try:
649 parsed = json.loads(candidate)
650 except Exception:
651 continue
652 if isinstance(parsed, dict):
653 return parsed
654 return {}

657def _text(value: Any) -> str:
658 """Langflow Message/Data/dict 등에서 LLM 응답 텍스트를 꺼냅니다."""
659
660 if value is None:
661 return ""
662 if isinstance(value, str):
663 return value
664 data = getattr(value, "data", None)
665 if isinstance(data, dict):
666 for key in ("llm_text", "text", "content", "response", "message"):
667 if isinstance(data.get(key), str):
668 return data[key]
669 for attr in ("text", "content"):
670 text = getattr(value, attr, None)
671 if isinstance(text, str):
672 return text
673 if isinstance(value, dict):
674 return json.dumps(value, ensure_ascii=False, default=str)
675 return str(value)
