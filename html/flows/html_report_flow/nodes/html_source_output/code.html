29def _payload(value: Any) -> dict[str, Any]:
30 """Langflow Data/Message/dict/JSON 문자열을 일반 dict로 맞춥니다."""
31
32 if isinstance(value, dict):
33 return deepcopy(value)
34 data = getattr(value, "data", None)
35 if isinstance(data, dict):
36 return deepcopy(data)
37 text = getattr(value, "text", None) or getattr(value, "content", None)
38 if isinstance(text, str) and text.strip():
39 try:
40 parsed = json.loads(text)
41 except Exception:
42 return {"text": text}
43 return deepcopy(parsed) if isinstance(parsed, dict) else {"text": text}
44 return {}
