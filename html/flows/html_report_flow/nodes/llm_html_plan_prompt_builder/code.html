159def _component_contracts(catalog: dict[str, Any]) -> list[dict[str, Any]]:
160 """카탈로그를 LLM이 읽기 쉬운 컴포넌트 계약 목록으로 줄입니다."""
161
162 components = _list(catalog.get("components"))
163 recommended = {
164 str(item.get("component_id")): item
165 for item in _list(catalog.get("recommended_components"))
166 if isinstance(item, dict)
167 }
168 result = []
169 for item in components:
170 if not isinstance(item, dict):
171 continue
172 component_id = str(item.get("component_id") or "")
173 rec = _dict(recommended.get(component_id))
174 result.append(
175 _compact_dict(
176 {
177 "component_id": component_id,
178 "display_name": item.get("display_name") or component_id,
179 "description": item.get("description") or "",
180 "layout_role": item.get("layout_role") or "",
181 "data_contract": _dict(item.get("data_contract")),
182 "recommended": bool(rec),
183 "suggested_bindings": _dict(rec.get("suggested_bindings")),
184 "reason": rec.get("reason") or "",
185 "template_guidance": item.get("template_guidance") or item.get("usage_guidance") or "",
186 "style_defaults": _dict(item.get("style_defaults")),
187 "default_spec": _dict(item.get("default_spec")),
188 "example_spec": _dict(item.get("example_spec")),
189 }
190 )
191 )

455def _payload(value: Any) -> dict[str, Any]:
456 """Langflow Data/Message/dict/JSON 문자열을 일반 dict로 맞춥니다."""
457
458 if isinstance(value, dict):
459 return deepcopy(value)
460 data = getattr(value, "data", None)
461 if isinstance(data, dict):
462 return deepcopy(data)
463 text = getattr(value, "text", None) or getattr(value, "content", None)
464 if isinstance(text, str) and text.strip():
465 try:
466 parsed = json.loads(text)
467 except Exception:
468 return {"text": text}
469 return deepcopy(parsed) if isinstance(parsed, dict) else {"text": text}
470 return {}

480 """list면 복사본을, 아니면 빈 list를 반환합니다."""

485def _rows(value: Any) -> list[dict[str, Any]]:
