119def _prioritize_recommendations(recommendations: list[Any], preferred_ids: list[str]) -> list[dict[str, Any]]:
120 """00번 표시 요구에서 직접 요청한 요소를 추천 목록 앞쪽으로 옮깁니다."""
121
122 pinned_ids = {"report_header", "scope_summary"}
123 by_id: dict[str, dict[str, Any]] = {}
124 original_order: list[str] = []
125
126 for item in recommendations:
127 if not isinstance(item, dict):
128 continue
129 component_id = str(item.get("component_id") or "").strip()
130 if not component_id:
131 continue
132 if component_id not in by_id:
133 by_id[component_id] = item
134 original_order.append(component_id)
135
136 result: list[dict[str, Any]] = []
137
138 def append_component(component_id: str) -> None:
139 if not component_id or any(existing.get("component_id") == component_id for existing in result):
140 return
141 result.append(deepcopy(by_id.get(component_id) or {"component_id": component_id}))
142
143 for component_id in original_order:
144 if component_id in pinned_ids:
145 append_component(component_id)
146 for component_id in preferred_ids:
147 if component_id not in pinned_ids:
148 append_component(component_id)
149 for component_id in original_order:
150 append_component(component_id)
151 return result

281def _resolve_max_blocks(max_blocks: Any, request: dict[str, Any], profile: dict[str, Any]) -> tuple[int, str]:
282 """수동 블록 수 대신 00번의 표시 요구를 우선 사용해 블록 수 제한을 정합니다."""
283
284 raw = str(max_blocks or "auto").strip().lower()
285 if raw not in {"", "auto", "\uc790\ub3d9"}:
286 return max(3, min(_positive_int(max_blocks, 8), 10)), "manual"
287
288 visual_request = _dict(request.get("visual_request"))
289 try:
290 target = int(visual_request.get("target_block_count") or 0)
291 except Exception:
292 target = 0
293 if target:
294 return max(3, min(target, 10)), "visual_request"
295
296 text = " ".join(
297 str(item or "")
298 for item in (
299 request.get("question"),
300 request.get("view_request"),
301 profile.get("intent_text"),
302 )
303 ).lower()
304 if _contains(text, ["\uac04\ub2e8", "\uc9e7\uac8c", "simple", "brief"]):
305 return 5, "auto_simple"
306 if _contains(text, ["\uc0c1\uc138", "\ud48d\ubd80", "detailed", "full"]):
307 return 10, "auto_detailed"
308 return 8, "auto_default"
