151def parse_find_item_response(xml_text: str, keyword: str = "", max_count: int = 10) -> list[dict[str, Any]]:
152 root = _parse_xml(xml_text, "FindItem")
153 selected: list[dict[str, Any]] = []
154 needle = keyword.strip().casefold()
155 for position, message in enumerate(root.findall(".//t:Items/t:Message", NS), 1):
156 item_id_node = message.find("t:ItemId", NS)
157 if item_id_node is None or not item_id_node.get("Id"):
158 continue
159 subject = _text(message, "t:Subject")
160 if needle and needle not in subject.casefold():
161 continue
162 selected.append(
163 {
164 "item_id": item_id_node.get("Id", ""),
165 "change_key": item_id_node.get("ChangeKey", ""),
166 "mail_index": len(selected) + 1,
167 "subject": subject,
168 "sender": _text(message, "t:Sender/t:Mailbox/t:Name"),
169 "sender_email": _text(message, "t:Sender/t:Mailbox/t:EmailAddress"),
170 "received_time": _text(message, "t:DateTimeReceived"),
171 "is_read": _bool_text(_text(message, "t:IsRead")),
172 "has_attachments": _bool_text(_text(message, "t:HasAttachments")),
173 "find_position": position,
174 }
175 )
176 if len(selected) >= max_count:
177 break
