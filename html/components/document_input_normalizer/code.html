96def _as_bool(value: Any, default: bool = False) -> bool:
97 if isinstance(value, bool):
98 return value
99 text = str(value or "").strip().lower()
100 if text in {"1", "true", "yes", "y", "on"}:
101 return True
102 if text in {"0", "false", "no", "n", "off"}:
103 return False
104 return default

115def _string_list(value: Any) -> list[str]:
116 if value is None:
117 return []
118 if isinstance(value, str):
119 items = re.split(r"[,;\n]", value)
120 elif isinstance(value, (list, tuple, set)):
121 items = list(value)
122 else:
123 items = [value]
124
125 result: list[str] = []
126 for item in items:
127 text = str(item or "").strip().lower()
128 if text and text not in result:
129 result.append(text)
130 return result

145def _payload_from_value(value: Any) -> Any:
146 """Convert Langflow Data/Message, JSON text, dict, or list to a plain value."""
147 if value is None:
148 return None
149 if isinstance(value, (dict, list)):
150 return value
151
152 data = getattr(value, "data", None)
153 if isinstance(data, (dict, list)):
154 return data
155
156 text = _text_value(value).strip()
157 if not text:
158 return None
159
160 candidate = text
161 if candidate.startswith("```") and candidate.endswith("```"):
162 candidate = re.sub(r"^```(?:json)?\s*", "", candidate, flags=re.IGNORECASE)
163 candidate = re.sub(r"\s*```$", "", candidate)
164 try:
165 return json.loads(candidate)
166 except Exception:
167 return {"content": text}

170def _document_items(payload: Any) -> list[Any]:
171 if isinstance(payload, list):
172 return payload
173 if not isinstance(payload, dict):
174 return []
175
176 for key in ("documents", "rows", "items"):
177 items = payload.get(key)
178 if isinstance(items, list):
179 return items
180 if isinstance(items, dict):
181 return [items]
182
183 nested = payload.get("data")
184 if isinstance(nested, list):
185 return nested
186 if isinstance(nested, dict) and nested is not payload:
187 nested_items = _document_items(nested)
188 if nested_items:
189 return nested_items
190
191 if any(key in payload for key in ("content", "text", "page_content", "deleted", "active")):
192 return [payload]
193 return []

208def _stable_document_id(item: dict[str, Any], content: str, position: int) -> str:
209 supplied = _safe_text(

216 identity = "|".join(

236def _expand_pages(item: dict[str, Any], page_break_marker: str) -> list[dict[str, Any]]:
237 pages = item.get("pages")
238 if isinstance(pages, list) and pages:
239 expanded: list[dict[str, Any]] = []
240 for index, page_item in enumerate(pages, start=1):
241 copied = dict(item)
242 copied.pop("pages", None)
243 if isinstance(page_item, dict):
244 copied.update(page_item)
245 else:
246 copied["content"] = str(page_item or "")
247 copied.setdefault("page", index)
248 expanded.append(copied)
249 return expanded
250
251 content = _content_from_item(item)
252 marker = str(page_break_marker or "").strip()
253 if marker and marker in content and _page_number(item.get("page")) is None:
254 expanded = []
255 for index, part in enumerate(content.split(marker), start=1):
256 if not part.strip():
257 continue
258 copied = dict(item)
259 copied["content"] = part.strip()
260 copied["page"] = index
261 copied["source_locator"] = f"page:{index}"
262 expanded.append(copied)
263 return expanded
264 return [item]

267def normalize_document_input(
268 value: Any,
269 *,
270 use_demo_corpus: bool = True,
271 default_source_name: str = "enterprise_demo_corpus",
272 default_tenant_id: str = "agent-ground",
273 default_classification: str = "internal",
274 default_allowed_roles: Any = "employee",
275 default_allowed_groups: Any = "all-employees",
276 page_break_marker: str = DEFAULT_PAGE_BREAK_MARKER,
277 max_documents: int = 100,
278 max_chars_per_document: int = 200_000,
279) -> dict[str, Any]:
280 """Normalize document-like input into the public ingestion contract.
281
282 The function is intentionally independent of Langflow classes so contract
283 tests can call it directly.
284 """
285 max_documents = _clamp_int(max_documents, 100, 1, 2_000)
286 max_chars_per_document = _clamp_int(max_chars_per_document, 200_000, 1_000, 2_000_000)
287 classification_default = str(default_classification or "internal").strip().lower()
288 if classification_default not in VALID_CLASSIFICATIONS:
289 classification_default = "internal"
290
291 payload = _payload_from_value(value)
292 raw_items = _document_items(payload)
293 demo_mode = False
294 if not raw_items and _as_bool(use_demo_corpus, default=True):
295 raw_items = [dict(item) for item in DEMO_DOCUMENTS]
296 demo_mode = True
297
298 errors: list[str] = []
299 warnings: list[str] = []
300 documents: list[dict[str, Any]] = []
301 rejected_count = 0
302 truncated_count = 0
303
304 if not raw_items:
305 errors.append("No document input was provided and the demo corpus is disabled.")
306
307 if len(raw_items) > max_documents:
308 raw_items = raw_items[:max_documents]
309 warnings.append("Document count exceeded the configured limit; remaining items were not processed.")
310
311 expanded_items: list[dict[str, Any]] = []
312 for raw_item in raw_items:
313 if isinstance(raw_item, str):
314 raw_item = {"content": raw_item}
315 if not isinstance(raw_item, dict):
316 rejected_count += 1
317 continue
318 expanded_items.extend(_expand_pages(dict(raw_item), page_break_marker))
319
320 if len(expanded_items) > max_documents:
321 expanded_items = expanded_items[:max_documents]
322 warnings.append("Expanded page count exceeded the configured document limit.")
323
324 default_roles = _string_list(default_allowed_roles)
325 default_groups = _string_list(default_allowed_groups)
326 for position, item in enumerate(expanded_items, start=1):
327 content = _content_from_item(item)
328 deleted = _as_bool(item.get("deleted"), default=False)
329 active = _as_bool(item.get("active"), default=not deleted)
330
331 # Tombstones are allowed to omit content; active documents are not.
332 if not content and not deleted and active:
333 rejected_count += 1
334 continue
335 if len(content) > max_chars_per_document:
336 content = content[:max_chars_per_document]
337 truncated_count += 1
338
339 document_id = _stable_document_id(item, content, position)
340 page = _page_number(item.get("page") or item.get("page_number"))
341 source_locator = _safe_text(item.get("source_locator"), 1_000)
342 if not source_locator:
343 source_locator = f"page:{page}" if page is not None else f"document:{document_id}"
344
345 classification = str(item.get("classification") or classification_default).strip().lower()
346 if classification not in VALID_CLASSIFICATIONS:
347 classification = classification_default
348 warnings.append("A document classification was invalid and was replaced by the configured default.")
349
350 roles = _string_list(item.get("allowed_roles"))
351 groups = _string_list(item.get("allowed_groups"))
352 if classification != "public":
353 if not roles:
354 roles = list(default_roles)
355 if not groups:
356 groups = list(default_groups)
357
358 source_name = _safe_text(
359 item.get("source_name") or item.get("file_name") or default_source_name,
360 300,
361 )
362 title = _safe_text(item.get("title") or item.get("name") or source_name or document_id, 500)
363 version = _safe_text(item.get("version") or item.get("document_version") or "1", 100)
364 content_hash = hashlib.sha256(content.encode("utf-8")).hexdigest()
365
366 documents.append(
367 {
368 "document_id": document_id,
369 "title": title,
370 "version": version,
371 "content": content,
372 "content_hash": content_hash,
373 "page": page,
374 "source_locator": source_locator,
375 "source_url": _safe_text(item.get("source_url") or item.get("url"), 2_000),
376 "source_name": source_name,
377 "tenant_id": _safe_text(item.get("tenant_id") or default_tenant_id, 200).lower(),
378 "visibility": _safe_text(item.get("visibility") or classification, 100).lower(),
379 "classification": classification,
380 "allowed_roles": roles,
381 "allowed_groups": groups,
382 "active": bool(active and not deleted),
383 "deleted": deleted,
384 }
385 )
386
387 if rejected_count:
388 warnings.append("One or more document items were rejected because their shape or content was invalid.")
389 if truncated_count:
390 warnings.append("One or more documents exceeded the character limit and were truncated.")
391 if not documents and not errors:
392 errors.append("No valid documents remained after normalization.")
393
394 success = bool(documents) and not errors
395 return {
396 "success": success,
397 "stage": "document_input_normalizer",
398 "documents": documents,
399 "document_count": len(documents),
400 "demo_mode": demo_mode,
401 "errors": errors,
402 "warnings": warnings,
403 "trace": [
404 {
405 "stage": "document_input_normalizer",
406 "status": "success" if success else "failed",
407 "document_count": len(documents),
408 "rejected_count": rejected_count,
409 "truncated_count": truncated_count,
410 "demo_mode": demo_mode,
411 }
412 ],
413 }
