24def _text(value: Any) -> str:
25 if value is None:
26 return ""
27 if isinstance(value, str):
28 return value.strip()
29 for attribute in ("text", "content"):
30 candidate = getattr(value, attribute, None)
31 if isinstance(candidate, str):
32 return candidate.strip()
33 return str(value).strip()

36def _payload(value: Any) -> dict[str, Any]:
37 if isinstance(value, dict):

39 data = getattr(value, "data", None)

61def _catalog_entry(skill_catalog: Any) -> tuple[dict[str, Any] | None, str]:
62 catalog = _payload(skill_catalog)
63 if catalog.get("contract") != CATALOG_CONTRACT:
64 return None, "검증된 데모 Skill 카탈로그가 연결되지 않았습니다."
65 skills = catalog.get("skills")
66 if not isinstance(skills, list):
67 return None, "Skill 카탈로그의 skills 목록을 확인할 수 없습니다."
68 matches = [entry for entry in skills if isinstance(entry, dict) and entry.get("tool_name") == TOOL_NAME]
69 if len(matches) != 1:
70 return None, f"카탈로그에서 {TOOL_NAME} 항목을 정확히 하나 확인해야 합니다."
71 entry = matches[0]
72 if entry.get("enabled") is not True:
73 return None, "휴가 정책 점검 Skill이 카탈로그에서 사용 중지되어 있습니다."
74 allowed = entry.get("allowed_actions")
75 forbidden = entry.get("forbidden_actions")
76 if not isinstance(allowed, list) or REQUIRED_ACTION not in allowed:
77 return None, f"카탈로그에 필수 허용 action '{REQUIRED_ACTION}'이 없습니다."
78 if not isinstance(forbidden, list) or set(allowed) & set(forbidden):
79 return None, "카탈로그의 허용 action과 금지 action 정책이 올바르지 않습니다."
80 return entry, ""

91def _parse_iso_date(value: str, label: str) -> date:
92 try:
93 parsed = date.fromisoformat(value)
94 except ValueError as exc:
95 raise ValueError(f"{label} '{value}'은 유효한 ISO 날짜가 아닙니다.") from exc
96 if parsed.isoformat() != value:
97 raise ValueError(f"{label}은 YYYY-MM-DD 형식이어야 합니다.")
98 return parsed

101def _parse_holidays(value: Any) -> list[date]:
102 if value is None or value == "":
103 return []
104 data = getattr(value, "data", None)
105 if data is not None:
106 value = data
107 if isinstance(value, str):
108 text = value.strip()
109 if not text:
110 return []
111 try:
112 value = json.loads(text)
113 except json.JSONDecodeError as exc:
114 raise ValueError(f"휴일 JSON을 해석할 수 없습니다: {exc.msg}") from exc
115 if isinstance(value, dict):
116 value = value.get("holiday_dates", value.get("holidays"))
117 if not isinstance(value, list):
118 raise ValueError("휴일 JSON은 ISO 날짜 문자열 목록이어야 합니다.")
119 holidays: list[date] = []
120 for index, item in enumerate(value, start=1):
121 holiday = _parse_iso_date(str(item or "").strip(), f"{index}번째 휴일")
122 if holiday not in holidays:
123 holidays.append(holiday)
124 return sorted(holidays)

143def run_leave_policy(request: Any, skill_catalog: Any, holiday_dates_json: Any = "") -> dict[str, Any]:
144 """요청의 ISO 날짜 두 개를 포함 범위로 계산하고 평일·휴일을 구분합니다."""
145
146 request_text = _text(request)
147 entry, catalog_error = _catalog_entry(skill_catalog)
148 if entry is None:
149 return _blocked_result(request_text, catalog_error)
150 maximum, rule_error = _demo_maximum(entry)
151 if maximum is None:
152 return _blocked_result(request_text, rule_error)
153
154 skill = {
155 "skill_id": str(entry.get("skill_id") or "leave_policy"),
156 "tool_name": TOOL_NAME,
157 "name": str(entry.get("name") or "휴가 정책 점검"),
158 }
159 governance = {
160 "authorized": True,
161 "required_action": REQUIRED_ACTION,
162 "catalog_enabled": True,
163 "external_write_performed": False,
164 "external_send_performed": False,
165 "decision_effect": "advisory_only",
166 }
167
168 date_tokens = ISO_DATE_PATTERN.findall(request_text)
169 if len(date_tokens) != 2:
170 return {
171 "status": "needs_input",
172 "skill": skill,
173 "result": {
174 "executed": False,
175 "reason": "시작일과 종료일에 해당하는 ISO 날짜를 정확히 두 개 입력해 주세요.",
176 "detected_iso_date_count": len(date_tokens),
177 },
178 "governance": governance,
179 "trace": _request_trace(request_text),
180 "disclaimer": DISCLAIMER,
181 }
182 try:
183 start_date = _parse_iso_date(date_tokens[0], "시작일")
184 end_date = _parse_iso_date(date_tokens[1], "종료일")
185 holidays = _parse_holidays(holiday_dates_json)
186 except ValueError as exc:
187 return {
188 "status": "needs_input",
189 "skill": skill,
190 "result": {"executed": False, "reason": str(exc)},
191 "governance": governance,
192 "trace": _request_trace(request_text),
193 "disclaimer": DISCLAIMER,
194 }
195 if end_date < start_date:
196 return {
197 "status": "needs_input",
198 "skill": skill,
199 "result": {"executed": False, "reason": "종료일은 시작일과 같거나 이후여야 합니다."},
200 "governance": governance,
201 "trace": _request_trace(request_text),
202 "disclaimer": DISCLAIMER,
203 }
204
205 cursor = start_date
206 weekdays: list[date] = []
207 weekend_days = 0
208 while cursor <= end_date:
209 if cursor.weekday() < 5:
210 weekdays.append(cursor)
211 else:
212 weekend_days += 1
213 cursor += timedelta(days=1)
214
215 excluded_holidays = [holiday for holiday in holidays if start_date <= holiday <= end_date and holiday.weekday() < 5]
216 ignored_holidays = [holiday for holiday in holidays if holiday not in excluded_holidays]
217 chargeable_days = len(weekdays) - len(excluded_holidays)
218 needs_review = chargeable_days > maximum
219 return {
220 "status": "completed",
221 "skill": skill,
222 "result": {
223 "executed": True,
224 "start_date": start_date.isoformat(),
225 "end_date": end_date.isoformat(),
226 "calendar_days_inclusive": (end_date - start_date).days + 1,
227 "weekday_count": len(weekdays),
228 "weekend_day_count": weekend_days,
229 "excluded_holidays": [value.isoformat() for value in excluded_holidays],
230 "ignored_holidays": [value.isoformat() for value in ignored_holidays],
231 "chargeable_days": chargeable_days,
232 "demo_max_chargeable_weekdays": maximum,
233 "policy_result": "담당자 검토 필요" if needs_review else "데모 기준 내",
234 "decision_code": "manual_review_required" if needs_review else "within_demo_guideline",
235 },
236 "governance": governance,
237 "trace": _request_trace(request_text),
238 "disclaimer": DISCLAIMER,
239 }
