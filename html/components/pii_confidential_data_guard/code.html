21def _as_bool(value: Any, default: bool = False) -> bool:
22 if isinstance(value, bool):
23 return value
24 text = str(value or "").strip().lower()
25 if text in {"1", "true", "yes", "y", "on"}:
26 return True
27 if text in {"0", "false", "no", "n", "off"}:
28 return False
29 return default

32def _payload_from_value(value: Any) -> dict[str, Any]:
33 if value is None:
34 return {}
35 if isinstance(value, dict):
36 return value
37 if isinstance(value, list):
38 return {"documents": value}
39
40 data = getattr(value, "data", None)
41 if isinstance(data, dict):
42 return data
43 if isinstance(data, list):
44 return {"documents": data}
45
46 text = getattr(value, "text", None) or getattr(value, "content", None)
47 if isinstance(value, str):
48 text = value
49 if isinstance(text, str) and text.strip():
50 try:
51 parsed = json.loads(text)
52 except Exception:
53 return {}
54 if isinstance(parsed, dict):
55 return parsed
56 if isinstance(parsed, list):
57 return {"documents": parsed}
58 return {}

61def _pattern_specs(
62 *,
63 detect_emails: bool,
64 detect_phone_numbers: bool,
65 detect_national_ids: bool,
66 detect_employee_ids: bool,
67 detect_secrets: bool,
68) -> list[tuple[str, re.Pattern[str], str | Callable[[re.Match[str]], str]]]:

81 if detect_national_ids:
82 specs.append(

84 "national_id",

87 )
88 )
89 if detect_emails:
90 specs.append(
91 (
92 "email",
93 re.compile(r"(?i)(?<![\w.+-])[\w.+-]+@[a-z0-9-]+(?:\.[a-z0-9-]+)+(?![\w.-])"),

95 )
96 )
97 if detect_phone_numbers:
98 specs.append(
99 (
100 "phone",
101 re.compile(r"(?<!\d)(?:(?:\+?82[- ]?)?0?1[016789]|0\d{1,2})[- ]?\d{3,4}[- ]?\d{4}(?!\d)"),
102 "[REDACTED:PHONE]",
103 )
104 )
105 if detect_employee_ids:
106 specs.append(

108 "employee_id",

110 "[REDACTED:EMPLOYEE_ID]",

116def _scan_and_redact(
117 text: Any,
118 specs: list[tuple[str, re.Pattern[str], str | Callable[[re.Match[str]], str]]],
119) -> tuple[str, dict[str, int]]:
120 safe_text = str(text or "")
121 counts: dict[str, int] = {}
122 for name, pattern, replacement in specs:
123 matches = list(pattern.finditer(safe_text))
124 if not matches:
125 continue
126 counts[name] = counts.get(name, 0) + len(matches)
127 safe_text = pattern.sub(replacement, safe_text)
128 return safe_text, counts

131def _merge_counts(target: dict[str, int], incoming: dict[str, int]) -> None:
132 for name, count in incoming.items():
133 target[name] = target.get(name, 0) + int(count)

136def _safe_document_id(
137 value: Any,
138 specs: list[tuple[str, re.Pattern[str], str | Callable[[re.Match[str]], str]]],
139) -> tuple[str, dict[str, int]]:
140 original = str(value or "").strip()
141 redacted, counts = _scan_and_redact(original, specs)
142 if counts:
143 # A redaction token is not a useful unique ID. Hash the original value
144 # and expose only the non-reversible short identifier.
145 return "protected-doc-" + hashlib.sha256(original.encode("utf-8")).hexdigest()[:20], counts
146 return redacted, counts

149def _upgrade_classification(value: Any) -> str:
150 classification = str(value or "internal").strip().lower()
151 if classification not in CLASSIFICATION_ORDER:
152 classification = "internal"
153 if CLASSIFICATION_ORDER[classification] < CLASSIFICATION_ORDER["confidential"]:
154 return "confidential"
155 return classification

158def guard_documents(
159 value: Any,
160 *,
161 guard_mode: str = "redact",
162 detect_emails: bool = True,
163 detect_phone_numbers: bool = True,
164 detect_national_ids: bool = True,
165 detect_employee_ids: bool = True,
166 detect_secrets: bool = True,
167 upgrade_classification: bool = True,
168) -> dict[str, Any]:
169 """Redact or block sensitive patterns without returning their raw values."""
170 payload = _payload_from_value(value)
171 documents_value = payload.get("documents")
172 documents = documents_value if isinstance(documents_value, list) else []
173
174 selected_mode = str(guard_mode or "redact").strip().lower()
175 warnings: list[str] = []
176 errors: list[str] = []
177 if selected_mode not in {"redact", "block"}:
178 selected_mode = "redact"
179 warnings.append("Guard mode was invalid and was changed to redact.")
180
181 specs = _pattern_specs(
182 detect_emails=_as_bool(detect_emails, True),
183 detect_phone_numbers=_as_bool(detect_phone_numbers, True),
184 detect_national_ids=_as_bool(detect_national_ids, True),
185 detect_employee_ids=_as_bool(detect_employee_ids, True),
186 detect_secrets=_as_bool(detect_secrets, True),
187 )
188
189 safe_documents: list[dict[str, Any]] = []
190 blocked_document_ids: list[str] = []
191 match_counts: dict[str, int] = {}
192 affected_document_count = 0
193 invalid_document_count = 0
194
195 for raw_document in documents:
196 if not isinstance(raw_document, dict):
197 invalid_document_count += 1
198 continue
199
200 document = dict(raw_document)
201 safe_id, id_counts = _safe_document_id(document.get("document_id"), specs)
202 document_counts: dict[str, int] = {}
203 _merge_counts(document_counts, id_counts)
204
205 safe_string_fields: dict[str, str] = {}
206 for field in (
207 "title",
208 "content",
209 "source_locator",
210 "source_url",
211 "source_name",
212 "version",
213 "tenant_id",
214 "visibility",
215 ):
216 redacted, counts = _scan_and_redact(document.get(field), specs)
217 safe_string_fields[field] = redacted
218 _merge_counts(document_counts, counts)
219
220 safe_roles: list[str] = []
221 roles = document.get("allowed_roles") if isinstance(document.get("allowed_roles"), list) else []
222 for role in roles:
223 redacted, counts = _scan_and_redact(role, specs)
224 _merge_counts(document_counts, counts)
225 if redacted.strip() and redacted.strip().lower() not in safe_roles:
226 safe_roles.append(redacted.strip().lower())
227
228 safe_groups: list[str] = []
229 groups = document.get("allowed_groups") if isinstance(document.get("allowed_groups"), list) else []
230 for group in groups:
231 redacted, counts = _scan_and_redact(group, specs)
232 _merge_counts(document_counts, counts)
233 if redacted.strip() and redacted.strip().lower() not in safe_groups:
234 safe_groups.append(redacted.strip().lower())
235
236 if document_counts:
237 affected_document_count += 1
238 _merge_counts(match_counts, document_counts)
239 if selected_mode == "block":
240 blocked_document_ids.append(safe_id)
241 continue
242
243 # Rebuild from an allowlist instead of passing arbitrary upstream keys.
244 # Otherwise an unknown metadata field could carry the very raw value
245 # that this guard removed from content/title.
246 document = {
247 "document_id": safe_id,
248 "title": safe_string_fields["title"],
249 "version": safe_string_fields["version"] or "1",
250 "content": safe_string_fields["content"],
251 "content_hash": hashlib.sha256(safe_string_fields["content"].encode("utf-8")).hexdigest(),
252 "page": raw_document.get("page"),
253 "source_locator": safe_string_fields["source_locator"],
254 "source_url": safe_string_fields["source_url"],
255 "source_name": safe_string_fields["source_name"],
256 "tenant_id": safe_string_fields["tenant_id"].strip().lower(),
257 "visibility": safe_string_fields["visibility"].strip().lower(),
258 "classification": str(raw_document.get("classification") or "internal").strip().lower(),
259 "allowed_roles": safe_roles,
260 "allowed_groups": safe_groups,
261 "active": _as_bool(raw_document.get("active"), True),
262 "deleted": _as_bool(raw_document.get("deleted"), False),
263 }
264 if document_counts and _as_bool(upgrade_classification, True):
265 document["classification"] = _upgrade_classification(document.get("classification"))
266 document["security_scan"] = {
267 "guard_mode": selected_mode,
268 "sensitive_pattern_count": sum(document_counts.values()),
269 "pattern_types": sorted(document_counts),
270 }
271 safe_documents.append(document)
272
273 if not documents:
274 errors.append("No documents were available for the confidentiality guard.")
275 if invalid_document_count:
276 warnings.append("One or more invalid document items were skipped by the confidentiality guard.")
277 if affected_document_count:
278 action = "blocked" if selected_mode == "block" else "redacted"
279 warnings.append(f"Sensitive patterns were detected and {action} according to the configured policy.")
280 if documents and not safe_documents:
281 errors.append("No documents remained after the confidentiality policy was applied.")
282
283 success = bool(safe_documents) and not errors
284 return {
285 "success": success,
286 "stage": "pii_confidential_data_guard",
287 "documents": safe_documents,
288 "document_count": len(safe_documents),
289 "guard_mode": selected_mode,
290 "affected_document_count": affected_document_count,
291 "blocked_document_count": len(blocked_document_ids),
292 "blocked_document_ids": blocked_document_ids,
293 "match_counts": match_counts,
294 "errors": errors,
295 "warnings": warnings,
296 "trace": [
297 {
298 "stage": "pii_confidential_data_guard",
299 "status": "success" if success else "failed",
300 "guard_mode": selected_mode,
301 "input_document_count": len(documents),
302 "safe_document_count": len(safe_documents),
303 "affected_document_count": affected_document_count,
304 "blocked_document_count": len(blocked_document_ids),
305 "sensitive_pattern_count": sum(match_counts.values()),
306 }
307 ],
308 }
