41# 1. Goodocs(auth: dict) 형태로 생성할 수 있어야 합니다.

47# 설치되어 있어야 하므로, 파일 하나만 등록하는 Standalone 방식과 다릅니다.

67def _secret_text(value: Any) -> str:
68 """SecretStr 또는 일반 문자열을 실제 인증 문자열로 변환합니다."""
69 if isinstance(value, SecretStr):
70 return value.get_secret_value().strip()
71 get_secret_value = getattr(value, "get_secret_value", None)
72 if callable(get_secret_value):
73 return str(get_secret_value() or "").strip()
74 return str(value or "").strip()

77def _validated_max_rows(value: Any) -> int:
78 """최대 출력 행 수를 안전한 정수 범위로 검증합니다."""
79 try:
80 max_rows = int(value)
81 except (TypeError, ValueError) as exc:
82 raise ValueError("최대 출력 행 수는 정수로 입력해 주세요.") from exc
83
84 if not 1 <= max_rows <= MAX_ALLOWED_ROWS:
85 raise ValueError(f"최대 출력 행 수는 1~{MAX_ALLOWED_ROWS:,} 사이여야 합니다.")
86 return max_rows

89def _build_auth(
90 document_id: Any,
91 user_id: Any,
92 token_source: Any,
93 token_key: Any,
94 sheet_name: Any = "",
95) -> dict[str, str]:
96 """입력값을 검증하고 실제 GooDocs 모듈에 전달할 인증 사전을 만듭니다."""
97 values = {
98 "문서 ID": str(document_id or "").strip(),
99 "사용자 ID": str(user_id or "").strip(),
100 "토큰 소스": _secret_text(token_source),
101 "토큰 키": _secret_text(token_key),
102 }
103 missing = [name for name, value in values.items() if not value]
104 if missing:
105 raise ValueError(f"필수 입력값이 없습니다: {', '.join(missing)}")
106
107 auth = {
108 "USER_ID": values["사용자 ID"],
109 "DOC_ID": values["문서 ID"],
110 "TOKEN_SOURCE": values["토큰 소스"],
111 "TOKEN_KEY": values["토큰 키"],
112 }
113 normalized_sheet_name = str(sheet_name or "").strip()
114 if normalized_sheet_name:
115 auth["SHEET_NAME"] = normalized_sheet_name
116 return auth

125def _is_empty_value(value: Any) -> bool:
126 """시트의 완전히 빈 행을 판별하기 위한 빈 값 규칙을 적용합니다."""
127 if value is None:
128 return True
129 if isinstance(value, str):
130 return value.strip() == ""
131 try:
132 is_na = pd.isna(value)
133 if isinstance(is_na, bool):
134 return is_na
135 # numpy.bool_은 bool과 다른 타입이므로 bool 변환을 한 번 더 시도합니다.
136 if type(is_na).__name__ == "bool_":
137 return bool(is_na)
138 except Exception:
139 pass
140 return False

143def _records_from_result(result: Any) -> list[dict[str, Any]]:
144 """실제 모듈의 반환값을 행 순서가 유지되는 레코드 목록으로 변환합니다."""
145 if isinstance(result, pd.DataFrame):
146 if not result.columns.is_unique:
147 raise ValueError("GooDocs 조회 결과에 중복된 열 이름이 있습니다.")
148 records = result.reset_index(drop=True).to_dict(orient="records")
149 elif isinstance(result, list):
150 if not all(isinstance(row, dict) for row in result):
151 raise TypeError("GooDocs 조회 결과 목록의 모든 항목은 dict 행이어야 합니다.")
152 records = result
153 else:
154 raise TypeError(
155 "GooDocs read_all()은 pandas DataFrame 또는 list[dict]를 반환해야 합니다."
156 )
157
158 cleaned_rows: list[dict[str, Any]] = []
159 for row in records:
160 cleaned = {
161 str(column): value
162 for column, value in row.items()
163 if not _is_system_column(column)
164 }
165 if cleaned and any(not _is_empty_value(value) for value in cleaned.values()):
166 cleaned_rows.append(cleaned)
167 return cleaned_rows

170def read_goodocs_rows(
171 document_id: Any,
172 user_id: Any,
173 token_source: Any,
174 token_key: Any,
175 sheet_name: Any = "",
176 max_rows: Any = DEFAULT_MAX_ROWS,
177 *,
178 client_factory: Callable[[dict[str, str]], Any] | None = None,
179) -> list[dict[str, Any]]:
180 """GooDocs를 한 번 조회하고 후속 처리에 바로 쓸 수 있는 행 목록을 반환합니다.
181
182 `client_factory`에 가짜 클라이언트를 전달할 수 있으므로 실제 사내 모듈이나
183 네트워크 없이도 인증 계약과 결과 정제 규칙을 단위 테스트할 수 있습니다.
184 """
185 auth = _build_auth(document_id, user_id, token_source, token_key, sheet_name)
186 output_limit = _validated_max_rows(max_rows)
187 factory = client_factory or Goodocs
188
189 try:
190 client = factory(auth)
191 read_all = getattr(client, "read_all", None)
192 if not callable(read_all):
193 raise TypeError("GooDocs 클라이언트에 호출 가능한 read_all()이 없습니다.")
194 result = read_all()
195 except GooDocsModuleNotConfiguredError:
196 raise
197 except Exception as exc:
198 # 인증값이나 외부 응답 본문이 오류 메시지에 섞이지 않도록 원문은 노출하지 않습니다.
199 raise RuntimeError(
200 "GooDocs 문서를 조회하지 못했습니다. 실제 모듈 설정과 인증 정보를 확인해 주세요. "
201 f"(오류 유형: {type(exc).__name__})"
202 ) from None
203
204 try:
205 rows = _records_from_result(result)
206 except (TypeError, ValueError):
207 raise
208 except Exception as exc:
209 raise RuntimeError(
210 "GooDocs 조회 결과를 데이터 테이블로 변환하지 못했습니다. "
211 f"(오류 유형: {type(exc).__name__})"
212 ) from None
213 return rows[:output_limit]
