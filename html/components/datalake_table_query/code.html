77def _sql_guard_text(sql: str) -> str:
78 """SQL 주석과 문자열 literal을 같은 길이의 공백으로 바꿉니다."""
79
80 result: list[str] = []
81 index = 0
82 state = "normal"
83 while index < len(sql):
84 char = sql[index]
85 next_char = sql[index + 1] if index + 1 < len(sql) else ""
86
87 if state == "normal":
88 if char == "-" and next_char == "-":
89 result.extend((" ", " "))
90 index += 2
91 state = "line_comment"
92 continue
93 if char == "/" and next_char == "*":
94 result.extend((" ", " "))
95 index += 2
96 state = "block_comment"
97 continue
98 if char == "'":
99 result.append(" ")
100 index += 1
101 state = "single_quote"
102 continue
103 if char == '"':
104 result.append(" ")
105 index += 1
106 state = "double_quote"
107 continue
108 result.append(char)
109 index += 1
110 continue
111
112 if state == "line_comment":
113 if char in "\r\n":
114 result.append(char)
115 state = "normal"
116 else:
117 result.append(" ")
118 index += 1
119 continue
120
121 if state == "block_comment":
122 if char == "*" and next_char == "/":
123 result.extend((" ", " "))
124 index += 2
125 state = "normal"
126 else:
127 result.append("\n" if char == "\n" else " ")
128 index += 1
129 continue
130
131 quote_char = "'" if state == "single_quote" else '"'
132 if char == quote_char and next_char == quote_char:
133 result.extend((" ", " "))
134 index += 2
135 continue
136 result.append("\n" if char == "\n" else " ")
137 index += 1
138 if char == quote_char:
139 state = "normal"
140
141 if state in {"single_quote", "double_quote", "block_comment"}:
142 raise ValueError("조회 SQL의 따옴표 또는 블록 주석이 닫히지 않았습니다.")
143 return "".join(result)

146def validate_read_only_sql(sql: Any) -> str:
147 """단일 SELECT/WITH 조회문인지 확인합니다."""
148
149 text = str(sql or "").strip()
150 if not text:
151 raise ValueError("조회 SQL을 입력해 주세요.")
152 if re.search(r"/\*!", text):
153 raise ValueError("실행 가능한 MySQL 주석(/*! ... */)은 조회 SQL에 사용할 수 없습니다.")
154
155 guard_text = _sql_guard_text(text)
156 semicolon_indexes = [index for index, char in enumerate(guard_text) if char == ";"]
157 if len(semicolon_indexes) > 1:
158 raise ValueError("한 번에 하나의 조회 SQL만 실행할 수 있습니다.")
159 if semicolon_indexes:
160 semicolon_index = semicolon_indexes[0]
161 if guard_text[semicolon_index + 1 :].strip():
162 raise ValueError("조회 SQL 뒤에 다른 문장을 함께 실행할 수 없습니다.")
163 text = (text[:semicolon_index] + text[semicolon_index + 1 :]).strip()
164 guard_text = _sql_guard_text(text)
165
166 tokens = re.findall(r"[A-Za-z_][A-Za-z0-9_$]*", guard_text.upper())
167 if not tokens or tokens[0] not in {"SELECT", "WITH"}:
168 raise ValueError("조회 전용 Component에서는 SELECT 또는 WITH로 시작하는 SQL만 허용합니다.")
169
170 forbidden = sorted(set(tokens).intersection(FORBIDDEN_SQL_KEYWORDS))
171 if forbidden:
172 raise ValueError(f"조회 SQL에 허용되지 않는 명령이 포함되어 있습니다: {', '.join(forbidden)}")
173 return text

249def normalize_allowed_mysql_hosts(value: Any) -> tuple[str, ...]:
250 """정확한 host 또는 `.example.com` suffix 허용 목록을 정리합니다."""
251
252 raw_items = re.split(r"[,;\r\n]+", str(value or ""))
253 rules: list[str] = []
254 for raw in raw_items:
255 rule = raw.strip().lower().rstrip(".")
256 if not rule:
257 continue
258 if rule.startswith("*."):
259 rule = "." + rule[2:]
260 is_suffix = rule.startswith(".")
261 host_part = rule[1:] if is_suffix else rule
262 if not host_part or "://" in host_part or "/" in host_part or ":" in host_part:
263 raise ValueError("허용 MySQL host에는 scheme, path, port를 넣을 수 없습니다.")
264 try:
265 ipaddress.ip_address(host_part)
266 if is_suffix:
267 raise ValueError("IP 주소는 suffix가 아닌 정확한 주소로만 허용할 수 있습니다.")
268 except ValueError as exc:
269 if "suffix" in str(exc):
270 raise
271 labels = host_part.split(".")
272 if any(
273 not label
274 or len(label) > 63
275 or not re.fullmatch(r"[a-z0-9](?:[a-z0-9-]*[a-z0-9])?", label)
276 for label in labels
277 ):
278 raise ValueError(f"허용 MySQL host 형식이 올바르지 않습니다: {raw.strip()}") from exc
279 normalized = f".{host_part}" if is_suffix else host_part
280 if normalized not in rules:
281 rules.append(normalized)
282 if not rules:
283 raise ValueError("허용 MySQL host를 한 개 이상 입력해 주세요.")
284 return tuple(rules)

352def parse_mysql_endpoint(value: Any) -> tuple[str, int, str]:
353 """host:port, mysql://, jdbc:mysql:// 주소를 MySQL 연결값으로 나눕니다."""
354
355 text = str(value or "").strip()
356 if not text:
357 raise ValueError("Cluster 응답에 MySQL 접속 주소가 없습니다.")
358 if text.lower().startswith("jdbc:mysql://"):
359 text = text[5:]
360 if "://" not in text:
361 text = "mysql://" + text
362
363 parsed = urlsplit(text)
364 if parsed.scheme.lower() != "mysql" or not parsed.hostname:
365 raise ValueError("MySQL 접속 주소 형식이 올바르지 않습니다.")
366 if parsed.username or parsed.password:
367 raise ValueError("Cluster 응답의 MySQL 주소에 계정 정보가 포함되어서는 안 됩니다.")
368 if parsed.query or parsed.fragment:
369 raise ValueError("Cluster 응답의 MySQL 주소에는 query string이나 fragment가 없어야 합니다.")
370 try:
371 port = parsed.port
372 except ValueError as exc:
373 raise ValueError("MySQL 접속 주소의 port가 올바르지 않습니다.") from exc
374 if port is None or not 1 <= port <= 65535:
375 raise ValueError("MySQL 접속 주소에는 1~65535 범위의 port가 필요합니다.")
376 database = unquote(parsed.path.lstrip("/").split("/", 1)[0]) if parsed.path.strip("/") else ""
377 return parsed.hostname, port, database

412 cluster_wait_seconds: int,
413 poll_interval_seconds: int,
414 status_fetcher: Callable[[str, Mapping[str, str], int], Any],
415 sleep_fn: Callable[[float], Awaitable[Any]] = asyncio.sleep,
416 monotonic_fn: Callable[[], float] = time.monotonic,
417) -> tuple[str, int, str]:
418 """실제 전체 deadline 안에서 Cluster 상태와 허용된 MySQL 주소를 확인합니다."""
419
420 max_attempts = max(1, cluster_wait_seconds // poll_interval_seconds + 1)
421 deadline = monotonic_fn() + cluster_wait_seconds
422 last_status = "UNKNOWN"
423 for attempt in range(1, max_attempts + 1):
424 remaining = deadline - monotonic_fn()
425 if remaining <= 0:
426 break
427 per_request_timeout = max(1, min(request_timeout_seconds, int(remaining + 0.999)))
428 fetched = status_fetcher(status_url, headers, per_request_timeout)
429 if inspect.isawaitable(fetched):
430 try:
431 payload = await asyncio.wait_for(fetched, timeout=remaining)
432 except asyncio.TimeoutError as exc:
433 raise TimeoutError("Datalake Cluster 상태 조회가 전체 대기시간을 초과했습니다.") from exc
434 else:
435 payload = fetched
436 if not isinstance(payload, Mapping):
437 raise RuntimeError("Datalake Cluster API 응답이 JSON 객체가 아닙니다.")
438
439 last_status = str(payload.get("status") or "UNKNOWN").upper()
440 if last_status == "RUNNING":
441 endpoints = payload.get("endpoints")
442 if not isinstance(endpoints, Mapping):
443 raise RuntimeError("RUNNING 응답에 endpoints 객체가 없습니다.")
444 host, port, database = parse_mysql_endpoint(endpoints.get(endpoint_key))
445 return ensure_allowed_mysql_host(host, allowed_mysql_hosts), port, database
446
447 if attempt < max_attempts:
448 remaining = deadline - monotonic_fn()
449 if remaining <= 0:
450 break
451 await sleep_fn(min(poll_interval_seconds, remaining))
452 raise TimeoutError(f"Datalake Cluster가 제한시간 안에 RUNNING 상태가 되지 않았습니다. 마지막 상태: {last_status}")

537def _column_names(cursor: Any) -> list[str]:
538 """MySQL cursor에서 중복되지 않는 컬럼명을 생성합니다."""
539
540 raw_columns = getattr(cursor, "column_names", None)
541 if not raw_columns:
542 description = getattr(cursor, "description", None) or []
543 raw_columns = [getattr(item, "name", None) or item[0] for item in description]
544
545 columns: list[str] = []
546 counts: dict[str, int] = {}
547 for index, raw_name in enumerate(raw_columns or [], 1):
548 base_name = str(raw_name or f"COLUMN_{index}")
549 counts[base_name] = counts.get(base_name, 0) + 1
550 suffix = counts[base_name]
551 columns.append(base_name if suffix == 1 else f"{base_name}_{suffix}")
552 return columns

555def _table_value(value: Any) -> Any:
556 """StarRocks/MySQL 결과 값을 DataFrame 화면에서 안전하게 확인 가능한 값으로 바꿉니다."""
557
558 if value is None or isinstance(value, (str, int, float, bool)):
559 return value
560 if isinstance(value, (datetime, date)):
561 return value.isoformat()
562 if isinstance(value, Decimal):
563 return float(value)
564 if isinstance(value, (bytes, bytearray, memoryview)):
565 return base64.b64encode(bytes(value)).decode("ascii")
566 try:
567 if value != value:
568 return None
569 except Exception:
570 pass
571 return str(value)

574def _rows_to_records(rows: list[Any], columns: list[str]) -> list[dict[str, Any]]:
575 """MySQL cursor row를 컬럼명이 있는 record 목록으로 변환합니다."""
576
577 records: list[dict[str, Any]] = []
578 for row in rows:
579 if isinstance(row, Mapping):
580 record = {str(key): _table_value(item) for key, item in row.items()}
581 else:
582 record = {column: _table_value(item) for column, item in zip(columns, row, strict=False)}
583 records.append(record)
584 return records

589
590 closer = getattr(resource, "close", None)
591 if callable(closer):

593 closer()
594 except Exception:

598def execute_mysql_query(
599 *,
600 host: str,
601 port: int,
602 database: str,
603 user_id: str,
604 token: str,
605 sql: str,
606 bind_parameters: Mapping[str, Any] | None,
607 max_rows: int,
608 connect_timeout_seconds: int,
609 query_timeout_seconds: int,
610 ssl_ca_path: str,
611 connect_fn: Callable[..., Any] | None = None,
612) -> tuple[list[dict[str, Any]], list[str], bool]:
613 """StarRocks MySQL endpoint에서 SQL을 실행합니다.
614
615 ``connect_fn``을 주입할 수 있어 실제 Datalake 없이 connection/cursor 가짜
616 객체로 native bind, 최대 행 수, 리소스 종료 계약을 테스트할 수 있습니다.
617 """
618
619 connection_factory = connect_fn or _load_mysql_connect()
620 connection = None
621 cursor = None
622 try:
623 connect_kwargs: dict[str, Any] = {
624 "host": host,
625 "port": port,
626 "user": user_id,
627 "password": token,
628 "use_pure": True,
629 "ssl_disabled": False,
630 "ssl_ca": ssl_ca_path,
631 "ssl_verify_cert": True,
632 "ssl_verify_identity": True,
633 "auth_plugin": MYSQL_AUTH_PLUGIN,
634 "allow_local_infile": False,
635 "connection_timeout": connect_timeout_seconds,
636 "read_timeout": query_timeout_seconds,
637 "write_timeout": query_timeout_seconds,
638 }
639 if database:
640 connect_kwargs["database"] = database
641 connection = connection_factory(**connect_kwargs)
642 cursor = connection.cursor()
643 parameters = dict(bind_parameters or {})
644 if parameters:
645 cursor.execute(sql, parameters)
646 else:
647 cursor.execute(sql)
648
649 columns = _column_names(cursor)
650 if not columns:
651 raise RuntimeError("Datalake 조회 결과에 컬럼 정보가 없습니다.")
652 fetched = list(cursor.fetchmany(max_rows + 1))
653 truncated = len(fetched) > max_rows
654 return _rows_to_records(fetched[:max_rows], columns), columns, truncated
655 finally:
656 _close_quietly(cursor)
657 _close_quietly(connection)

660def _safe_error_text(error: Exception, sensitive_values: list[str]) -> str:
661 """인증 정보와 내부 주소가 노출되지 않도록 오류 문자열을 정리합니다."""
662
663 if error.__class__.__module__ not in {"builtins", __name__}:
664 return f"외부 MySQL driver 오류가 발생했습니다 ({error.__class__.__name__})."
665 text = str(error).strip() or error.__class__.__name__
666 for value in sensitive_values:
667 if value:
668 text = text.replace(value, "[보호됨]")
669 return text[:500]
