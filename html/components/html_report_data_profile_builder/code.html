133def _profile_column(column: str, rows: list[dict[str, Any]]) -> dict[str, Any]:
134 """한 컬럼의 값 분포를 보고 타입, 결측 수, 샘플 값, 숫자 통계를 계산합니다."""
135
136 values = [row.get(column) for row in rows]
137 non_empty = [value for value in values if value not in (None, "", [], {})]
138 unique_values = _unique(non_empty)
139 inferred = _infer_type(column, non_empty, unique_values)
140 numeric_values = [number for number in (_number(value) for value in non_empty) if number is not None]
141 result: dict[str, Any] = {
142 "name": column,
143 "inferred_type": inferred,
144 "non_empty_count": len(non_empty),
145 "null_count": len(values) - len(non_empty),
146 "unique_count": len(unique_values),
147 "sample_values": _sample(non_empty),
148 "top_values": _top_values(non_empty),
149 "is_metric_candidate": inferred == "numeric",
150 "is_dimension_candidate": inferred in {"dimension", "id", "status"},
151 "is_time_candidate": inferred == "time",
152 "is_text_candidate": inferred == "text",
153 }
154 if DELTA_RE.search(column):
155 # 증감률/변화량처럼 보이는 컬럼은 카드나 해석에서 별도로 활용할 수 있게 표시합니다.
156 result["semantic_hint"] = "delta_or_rate"
157 if STATUS_RE.search(column):
158 # 오류/경고/상태 컬럼은 예외 테이블 후보가 됩니다.
159 result["semantic_hint"] = "status_or_exception"
160 if numeric_values:
161 result["numeric_stats"] = {
162 "min": min(numeric_values),
163 "max": max(numeric_values),
164 "sum": sum(numeric_values),
165 "avg": sum(numeric_values) / len(numeric_values),
166 }
167 return result

170def _infer_type(column: str, values: list[Any], unique_values: list[Any]) -> str:
171 """컬럼명과 실제 값 샘플을 함께 보고 간단한 타입을 추정합니다."""
172
173 if not values:
174 return "empty"
175 if DATE_RE.search(column) or _date_ratio(values) >= 0.6:
176 return "time"
177 if _numeric_ratio(values) >= 0.8:
178 return "numeric"
179 if STATUS_RE.search(column):
180 return "status"
181 if ID_RE.search(column) and len(unique_values) >= min(len(values), 20):
182 return "id"
183 if sum(len(str(value)) for value in values) / max(len(values), 1) >= 50:
184 return "text"
185 return "dimension"

189 """컬럼별 프로파일을 숫자/범주/날짜/상태 같은 그룹으로 다시 묶습니다."""
190
191 groups = {
192 "numeric_columns": [],
193 "dimension_columns": [],
194 "time_columns": [],
195 "text_columns": [],
196 "id_columns": [],
197 "status_columns": [],
198 "delta_columns": [],
199 }
200 for profile in column_profiles:
201 name = str(profile.get("name") or "")

203 if inferred == "numeric":
204 groups["numeric_columns"].append(name)
205 if inferred in {"dimension", "status"}:
206 groups["dimension_columns"].append(name)
207 if inferred == "time":
208 groups["time_columns"].append(name)
209 if inferred == "text":
210 groups["text_columns"].append(name)
211 if inferred == "id":
212 groups["id_columns"].append(name)
213 if inferred == "status" or profile.get("semantic_hint") == "status_or_exception":
214 groups["status_columns"].append(name)
215 if profile.get("semantic_hint") == "delta_or_rate":
216 groups["delta_columns"].append(name)
217 return groups

241def _payload(value: Any) -> dict[str, Any]:
242 """Langflow Data/Message/dict/JSON 문자열을 일반 dict로 맞춥니다."""
243
244 if isinstance(value, dict):
245 return deepcopy(value)
246 data = getattr(value, "data", None)
247 if isinstance(data, dict):
248 return deepcopy(data)

250 if isinstance(text, str) and text.strip():
251 try:
252 parsed = json.loads(text)
253 except Exception:
254 return {"text": text}
255 return deepcopy(parsed) if isinstance(parsed, dict) else {"text": text}
256 return {}

356def _top_values(values: list[Any], limit: int = 12) -> list[dict[str, Any]]:
357 """범주/상태/코드 값의 빈도를 LLM이 이해하기 쉬운 형태로 요약합니다.
358
359 예를 들어 `ALERT_LEVEL` 컬럼에 HIGH/WARN/NORMAL 값이 있으면
360 LLM이 정확한 값 문자열을 사용해 filter_rules나 highlight_rules를 만들 수 있습니다.
361 """
362
363 counts: dict[str, dict[str, Any]] = {}
364 for value in values:
365 if value in (None, "", [], {}):
366 continue
367 signature = json.dumps(value, ensure_ascii=False, sort_keys=True, default=str)
368 if signature not in counts:
369 counts[signature] = {"value": deepcopy(value), "count": 0}
370 counts[signature]["count"] += 1
371 ranked = sorted(counts.values(), key=lambda item: (-int(item["count"]), str(item["value"])))
372 return ranked[:limit]

375def _unique(values: list[Any]) -> list[Any]:
376 """값 목록에서 빈 값과 중복 값을 제거합니다."""
377
378 result: list[Any] = []
379 signatures: set[str] = set()
380 for value in values:
381 if value in (None, "", [], {}):
382 continue
383 signature = json.dumps(value, ensure_ascii=False, sort_keys=True, default=str)
384 if signature not in signatures:
385 signatures.add(signature)
386 result.append(deepcopy(value))
387 return result
