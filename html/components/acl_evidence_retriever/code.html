37def _payload(value: Any) -> dict[str, Any]:
38 if value is None:
39 return {}
40 if isinstance(value, dict):
41 return value
42 data = getattr(value, "data", None)
43 if isinstance(data, dict):
44 return data
45 text = _text(value)

70 if value is None:

81def _bool(value: Any, default: bool = False) -> bool:
82 if isinstance(value, bool):
83 return value
84 text = str(value or "").strip().lower()
85 if text in {"1", "true", "yes", "y", "active"}:
86 return True
87 if text in {"0", "false", "no", "n", "inactive", "deleted"}:

92def _strings(value: Any, limit: int = 64) -> list[str]:
93 if isinstance(value, str):
94 values = re.split(r"[,;\n]", value)
95 else:
96 values = _list(value)
97 result: list[str] = []
98 for item in values:
99 text = str(item or "").strip().lower()
100 if text and text not in result:
101 result.append(text[:128])
102 if len(result) >= limit:
103 break
104 return result

107def _clean_text(value: Any, limit: int) -> str:
108 text = unicodedata.normalize("NFKC", _text(value))
109 text = re.sub(r"[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]", " ", text)
110 text = re.sub(r"[\u200b-\u200f\u2060\ufeff]", "", text)
111 return text[:limit]

119def _records(index_value: Any) -> list[dict[str, Any]]:
120 payload = _payload(index_value)
121 nested = _dict(payload.get("document_index"))
122 if nested:
123 payload = nested
124 raw_items: list[Any] = []
125 for key in ("chunks", "documents", "records", "items", "ingest_data"):
126 if isinstance(payload.get(key), list):
127 raw_items = payload[key]
128 break
129 if not raw_items and any(key in payload for key in ("text", "content", "page_content")):
130 raw_items = [payload]
131
132 expanded: list[dict[str, Any]] = []
133 for raw in raw_items:
134 if not isinstance(raw, dict):
135 continue
136 child_chunks = raw.get("chunks")
137 if isinstance(child_chunks, list):
138 parent = {key: value for key, value in raw.items() if key != "chunks"}
139 for child in child_chunks:
140 if isinstance(child, dict):
141 expanded.append({**parent, **child})
142 else:
143 expanded.append(raw)
144 return expanded

153def _is_authorized(record: dict[str, Any], actor: dict[str, Any]) -> bool:
154 item = _merged_record(record)
155 metadata = _dict(item.get("metadata"))
156 acl = _dict(item.get("acl")) or _dict(metadata.get("acl"))
157
158 # ACL metadata is mandatory. Top-level compatibility fields are accepted
159 # only when they make an explicit ACL, never as an implicit public default.
160 acl_keys = {
161 "tenant_id", "visibility", "classification", "security_classification",
162 "allowed_users", "allowed_roles", "allowed_groups",
163 }
164 explicit_acl = bool(acl) or any(key in item for key in acl_keys) or any(key in metadata for key in acl_keys)
165 if not explicit_acl:
166 return False

168 def field(name: str, fallback: Any = None) -> Any:
169 if name in acl:
170 return acl.get(name)
171 if name in item:
172 return item.get(name)
173 return metadata.get(name, fallback)
174
175 if not _bool(item.get("active", metadata.get("active", True)), default=True):
176 return False
177 if str(item.get("status") or metadata.get("status") or "").strip().lower() in {"deleted", "inactive", "revoked"}:
178 return False
179
180 actor_tenant = str(actor.get("tenant_id") or "").strip().lower()
181 document_tenant = str(field("tenant_id") or "").strip().lower()
182 if not actor_tenant or (document_tenant and document_tenant not in {actor_tenant, "*"}):
183 return False
184
185 classification = _classification(field("classification", field("security_classification", "restricted")))
186 try:
187 actor_level = int(actor.get("clearance_level", CLASSIFICATION_LEVELS.get(actor.get("clearance"), -1)))
188 except Exception:
189 actor_level = -1
190 if CLASSIFICATION_LEVELS[classification] > actor_level:
191 return False
192
193 visibility = str(field("visibility") or "").strip().lower()
194 allowed_users = set(_strings(field("allowed_users")))
195 allowed_roles = set(_strings(field("allowed_roles")))
196 allowed_groups = set(_strings(field("allowed_groups")))
197 actor_user = str(actor.get("user_id") or "").strip().lower()
198 actor_roles = set(_strings(actor.get("roles")))
199 actor_groups = set(_strings(actor.get("groups")))
200
201 principal_rules_present = bool(allowed_users or allowed_roles or allowed_groups)
202 principal_match = (
203 (actor_user in allowed_users if allowed_users else False)
204 or bool(actor_roles & allowed_roles)
205 or bool(actor_groups & allowed_groups)
206 )
207 if principal_rules_present:
208 return principal_match
209 # A legacy/single-tenant row without tenant_id is accepted only when an
210 # explicit user/role/group ACL above matched. Visibility alone is not
211 # sufficient when tenant scope is absent.
212 if not document_tenant:
213 return False
214 return visibility in {"public", "internal", "company"}

222def _stem_token(token: str) -> str:
223 # Korean particles often attach directly to English/Korean terms
224 # (``RAG를``, ``flow와``, ``문서는``). Removing only well-known particles
225 # improves exact lexical recall without turning this security component
226 # into a broad fuzzy matcher.
227 particles = (
228 "으로", "에서", "에게", "까지", "부터", "처럼", "보다", "하고", "이며", "이고",
229 "에는", "에서도", "은", "는", "이", "가", "을", "를", "와", "과", "로", "에", "의", "도", "만",
230 )
231 for suffix in particles:
232 if token.endswith(suffix) and len(token) - len(suffix) >= 2:
233 return token[: -len(suffix)]
234 return token

237def _tokens(text: str) -> list[str]:
238 normalized = unicodedata.normalize("NFKC", text).lower()
239 fillers = {
240 "그리고", "대한", "관련", "무엇", "어떻게", "어떤", "알려줘", "해주세요", "설명", "질문",
241 "경우", "내용", "what", "how", "why", "about", "please",
242 }
243 result: list[str] = []
244 for raw in re.findall(r"[a-z0-9가-힣_]{2,}", normalized):
245 token = _stem_token(raw)
246 if token and token not in fillers:
247 result.append(token)
248 return result

251def _score(question: str, content: str, title: str) -> float:
252 query_tokens = set(_tokens(question))
253 if not query_tokens:
254 return 0.0
255 content_tokens = set(_tokens(content))
256 title_tokens = set(_tokens(title))
257 overlap = len(query_tokens & content_tokens) / max(len(query_tokens), 1)
258 title_overlap = len(query_tokens & title_tokens) / max(len(query_tokens), 1)
259 phrase = 1.0 if unicodedata.normalize("NFKC", question).lower() in unicodedata.normalize("NFKC", content).lower() else 0.0
260 density = len(query_tokens & content_tokens) / max(math.sqrt(len(content_tokens) or 1), 1.0)
261 return round(min(1.0, overlap * 0.68 + title_overlap * 0.20 + phrase * 0.08 + min(density, 1.0) * 0.04), 6)

264def retrieve_authorized(request_value: Any, index_value: Any) -> dict[str, Any]:
265 request = _payload(request_value)
266 question = _clean_text(request.get("question"), 8000)
267 actor = _dict(request.get("actor"))
268 request_ready = bool(
269 request.get("success")
270 and _dict(request.get("security")).get("authorized_for_retrieval")
271 and actor.get("identity_verified")
272 )
273
274 ranked: list[dict[str, Any]] = []
275 injection_detected = False
276 if request_ready and question:
277 for raw_record in _records(index_value):
278 # Security boundary: no text, ID, title, or score is read into an
279 # output object until this record passes ACL.
280 if not _is_authorized(raw_record, actor):
281 continue
282 item = _merged_record(raw_record)
283 metadata = _dict(item.get("metadata"))
284 document_id = _clean_text(
285 item.get("document_id") or item.get("doc_id") or metadata.get("document_id"),
286 160,
287 )
288 page = item.get("page", item.get("page_number", metadata.get("page", metadata.get("page_number"))))
289 locator = _clean_text(
290 item.get("source_locator")
291 or metadata.get("source_locator")
292 or (f"page:{page}" if page not in (None, "") else ""),
293 240,
294 )
295 # Citation-ready identity is mandatory. External/malformed rows
296 # cannot become evidence merely because their text is relevant.
297 if not document_id or (page in (None, "") and not locator):
298 continue
299 content = _clean_text(
300 item.get("text") or item.get("content") or item.get("page_content"),
301 12000,
302 )
303 if not content:
304 continue
305 title = _clean_text(
306 item.get("title") or item.get("document_title") or item.get("source_file") or "사내 문서",
307 240,
308 )
309 score = _score(question, content, title)
310 if score <= 0.0:
311 continue
312 # Injection signals in irrelevant authorized documents must not
313 # block an unrelated question. Relevance is computed only after
314 # ACL, then a relevant suspicious record is excluded here.
315 if _has_injection_signal(content):
316 injection_detected = True
317 continue
318 ranked.append(
319 {
320 "record": item,
321 "content": content,
322 "title": title,
323 "score": score,
324 "document_id": document_id,
325 "page": page,
326 "locator": locator,
327 }
328 )
329
330 ranked.sort(key=lambda row: (-row["score"], row["title"].lower()))
331 evidence: list[dict[str, Any]] = []
332 for position, row in enumerate(ranked[:6], start=1):
333 item = row["record"]
334 metadata = _dict(item.get("metadata"))
335 page = row["page"]
336 version = _clean_text(item.get("version") or metadata.get("version"), 80)
337 locator = row["locator"]
338 evidence.append(
339 {
340 "evidence_id": f"E{position}",
341 "content": row["content"][:6000],
342 "score": row["score"],
343 "source": {
344 "document_id": row["document_id"],
345 "title": row["title"],
346 "version": version,
347 "page": page if isinstance(page, (int, float, str)) else "",
348 "locator": locator,
349 "classification": _classification(
350 item.get("classification") or metadata.get("classification") or "internal"
351 ),
352 },
353 }
354 )
355
356 return {
357 "contract": "agent_ground.enterprise_rag.retrieval.v1",
358 "success": bool(request_ready and question),
359 "request_id": str(request.get("request_id") or "")[:96],
360 "question": question,
361 "retrieval_status": "evidence_found" if evidence else "no_evidence",
362 "evidence": evidence,
363 "security": {
364 "acl_applied_before_scoring": True,
365 "missing_acl_policy": "deny",
366 "injection_signal_detected": injection_detected,
367 "injection_records_excluded": injection_detected,
368 "denied_information_exposed": False,
369 },
370 "warnings": ["Potential document instructions were excluded."] if injection_detected else [],
371 "errors": [] if request_ready else ["Authorized evidence is unavailable."],
372 }
