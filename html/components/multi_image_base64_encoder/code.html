118 if isinstance(item, (str, Path)):
119 return Path(item)

148def _read_file_bytes(item: Any, position: int, max_bytes: int) -> tuple[bytes, str]:
149 """오류에 원본 경로를 남기지 않고 크기 제한보다 한 바이트까지만 읽습니다."""
150 filename = _filename_from_item(item, position)
151 path = _path_from_item(item)
152
153 if path is not None:
154 try:
155 if not path.is_file():
156 raise ImageProcessingError("file_unavailable", "업로드 파일을 읽을 수 없습니다.")
157 declared_size = int(path.stat().st_size)
158 if declared_size > max_bytes:
159 raise ImageProcessingError(
160 "file_size_exceeded",
161 "파일이 개별 크기 제한을 초과했습니다.",
162 details={"actual_bytes": declared_size, "limit_bytes": max_bytes},
163 byte_size=declared_size,
164 )
165 with path.open("rb") as handle:
166 content = handle.read(max_bytes + 1)
167 except ImageProcessingError:
168 raise
169 except (OSError, PermissionError) as exc:
170 raise ImageProcessingError("file_read_failed", "업로드 파일을 읽지 못했습니다.") from exc
171 else:
172 stream = item.get("file") if isinstance(item, dict) else getattr(item, "file", item)
173 if isinstance(stream, (bytes, bytearray, memoryview)):
174 content = bytes(stream[: max_bytes + 1])
175 elif hasattr(stream, "read"):
176 original_position: int | None = None
177 try:
178 if hasattr(stream, "tell"):
179 original_position = int(stream.tell())
180 if hasattr(stream, "seek"):
181 stream.seek(0)
182 content = stream.read(max_bytes + 1)
183 except Exception as exc:
184 raise ImageProcessingError("file_read_failed", "업로드 파일을 읽지 못했습니다.") from exc
185 finally:
186 if original_position is not None and hasattr(stream, "seek"):
187 try:
188 stream.seek(original_position)
189 except Exception:
190 pass
191 if not isinstance(content, (bytes, bytearray, memoryview)):
192 raise ImageProcessingError("file_read_failed", "파일 내용이 바이너리 형식이 아닙니다.")
193 content = bytes(content)
194 else:
195 raise ImageProcessingError("file_unavailable", "지원하지 않는 파일 입력 형식입니다.")
196
197 if len(content) > max_bytes:
198 raise ImageProcessingError(
199 "file_size_exceeded",
200 "파일이 개별 크기 제한을 초과했습니다.",
201 details={"actual_bytes_at_least": max_bytes + 1, "limit_bytes": max_bytes},
202 byte_size=max_bytes + 1,
203 )
204 return content, filename

214def _validate_static_svg(content: bytes) -> None:
215 try:
216 text = content.decode("utf-8-sig")
217 except UnicodeDecodeError as exc:
218 raise ImageProcessingError("invalid_svg_encoding", "SVG는 UTF-8 정적 XML이어야 합니다.") from exc
219
220 lowered = text.lower()
221 if re.search(r"<!\s*(?:doctype|entity)\b", lowered) or "<?xml-stylesheet" in lowered:
222 raise ImageProcessingError("unsafe_svg", "외부 엔터티나 스타일시트를 포함한 SVG는 허용되지 않습니다.")
223 without_declaration = re.sub(r"^\s*<\?xml\s+[^?]*\?>", "", lowered, count=1)
224 if "<?" in without_declaration:
225 raise ImageProcessingError("unsafe_svg", "XML 처리 지시문을 포함한 SVG는 허용되지 않습니다.")
226
227 try:
228 root = ET.fromstring(text)
229 except ET.ParseError as exc:
230 raise ImageProcessingError("invalid_svg", "SVG XML 구조가 올바르지 않습니다.") from exc
231
232 root_namespace, root_name = _split_xml_name(str(root.tag))
233 if root_name.lower() != "svg" or root_namespace not in {"", _SVG_NAMESPACE}:
234 raise ImageProcessingError("invalid_svg", "루트 요소가 정적 SVG 형식이 아닙니다.")
235
236 stack: list[tuple[ET.Element, int]] = [(root, 1)]
237 node_count = 0
238 while stack:
239 element, depth = stack.pop()
240 node_count += 1
241 if node_count > 10000 or depth > 64:
242 raise ImageProcessingError("svg_complexity_exceeded", "SVG 구조가 안전 복잡도 제한을 초과했습니다.")
243
244 namespace, local_name = _split_xml_name(str(element.tag))
245 local_name_lower = local_name.lower()
246 if namespace not in {"", _SVG_NAMESPACE} or local_name_lower not in _STATIC_SVG_TAGS:
247 raise ImageProcessingError("unsafe_svg", "동적이거나 지원하지 않는 SVG 요소가 포함되어 있습니다.")
248
249 for raw_name, raw_value in element.attrib.items():
250 attribute_namespace, attribute_name = _split_xml_name(str(raw_name))
251 attribute_name_lower = attribute_name.lower()
252 if attribute_namespace not in {"", _XLINK_NAMESPACE, _XML_NAMESPACE}:
253 raise ImageProcessingError("unsafe_svg", "지원하지 않는 SVG 속성 네임스페이스가 포함되어 있습니다.")
254 if attribute_name_lower.startswith("on") or attribute_name_lower in {"src", "formaction"}:
255 raise ImageProcessingError("unsafe_svg", "실행 가능한 SVG 속성이 포함되어 있습니다.")
256
257 value = str(raw_value or "").strip()
258 value_lower = re.sub(r"\s+", "", value.lower())
259 if attribute_name_lower == "href" and (not value.startswith("#") or len(value) <= 1):
260 raise ImageProcessingError("unsafe_svg", "SVG의 href 속성은 문서 내부 조각 참조만 허용됩니다.")
261 if any(marker in value_lower for marker in _UNSAFE_SVG_VALUE_MARKERS) or value_lower.startswith("//"):
262 raise ImageProcessingError("unsafe_svg", "외부 또는 실행 가능한 SVG 참조가 포함되어 있습니다.")
263
264 children = list(element)
265 for child in reversed(children):
266 stack.append((child, depth + 1))

269def _detect_raster_format(content: bytes) -> tuple[str, str, set[str]] | None:
270 if (
271 len(content) >= 24
272 and content.startswith(b"\x89PNG\r\n\x1a\n")
273 and content[12:16] == b"IHDR"
274 and int.from_bytes(content[16:20], "big") > 0
275 and int.from_bytes(content[20:24], "big") > 0
276 ):
277 return "png", "image/png", {".png"}
278
279 stripped_jpeg = content.rstrip(b"\x00\x09\x0a\x0d\x20")
280 if len(stripped_jpeg) >= 6 and stripped_jpeg.startswith(b"\xff\xd8\xff") and stripped_jpeg.endswith(b"\xff\xd9"):
281 return "jpeg", "image/jpeg", {".jpg", ".jpeg"}
282
283 if len(content) >= 13 and content[:6] in {b"GIF87a", b"GIF89a"}:
284 width = int.from_bytes(content[6:8], "little")
285 height = int.from_bytes(content[8:10], "little")
286 if width > 0 and height > 0:
287 return "gif", "image/gif", {".gif"}
288
289 if len(content) >= 16 and content.startswith(b"RIFF") and content[8:12] == b"WEBP":
290 declared_size = int.from_bytes(content[4:8], "little") + 8
291 if 12 <= declared_size <= len(content):
292 return "webp", "image/webp", {".webp"}
293
294 if len(content) >= 26 and content.startswith(b"BM"):
295 declared_size = int.from_bytes(content[2:6], "little")
296 pixel_offset = int.from_bytes(content[10:14], "little")
297 if 26 <= declared_size <= len(content) and 14 <= pixel_offset <= declared_size:
298 return "bmp", "image/bmp", {".bmp"}
299
300 if len(content) >= 8 and content[:4] in {b"II*\x00", b"MM\x00*"}:
301 byte_order = "little" if content[:2] == b"II" else "big"
302 first_ifd_offset = int.from_bytes(content[4:8], byte_order)
303 if first_ifd_offset == 0 or first_ifd_offset < len(content):
304 return "tiff", "image/tiff", {".tif", ".tiff"}
305
306 return None

309def _validate_image(
310 content: bytes,
311 extension: str,
312 *,
313 allow_svg: bool,
314) -> tuple[str, str]:
315 if extension not in _SUPPORTED_EXTENSIONS:
316 raise ImageProcessingError(
317 "unsupported_extension",
318 "지원하지 않는 이미지 확장자입니다.",
319 details={"supported_extensions": sorted(_SUPPORTED_EXTENSIONS)},
320 )
321 if not content:
322 raise ImageProcessingError("empty_file", "빈 파일은 이미지로 변환할 수 없습니다.")
323
324 if extension == ".svg":
325 if not allow_svg:
326 raise ImageProcessingError("svg_disabled", "SVG 입력은 기본적으로 비활성화되어 있습니다.")
327 _validate_static_svg(content)
328 return "svg", "image/svg+xml"
329
330 detected = _detect_raster_format(content)
331 if detected is None:
332 raise ImageProcessingError("invalid_image_signature", "실제 이미지 파일 서명을 확인할 수 없습니다.")
333
334 image_format, mime_type, matching_extensions = detected
335 if extension not in matching_extensions:
336 raise ImageProcessingError(
337 "extension_signature_mismatch",
338 "파일 확장자와 실제 이미지 파일 서명이 일치하지 않습니다.",
339 details={
340 "detected_format": image_format,
341 "expected_extensions": sorted(matching_extensions),
342 },
343 )
344 return image_format, mime_type

369def encode_image_files(
370 image_files: Any,
371 *,
372 output_format: str = "base64",
373 error_policy: str = "reject_batch",
374 allow_svg: bool = False,
375 max_files: int = _DEFAULT_MAX_FILES,
376 max_file_size_mb: int = _DEFAULT_MAX_FILE_MB,
377 max_total_size_mb: int = _DEFAULT_MAX_TOTAL_MB,
378) -> dict[str, Any]:
379 """이미지를 검증하고 입력 순서를 유지한 Data 출력용 묶음으로 반환합니다."""
380 files = _normalize_files(image_files)
381 selected_format = output_format if output_format in {"base64", "data_url"} else "base64"
382 selected_policy = error_policy if error_policy in {"reject_batch", "skip_invalid"} else "reject_batch"
383 file_limit = _clamp_int(max_files, _DEFAULT_MAX_FILES, 1, _MAX_FILE_COUNT)
384 per_file_mb = _clamp_int(max_file_size_mb, _DEFAULT_MAX_FILE_MB, 1, _MAX_FILE_MB)
385 total_mb = _clamp_int(max_total_size_mb, _DEFAULT_MAX_TOTAL_MB, 1, _MAX_TOTAL_MB)
386 per_file_bytes = per_file_mb * 1024 * 1024
387 total_limit_bytes = total_mb * 1024 * 1024
388 recoverable = selected_policy == "skip_invalid"
389
390 items: list[dict[str, Any]] = []
391 errors: list[dict[str, Any]] = []
392 warnings: list[dict[str, Any]] = []
393 total_read_bytes = 0
394 encoded_source_bytes = 0
395
396 if not files:
397 errors.append(
398 {
399 "code": "no_files",
400 "message": "변환할 이미지 파일이 없습니다.",
401 "stage": "input_validation",
402 "recoverable": False,
403 }
404 )
405
406 for index, file_item in enumerate(files):
407 position = index + 1
408 filename = _filename_from_item(file_item, position)
409 if index >= file_limit:
410 errors.append(
411 _error_record(
412 ImageProcessingError(
413 "file_count_exceeded",
414 "파일 개수 제한을 초과한 입력입니다.",
415 details={"limit": file_limit},
416 ),
417 index=index,
418 position=position,
419 filename=filename,
420 recoverable=recoverable,
421 )
422 )
423 continue
424
425 try:
426 content, filename = _read_file_bytes(file_item, position, per_file_bytes)
427 total_read_bytes += len(content)
428 extension = Path(filename).suffix.lower()
429 image_format, mime_type = _validate_image(content, extension, allow_svg=bool(allow_svg))
430
431 if encoded_source_bytes + len(content) > total_limit_bytes:
432 raise ImageProcessingError(
433 "total_size_exceeded",
434 "누적 이미지 크기가 전체 제한을 초과했습니다.",
435 details={
436 "candidate_bytes": len(content),
437 "accepted_bytes": encoded_source_bytes,
438 "limit_bytes": total_limit_bytes,
439 },
440 )
441
442 encoded = base64.b64encode(content).decode("ascii")
443 encoded_value = f"data:{mime_type};base64,{encoded}" if selected_format == "data_url" else encoded
444 items.append(
445 {
446 "index": index,
447 "position": position,
448 "filename": filename,
449 "extension": extension,
450 "image_format": image_format,
451 "mime_type": mime_type,
452 "byte_size": len(content),
453 "sha256": hashlib.sha256(content).hexdigest(),
454 "encoding": selected_format,
455 "value": encoded_value,
456 }
457 )
458 encoded_source_bytes += len(content)
459 if image_format == "svg":
460 warnings.append(
461 {
462 "code": "static_svg_enabled",
463 "message": "정적 SVG 검사에 통과한 파일만 포함했습니다.",
464 "index": index,
465 "position": position,
466 }
467 )
468 except ImageProcessingError as error:
469 if error.byte_size is not None:
470 total_read_bytes += max(0, error.byte_size)
471 errors.append(
472 _error_record(
473 error,
474 index=index,
475 position=position,
476 filename=filename,
477 recoverable=recoverable,
478 )
479 )
480
481 rejected_count = sum(1 for error in errors if isinstance(error.get("position"), int))
482 batch_rejected = selected_policy == "reject_batch" and bool(errors)
483 if batch_rejected:
484 items = []
485 encoded_source_bytes = 0
486
487 success = bool(items) and (selected_policy == "skip_invalid" or not errors)
488 if selected_policy == "skip_invalid" and errors and items:
489 warnings.append(
490 {
491 "code": "invalid_files_skipped",
492 "message": "검증에 실패한 파일을 제외하고 유효한 파일만 변환했습니다.",
493 "skipped_count": rejected_count,
494 }
495 )
496
497 result = {
498 "success": success,
499 "output_format": selected_format,
500 "error_policy": selected_policy,
501 "order_preserved": True,
502 "input_count": len(files),
503 "encoded_count": len(items),
504 "rejected_count": rejected_count,
505 "error_count": len(errors),
506 "batch_rejected": batch_rejected,
507 "total_read_bytes": total_read_bytes,
508 "encoded_source_bytes": encoded_source_bytes,
509 "items": items,
510 "errors": errors,
511 "warnings": warnings,
512 "limits": {
513 "max_files": file_limit,
514 "max_file_size_bytes": per_file_bytes,
515 "max_total_size_bytes": total_limit_bytes,
516 "svg_enabled": bool(allow_svg),
517 },
518 "trace": [
519 {
520 "stage": "multi_image_base64_encoder",
521 "status": "success" if success else "failed",
522 "input_count": len(files),
523 "encoded_count": len(items),
524 "rejected_count": rejected_count,
525 "error_count": len(errors),
526 }
527 ],
528 }
529 return result
