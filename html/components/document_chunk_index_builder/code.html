61def _version_key(value: Any) -> tuple[tuple[int, Any], ...]:
62 """Return a deterministic natural-sort key for common document versions."""
63 text = str(value or "1").strip().lower()
64 parts = re.findall(r"\d+|[^\d]+", text)
65 key: list[tuple[int, Any]] = []
66 for part in parts:
67 cleaned = part.strip("._+- ")
68 if not cleaned:
69 continue
70 if cleaned.isdigit():
71 key.append((1, int(cleaned)))
72 else:
73 key.append((0, cleaned))
74 return tuple(key) or ((1, 1),)

91def _split_text(text: str, chunk_chars: int, overlap_chars: int) -> list[str]:
92 text = _normalized_text(text)
93 if not text:
94 return []
95
96 chunk_chars = max(50, int(chunk_chars))
97 overlap_chars = max(0, min(int(overlap_chars), chunk_chars // 2))
98 chunks: list[str] = []
99 start = 0
100 text_length = len(text)
101 while start < text_length:
102 proposed_end = min(text_length, start + chunk_chars)
103 end = proposed_end
104 if proposed_end < text_length:
105 # Prefer a word boundary in the second half of the window. If no
106 # boundary exists (for example, a long identifier), use the exact
107 # character limit so the loop always makes progress.
108 boundary = text.rfind(" ", start + chunk_chars // 2, proposed_end)
109 if boundary > start:
110 end = boundary
111
112 chunk = text[start:end].strip()
113 if chunk:
114 chunks.append(chunk)
115 if end >= text_length:
116 break
117 next_start = end - overlap_chars
118 start = next_start if next_start > start else end
119 return chunks

198 document_id = str(document.get("document_id") or "").strip()
199 version = str(document.get("version") or "1").strip()

205 chunk_content_hash = hashlib.sha256(chunk.encode("utf-8")).hexdigest()

234def build_document_chunk_index(
235 value: Any,
236 *,
237 chunk_chars: int = 900,
238 overlap_chars: int = 120,
239 max_chunks: int = 5_000,
240 latest_version_only: bool = True,
241) -> dict[str, Any]:
242 """Build a deterministic, payload-local retrieval index.
243
244 This component deliberately does not claim persistent vector-store writes.
245 `operations` describes how a later persistent indexer should replace or
246 delete versions, while `chunks` is immediately usable by the model-free
247 lexical retriever in the example flow.
248 """
249 payload = _payload_from_value(value)
250 raw_documents = payload.get("documents")
251 documents = raw_documents if isinstance(raw_documents, list) else []
252 chunk_chars = _clamp_int(chunk_chars, 900, 50, 20_000)
253 overlap_chars = _clamp_int(overlap_chars, 120, 0, chunk_chars // 2)
254 max_chunks = _clamp_int(max_chunks, 5_000, 1, 50_000)
255 select_latest = _as_bool(latest_version_only, True)
256
257 errors: list[str] = []
258 warnings: list[str] = []
259 if select_latest:
260 selected_documents, delete_operations, invalid_document_count = _latest_document_records(documents)
261 else:
262 selected_documents, delete_operations, invalid_document_count = _all_active_records(documents)
263
264 chunks_by_id: dict[str, dict[str, Any]] = {}
265 source_chunk_count = 0
266 limit_reached = False
267 for document in selected_documents:
268 content = str(document.get("content") or "").strip()
269 if not content:
270 invalid_document_count += 1
271 continue
272 for chunk_index, chunk in enumerate(_split_text(content, chunk_chars, overlap_chars), start=1):
273 source_chunk_count += 1
274 record = _chunk_record(document, chunk, chunk_index)
275 chunks_by_id.setdefault(record["id"], record)
276 if len(chunks_by_id) >= max_chunks:
277 limit_reached = True
278 break
279 if limit_reached:
280 break
281
282 chunks = list(chunks_by_id.values())
283 deduplicated_chunk_count = max(0, source_chunk_count - len(chunks))
284
285 chunk_ids_by_document: dict[tuple[str, str], list[str]] = {}
286 for chunk in chunks:
287 metadata = chunk["metadata"]
288 key = (metadata["document_id"], metadata["version"])
289 chunk_ids_by_document.setdefault(key, []).append(chunk["id"])
290
291 upsert_operations: list[dict[str, Any]] = []
292 for (document_id, version), chunk_ids in sorted(chunk_ids_by_document.items()):
293 upsert_operations.append(
294 {
295 "operation": "replace_document_version" if select_latest else "upsert_document_version",
296 "document_id": document_id,
297 "version": version,
298 "delete_previous_versions": select_latest,
299 "upsert_chunk_ids": chunk_ids,
300 }
301 )
302 operations = delete_operations + upsert_operations
303
304 if not documents:
305 errors.append("No documents were available for chunk indexing.")
306 if invalid_document_count:
307 warnings.append("One or more document records were skipped because required indexing fields were missing.")
308 if deduplicated_chunk_count:
309 warnings.append("Duplicate chunks were removed by their stable chunk IDs.")
310 if limit_reached:
311 warnings.append("The maximum chunk limit was reached; remaining content was not indexed.")
312 if documents and not chunks and delete_operations:
313 warnings.append("The current payload contains deletion or inactive records and no active chunks.")
314 if documents and not chunks and not delete_operations:
315 errors.append("No active text chunks could be built from the document payload.")
316
317 # Deletion-only payloads are valid ingestion plans. They are intentionally
318 # successful even though the model-free current-run index is empty.
319 success = bool(chunks or delete_operations) and not errors
320 selected_document_ids = sorted({chunk["metadata"]["document_id"] for chunk in chunks})
321 selected_versions = {
322 document_id: sorted(
323 {chunk["metadata"]["version"] for chunk in chunks if chunk["metadata"]["document_id"] == document_id},
324 key=_version_key,
325 )[-1]
326 for document_id in selected_document_ids
327 }
328
329 return {
330 "success": success,
331 "stage": "document_chunk_index_builder",
332 "backend": "payload_lexical_v1",
333 "persistence": "ephemeral_current_run",
334 "chunks": chunks,
335 "chunk_count": len(chunks),
336 "document_count": len(selected_document_ids),
337 "selected_versions": selected_versions,
338 "operations": operations,
339 "operation_count": len(operations),
340 "delete_operation_count": len(delete_operations),
341 "deduplicated_chunk_count": deduplicated_chunk_count,
342 "settings": {
343 "chunk_chars": chunk_chars,
344 "overlap_chars": overlap_chars,
345 "max_chunks": max_chunks,
346 "latest_version_only": select_latest,
347 },
348 "errors": errors,
349 "warnings": warnings,
350 "trace": [
351 {
352 "stage": "document_chunk_index_builder",
353 "status": "success" if success else "failed",
354 "backend": "payload_lexical_v1",
355 "input_document_count": len(documents),
356 "indexed_document_count": len(selected_document_ids),
357 "chunk_count": len(chunks),
358 "delete_operation_count": len(delete_operations),
359 "deduplicated_chunk_count": deduplicated_chunk_count,
