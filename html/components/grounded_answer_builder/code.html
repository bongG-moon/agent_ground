37def _text(value: Any) -> str:
38 if value is None:
39 return ""
40 if isinstance(value, str):
41 return value.strip()
42 data = getattr(value, "data", None)
43 if isinstance(data, dict):
44 for key in ("answer", "text", "content", "output"):
45 if isinstance(data.get(key), str):
46 return data[key].strip()
47 for attr in ("text", "content"):
48 candidate = getattr(value, attr, None)
49 if isinstance(candidate, str):
50 return candidate.strip()
51 return str(value).strip()

60def _parse_llm(value: Any) -> dict[str, Any]:
61 if isinstance(value, dict):
62 return value
63 data = getattr(value, "data", None)
64 if isinstance(data, dict) and any(key in data for key in ("answer", "used_evidence_ids", "unsupported")):
65 return data
66 raw = _text(value)
67 if not raw:
68 return {}
69 cleaned = re.sub(r"^```(?:json)?\s*|\s*```$", "", raw.strip(), flags=re.IGNORECASE)
70 candidates = [cleaned]
71 start, end = cleaned.find("{"), cleaned.rfind("}")
72 if 0 <= start < end:
73 candidates.append(cleaned[start : end + 1])
74 for candidate in candidates:
75 try:
76 parsed = json.loads(candidate)
77 if isinstance(parsed, dict):
78 return parsed
79 except Exception:
80 continue
81 ids = re.findall(r"\bE\d+\b", raw)
82 return {"answer": raw, "used_evidence_ids": ids, "unsupported": False}

85def _tokens(text: str) -> set[str]:
86 normalized = unicodedata.normalize("NFKC", text).lower()
87 stop = {"그리고", "그러나", "대한", "관련", "입니다", "합니다", "있습니다", "the", "and", "that", "with"}
88 return {token for token in re.findall(r"[a-z0-9가-힣_]{2,}", normalized) if token not in stop}

92 return set(re.findall(r"(?<![A-Za-z])\d+(?:[.,]\d+)?%?", text))

95def _safe_llm_answer(answer: str, evidence_text: str) -> bool:
96 if not answer or len(answer) > 4000:
97 return False
98 lowered = unicodedata.normalize("NFKC", answer).lower()
99 if any(marker in lowered for marker in ("ignore previous", "system prompt", "developer message", "<script", "시스템 프롬프트")):
100 return False
101 answer_without_ids = re.sub(r"\[?E\d+\]?", " ", answer)
102 answer_tokens = _tokens(answer_without_ids)
103 evidence_tokens = _tokens(evidence_text)
104 if not answer_tokens or not evidence_tokens:
105 return False
106 overlap = len(answer_tokens & evidence_tokens) / max(len(answer_tokens), 1)
107 if overlap < 0.16:
108 return False
109 # Numbers are high-risk factual claims. Every number in a model answer
110 # must occur verbatim in at least one authorized evidence chunk.
111 if not _numbers(answer_without_ids).issubset(_numbers(evidence_text)):
112 return False
113 return True

116def _deterministic_answer(evidence: list[dict[str, Any]]) -> tuple[str, list[str]]:
117 bullets: list[str] = []
118 used: list[str] = []
119 for item in evidence[:2]:
120 evidence_id = str(item.get("evidence_id") or "")
121 content = " ".join(str(item.get("content") or "").split())[:420]
122 if evidence_id and content:
123 bullets.append(f"- {content} [{evidence_id}]")
124 used.append(evidence_id)
125 if not bullets:
126 return "확인 가능한 사내 문서 근거가 충분하지 않아 답변할 수 없습니다. 질문을 구체화하거나 문서 담당자에게 확인해 주세요.", []
127 return "확인된 사내 문서 근거는 다음과 같습니다.\n" + "\n".join(bullets), used

130def build_grounded_answer(gate_value: Any, llm_value: Any = "") -> dict[str, Any]:
131 gate = _payload(gate_value)
132 abstention = str(
133 gate.get("abstention_message")
134 or "확인 가능한 사내 문서 근거가 충분하지 않아 답변할 수 없습니다. 질문을 구체화하거나 문서 담당자에게 확인해 주세요."
135 )
136 if not gate.get("allowed"):
137 return {
138 "contract": "agent_ground.enterprise_rag.answer.v1",
139 "success": True,
140 "supported": False,
141 "answer_text": abstention,
142 "used_evidence_ids": [],
143 "evidence": [],
144 "answer_mode": "abstention",
145 }
146
147 evidence = [item for item in gate.get("evidence", []) if isinstance(item, dict)]
148 allowed_by_id = {
149 str(item.get("evidence_id")): item
150 for item in evidence
151 if str(item.get("evidence_id") or "") in set(str(value) for value in gate.get("citation_allowlist", []))
152 }
153 parsed = _parse_llm(llm_value)
154 raw_ids = parsed.get("used_evidence_ids")
155 if isinstance(raw_ids, str):
156 raw_ids = re.findall(r"\bE\d+\b", raw_ids)
157 if not isinstance(raw_ids, list):
158 raw_ids = []
159 # The model can only narrow the server-created allowlist; it cannot add a
160 # source by naming an arbitrary ID.
161 used_ids: list[str] = []
162 for value in raw_ids:
163 evidence_id = str(value or "").strip()
164 if evidence_id in allowed_by_id and evidence_id not in used_ids:
165 used_ids.append(evidence_id)
166
167 candidate_answer = str(parsed.get("answer") or "").strip()
168 candidate_answer = re.sub(r"\[?E\d+\]?", "", candidate_answer).strip()
169 selected_text = "\n".join(str(allowed_by_id[item].get("content") or "") for item in used_ids)
170 llm_is_grounded = bool(
171 used_ids
172 and not _bool(parsed.get("unsupported"))
173 and _safe_llm_answer(candidate_answer, selected_text)
174 )
175
176 if llm_is_grounded:
177 answer_text = candidate_answer
178 # Evidence markers are reintroduced from the intersected IDs, not from
179 # arbitrary model text. The citation component will map them to [1].
180 answer_text += " " + " ".join(f"[{item}]" for item in used_ids)
181 mode = "llm_grounded"
182 else:
183 answer_text, used_ids = _deterministic_answer(evidence)
184 mode = "deterministic_fallback"
185
186 selected = [allowed_by_id[item] for item in used_ids if item in allowed_by_id]
187 return {
188 "contract": "agent_ground.enterprise_rag.answer.v1",
189 "success": True,
190 "supported": bool(selected),
191 "answer_text": answer_text,
192 "used_evidence_ids": used_ids,
193 "evidence": selected,
194 "answer_mode": mode,
195 }
