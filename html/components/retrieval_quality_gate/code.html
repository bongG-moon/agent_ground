16def _payload(value: Any) -> dict[str, Any]:
17 if value is None:
18 return {}
19 if isinstance(value, dict):
20 return value
21 data = getattr(value, "data", None)
22 if isinstance(data, dict):
23 return data
24 text = getattr(value, "text", None) or getattr(value, "content", None)
25 if isinstance(text, str):
26 try:
27 parsed = json.loads(text)
28 return parsed if isinstance(parsed, dict) else {}
29 except Exception:
30 return {}
31 return {}

34def _dict(value: Any) -> dict[str, Any]:
35 return value if isinstance(value, dict) else {}
36

38def _evidence(value: Any) -> list[dict[str, Any]]:
39 if not isinstance(value, list):
40 return []
41 result: list[dict[str, Any]] = []
42 seen: set[str] = set()
43 for item in value:
44 if not isinstance(item, dict):
45 continue
46 evidence_id = str(item.get("evidence_id") or "").strip()
47 content = str(item.get("content") or "").strip()
48 if not evidence_id or not content or evidence_id in seen:
49 continue
50 try:
51 score = max(0.0, min(1.0, float(item.get("score", 0.0))))
52 except Exception:
53 score = 0.0
54 normalized = dict(item)
55 normalized["evidence_id"] = evidence_id
56 normalized["content"] = content[:6000]
57 normalized["score"] = score
58 result.append(normalized)
59 seen.add(evidence_id)
60 return result

63def apply_quality_gate(retrieval_value: Any) -> dict[str, Any]:
64 retrieval = _payload(retrieval_value)
65 candidates = _evidence(retrieval.get("evidence"))
66 security = _dict(retrieval.get("security"))
67
68 top_score = max((float(item.get("score", 0.0)) for item in candidates), default=0.0)
69 injection_signal = bool(security.get("injection_signal_detected"))
70 retrieval_valid = bool(
71 retrieval.get("success")
72 and security.get("acl_applied_before_scoring") is True
73 and security.get("denied_information_exposed") is False
74 )
75 allowed = bool(retrieval_valid and candidates and top_score >= 0.12 and not injection_signal)
76
77 # Evidence is removed entirely on a failed gate so an accidentally wired
78 # downstream LLM still cannot receive low-confidence or suspicious text.
79 passed_evidence = candidates if allowed else []
80 allowlist = [item["evidence_id"] for item in passed_evidence]
81 return {
82 "contract": "agent_ground.enterprise_rag.gate.v1",
83 "success": True,
84 "request_id": str(retrieval.get("request_id") or "")[:96],
85 "question": str(retrieval.get("question") or "")[:8000],
86 "allowed": allowed,
87 "decision": "answer" if allowed else "abstain",
88 # Deliberately identical for missing evidence, ACL denial, low score,
89 # and document-injection detection. End users cannot probe document
90 # existence by comparing denial messages.
91 "reason_code": "sufficient_authorized_evidence" if allowed else "insufficient_authorized_evidence",
92 "abstention_message": ABSTENTION_MESSAGE,
93 "evidence": passed_evidence,
94 "citation_allowlist": allowlist,
95 "quality": {
96 "minimum_top_score": 0.12,
97 "top_score": round(top_score, 6) if allowed else 0.0,
98 "evidence_count": len(passed_evidence),
99 },
100 "security": {
101 "acl_verified": bool(security.get("acl_applied_before_scoring")),
102 "injection_free": not injection_signal,
103 "fail_closed": True,
104 },
105 "errors": [],
106 "warnings": [],
107 }

119 def build_gate(self) -> Data:
120 result = apply_quality_gate(getattr(self, "retrieval", None))
121 self.status = "답변 생성 허용" if result["allowed"] else "근거 부족으로 답변 중단"
122 return Data(data=result)
