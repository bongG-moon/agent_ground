59def _single_chat_input_id(vertices: Any) -> str:
60 """현재 하위 Flow에서 사용자 질문을 받을 Chat Input ID를 정확히 하나만 찾습니다.
61
62 하위 Flow를 다시 import하면 node ID가 바뀔 수 있으므로 export 시점 ID를
63 저장하지 않고, 실행에 사용할 현재 그래프에서 매번 확인합니다.
64 """
65
66 chat_input_ids = [
67 str(vertex.id)
68 for vertex in list(vertices or [])
69 if (getattr(vertex, "data", {}) or {}).get("type") == "ChatInput"
70 or getattr(vertex, "display_name", "") == "Chat Input"
71 ]
72 if len(chat_input_ids) != 1:
73 raise ValueError("대상 Flow에는 사용자 입력용 Chat Input이 정확히 하나 있어야 합니다.")
74 return chat_input_ids[0]

81 때문에 provider가 하이픈과 물결표를 밑줄로 바꾸더라도 질문이 유실되지 않습니다.
82 """
83
84 node_id = str(chat_input_id or "").strip()

87
88 tool_values = flow_tweak_data.model_dump() if hasattr(flow_tweak_data, "model_dump") else flow_tweak_data

185 async def get_graph(
186 self,
187 flow_name_selected: str | None = None,
188 flow_id_selected: str | None = None,
189 updated_at: str | None = None,
190 ):
191 del flow_id_selected, updated_at
192 flow_name = str(flow_name_selected or getattr(self, "flow_name_selected", "") or "").strip()
193 if not flow_name:
194 raise ValueError("대상 Flow 이름이 필요합니다.")
195 if len(flow_name) > _MAX_FLOW_NAME_CHARS:
196 raise ValueError(f"대상 Flow 이름은 {_MAX_FLOW_NAME_CHARS}자 이하여야 합니다.")
197 if not getattr(self, "user_id", None):
198 raise ValueError("현재 Langflow 사용자 ID를 확인할 수 없습니다.")
199
200 parent_graph = getattr(self, "graph", None)
201 parent_flow_id = str(getattr(parent_graph, "flow_id", "") or "").strip()
202 allow_cross_folder = bool(getattr(self, "allow_cross_folder", False))
203 if allow_cross_folder:
204 candidates = await list_flows(user_id=self.user_id)
205 scope_label = "현재 사용자의 전체 폴더"
206 else:
207 if not parent_flow_id:
208 raise ValueError("같은 폴더를 확인할 부모 Flow ID가 없습니다.")
209 candidates = await list_flows_by_flow_folder(user_id=self.user_id, flow_id=parent_flow_id)
210 scope_label = "부모 Router와 같은 폴더"
211
212 matches: list[dict[str, Any]] = []
213 for candidate in candidates:
214 candidate_data = getattr(candidate, "data", None) or {}
215 if isinstance(candidate_data, dict) and str(candidate_data.get("name") or "") == flow_name:
216 matches.append(candidate_data)
217 if not matches:
218 raise ValueError(f"{scope_label}에서 이름이 정확히 일치하는 Flow를 찾지 못했습니다: {flow_name}")
219 if len(matches) > 1:
220 raise ValueError(f"{scope_label}에 같은 이름의 Flow가 여러 개 있습니다: {flow_name}")
221
222 actual_id = str(matches[0].get("id") or "").strip()
223 actual_updated_at = _as_iso_text(matches[0].get("updated_at"))
224 if not actual_id:
225 raise ValueError(f"대상 Flow의 실제 DB ID를 확인할 수 없습니다: {flow_name}")
226 if parent_flow_id and actual_id == parent_flow_id:
227 raise ValueError("현재 부모 Flow 자신을 하위 Run Flow 도구로 실행할 수 없습니다.")
228
229 self.flow_name_selected = flow_name
230 self.flow_id_selected = actual_id
231 self._attributes["flow_name_selected"] = flow_name
232 self._attributes["flow_id_selected"] = actual_id
233 self._attributes["flow_name_selected_updated_at"] = actual_updated_at
234 self._cached_flow_updated_at = actual_updated_at
235 graph = await super().get_graph(
236 flow_name_selected=flow_name,
237 flow_id_selected=actual_id,
238 updated_at=actual_updated_at,
239 )
240 self._resolved_chat_input_id = _single_chat_input_id(getattr(graph, "vertices", []))
241 return graph

243 def _is_cached_flow_up_to_date(self, cached_flow, updated_at: str | None) -> bool:
244 """microsecond를 보존해 같은 초 안의 Flow 수정도 캐시 무효화에 반영합니다."""
245 cached_timestamp = self._parse_timestamp(getattr(cached_flow, "updated_at", None))
246 current_timestamp = self._parse_timestamp(updated_at)
247 return bool(cached_timestamp and current_timestamp and cached_timestamp >= current_timestamp)

249 @staticmethod
250 def _parse_timestamp(value: str | None) -> datetime | None:
251 """ISO timestamp를 UTC 기준으로 변환하되 microsecond를 제거하지 않습니다."""
252 if not value:
253 return None
254 try:
255 parsed = datetime.fromisoformat(str(value).replace("Z", "+00:00"))
256 except ValueError:
257 return None
258 if parsed.tzinfo is None:
259 return parsed.replace(tzinfo=timezone.utc)
260 return parsed.astimezone(timezone.utc)
