24def publish_html_report(
25 payload_value: Any,
26 report_api_url: Any = "",
27 ttl_hours: Any = "24",
28) -> dict[str, Any]:
29 """HTML 생성 결과를 Report API에 저장하고 링크 정보를 payload에 추가합니다."""
30
31 payload = _payload(payload_value)
32 if not payload:
33 return {"report_publish": {"status": "error", "errors": ["empty payload"]}}
34
35 html_report = _dict(payload.get("html_report"))
36 report_plan = _dict(payload.get("report_plan"))
37 request = _dict(payload.get("request"))
38 html = _first_text(payload, html_report, ["html", "rendered_html", "html_string", "document_html"])
39 if not html.strip():
40 return {**payload, "report_publish": {"status": "error", "errors": ["html is empty"]}}
41
42 base_url = _clean(report_api_url)
43 if not base_url:
44 return {**payload, "report_publish": {"status": "error", "errors": ["report_api_url is empty"]}}
45
46 post_url = base_url.rstrip("/")
47 if not post_url.endswith("/reports"):
48 # 사용자가 `http://127.0.0.1:8010`만 넣어도 자동으로 `/reports`를 붙입니다.
49 post_url = post_url + "/reports"
50
51 create_request = {
52 # server.py의 CreateReportRequest 모델과 같은 형식으로 맞춥니다.
53 "html": html,
54 "title": _clean(html_report.get("title") or report_plan.get("title") or payload.get("title") or "HTML Report"),
55 "question": _clean(request.get("question") or payload.get("question")),
56 "view_request": _clean(request.get("view_request") or payload.get("view_request") or html_report.get("view_request")),
57 "available_datasets": _list(payload.get("available_datasets") or request.get("available_datasets")),
58 "report_plan": report_plan,
59 "ttl_hours": _positive_int(ttl_hours, 24),
60 "filename_hint": _clean(
61 html_report.get("filename_hint")
62 or report_plan.get("filename_hint")
63 or report_plan.get("title")
64 or html_report.get("title")
65 or "report"
66 ),
67 }
68
69 try:
70 response = _post_json(post_url, create_request, timeout=DEFAULT_TIMEOUT_SECONDS)
71 except Exception as exc:
72 # API 서버가 꺼져 있거나 URL이 틀려도 flow 전체가 죽지 않게 오류 payload를 반환합니다.
73 return {**payload, "report_publish": {"status": "error", "errors": [str(exc)], "post_url": post_url}}
74
75 next_html_report = _public_html_report(html_report, response)
76 next_payload = _compact_publish_payload(payload, next_html_report)
77 next_payload["report_publish"] = {
78 "status": "published",
79 "errors": [],
80 "download_url": response.get("download_url", ""),
81 "expires_at": response.get("expires_at", ""),
82 }
83 return next_payload

86def build_report_link_message(payload_value: Any) -> str:
87 """Report API 응답 payload를 사용자가 읽을 수 있는 짧은 링크 메시지로 바꿉니다."""
88
89 payload = _payload(payload_value)
90 html_report = _dict(payload.get("html_report"))
91 publish = _dict(payload.get("report_publish"))
92 title = _clean(html_report.get("title") or "HTML Report")
93 download_url = _clean(html_report.get("download_url") or publish.get("download_url"))
94 expires_at = _clean(html_report.get("expires_at") or publish.get("expires_at"))
95 if download_url:
96 lines = [f"HTML 리포트가 생성되었습니다: {title}", "", f"- 다운로드 링크: {download_url}"]
97 if expires_at:
98 lines.append(f"- 만료 시간: {_format_expires_at(expires_at)}")
99 return "\n".join(lines)
100 if publish.get("status") == "error":
101 errors = "; ".join(str(item) for item in _list(publish.get("errors"))) or "알 수 없는 오류"
102 return f"HTML은 생성했지만 Report API 링크 생성에 실패했습니다: {errors}"
103 return "공유 링크가 아직 없습니다. 04 HTML 렌더링 결과와 Report API URL 연결을 확인하세요."

131def _post_json(url: str, body: dict[str, Any], timeout: int) -> dict[str, Any]:
132 """표준 라이브러리만 사용해 JSON POST 요청을 보냅니다.
133
134 외부 패키지 `requests`가 없어도 동작하도록 `urllib.request`를 사용합니다.
135 """
136
137 data = json.dumps(body, ensure_ascii=False, default=str).encode("utf-8")
138 request = urllib.request.Request(url, data=data, method="POST", headers={"Content-Type": "application/json"})
139 try:
140 with urllib.request.urlopen(request, timeout=timeout) as response:
141 raw = response.read().decode("utf-8")
142 except urllib.error.HTTPError as exc:
143 detail = exc.read().decode("utf-8", errors="replace")
144 raise RuntimeError(f"Report API HTTP {exc.code}: {detail}") from exc
145 parsed = json.loads(raw or "{}")
146 if not isinstance(parsed, dict):
147 raise RuntimeError("Report API did not return a JSON object")
148 return parsed

151def _first_text(payload: dict[str, Any], html_report: dict[str, Any], keys: list[str]) -> str:
152 """여러 후보 key 중 가장 먼저 발견되는 HTML 문자열을 찾습니다."""
153
154 for source in (html_report, payload):
155 for key in keys:
156 value = source.get(key)
157 if isinstance(value, str) and value.strip():
158 return value
159 return ""

162def _payload(value: Any) -> dict[str, Any]:
163 """Langflow Data/Message/dict/JSON 문자열을 일반 dict로 맞춥니다."""
164
165 if isinstance(value, dict):
166 return deepcopy(value)
167 data = getattr(value, "data", None)
168 if isinstance(data, dict):
169 return deepcopy(data)
170 text = getattr(value, "text", None) or getattr(value, "content", None)
171 if isinstance(text, str):
172 try:
173 parsed = json.loads(text)
174 except Exception:
175 return {"text": text}
176 return deepcopy(parsed) if isinstance(parsed, dict) else {"text": text}
177 return {}

198def _format_expires_at(value: Any) -> str:
199 """서버가 준 UTC 만료 시간을 한국 시간 문자열로 바꿉니다."""
200
201 text = _clean(value)
202 if not text:
203 return ""
204 try:
205 parsed = datetime.fromisoformat(text.replace("Z", "+00:00"))
206 except Exception:
207 return text
208 if parsed.tzinfo is None:
209 parsed = parsed.replace(tzinfo=timezone.utc)
210 kst = parsed.astimezone(timezone(timedelta(hours=9)))
211 return kst.strftime("%Y-%m-%d %H:%M KST")

214def _positive_int(value: Any, default: int) -> int:
215 """값을 양의 정수로 바꾸고 실패하면 default를 사용합니다."""
216
217 try:
218 parsed = int(value)
219 except Exception:
220 parsed = default
221 return max(1, parsed)

237 def _published_payload(self) -> dict[str, Any]:
238 """같은 노드 실행 중 API POST가 중복 호출되지 않도록 결과를 캐시합니다."""
239
240 cached = getattr(self, "_cached_publish_result", None)
241 if isinstance(cached, dict):
242 return cached
243 result = publish_html_report(
244 getattr(self, "payload", None),
245 getattr(self, "report_api_url", ""),
246 getattr(self, "ttl_hours", "24"),
247 )
248 self._cached_publish_result = result
249 return result

251 def build_message(self) -> Message:
252 """공유 링크 메시지를 생성하고 Langflow status에 간단한 실행 상태를 표시합니다."""
253
254 result = self._published_payload()
255 publish = result.get("report_publish", {}) if isinstance(result, dict) else {}
