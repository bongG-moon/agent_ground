22def _text(value: Any) -> str:
23 if value is None:
24 return ""
25 if isinstance(value, str):
26 return value.strip()
27 for attribute in ("text", "content"):
28 candidate = getattr(value, attribute, None)
29 if isinstance(candidate, str):
30 return candidate.strip()
31 return str(value).strip()

34def _payload(value: Any) -> dict[str, Any]:
35 if isinstance(value, dict):
36 return value
37 data = getattr(value, "data", None)
38 if isinstance(data, dict):
39 return data
40 text = _text(value)
41 if not text:
42 return {}
43 try:
44 parsed = json.loads(text)
45 except json.JSONDecodeError:
46 return {}
47 return parsed if isinstance(parsed, dict) else {}

59def _catalog_entry(skill_catalog: Any) -> tuple[dict[str, Any] | None, str]:
60 catalog = _payload(skill_catalog)
61 if catalog.get("contract") != CATALOG_CONTRACT:
62 return None, "검증된 데모 Skill 카탈로그가 연결되지 않았습니다."
63 skills = catalog.get("skills")
64 if not isinstance(skills, list):
65 return None, "Skill 카탈로그의 skills 목록을 확인할 수 없습니다."
66 matches = [entry for entry in skills if isinstance(entry, dict) and entry.get("tool_name") == TOOL_NAME]
67 if len(matches) != 1:
68 return None, f"카탈로그에서 {TOOL_NAME} 항목을 정확히 하나 확인해야 합니다."
69 entry = matches[0]
70 if entry.get("enabled") is not True:
71 return None, "회의 후속 조치 Skill이 카탈로그에서 사용 중지되어 있습니다."
72 allowed = entry.get("allowed_actions")
73 forbidden = entry.get("forbidden_actions")
74 if not isinstance(allowed, list) or REQUIRED_ACTION not in allowed:
75 return None, f"카탈로그에 필수 허용 action '{REQUIRED_ACTION}'이 없습니다."
76 if not isinstance(forbidden, list) or set(allowed) & set(forbidden):
77 return None, "카탈로그의 허용 action과 금지 action 정책이 올바르지 않습니다."
78 rules = entry.get("rules")
79 if not isinstance(rules, dict) or str(rules.get("line_format") or "").strip() != "담당자 | 할 일 | YYYY-MM-DD":
80 return None, "회의 후속 조치 Skill의 입력 형식 규칙이 올바르지 않습니다."
81 return entry, ""

108def _is_optional_command_header(line: str, line_number: int) -> bool:

116def run_meeting_action(request: Any, skill_catalog: Any) -> dict[str, Any]:
117 """`담당자 | 할 일 | YYYY-MM-DD` 줄을 순서대로 구조화합니다."""
118
119 request_text = _text(request)
120 entry, catalog_error = _catalog_entry(skill_catalog)
121 if entry is None:
122 return _blocked_result(request_text, catalog_error)
123
124 skill = {
125 "skill_id": str(entry.get("skill_id") or "meeting_action"),
126 "tool_name": TOOL_NAME,
127 "name": str(entry.get("name") or "회의 후속 조치 정리"),
128 }
129 governance = {
130 "authorized": True,
131 "required_action": REQUIRED_ACTION,
132 "catalog_enabled": True,
133 "external_write_performed": False,
134 "external_send_performed": False,
135 "decision_effect": "advisory_only",
136 }
137 if not request_text:
138 return {
139 "status": "needs_input",
140 "skill": skill,
141 "result": {
142 "executed": False,
143 "reason": "'담당자 | 할 일 | YYYY-MM-DD' 형식으로 한 줄 이상 입력해 주세요.",
144 "action_items": [],
145 "count": 0,
146 "invalid_lines": [],
147 "ignored_header_line_count": 0,
148 },
149 "governance": governance,
150 "trace": _request_trace(request_text),
151 "disclaimer": DISCLAIMER,
152 }
153
154 action_items: list[dict[str, Any]] = []
155 invalid_lines: list[dict[str, Any]] = []
156 ignored_header_line_count = 0
157 for line_number, raw_line in enumerate(request_text.splitlines(), start=1):
158 line = raw_line.strip()
159 if not line:
160 continue
161 if _is_optional_command_header(line, line_number):
162 ignored_header_line_count += 1
163 continue
164 fields = [field.strip() for field in line.split("|")]
165 if len(fields) != 3:
166 invalid_lines.append({"line_number": line_number, "reason": "구분자 | 로 나눈 필드가 세 개가 아닙니다."})
167 continue
168 assignee, action, due_date = fields
169 if not assignee:
170 invalid_lines.append({"line_number": line_number, "reason": "담당자가 비어 있습니다."})
171 continue
172 if not action:
173 invalid_lines.append({"line_number": line_number, "reason": "할 일이 비어 있습니다."})
174 continue
175 if len(assignee) > 100:
176 invalid_lines.append({"line_number": line_number, "reason": "담당자는 100자 이하여야 합니다."})
177 continue
178 if len(action) > 500:
179 invalid_lines.append({"line_number": line_number, "reason": "할 일은 500자 이하여야 합니다."})
180 continue
181 if not _valid_due_date(due_date):
182 invalid_lines.append({"line_number": line_number, "reason": "기한은 유효한 YYYY-MM-DD 날짜여야 합니다."})
183 continue
184 action_items.append(
185 {
186 "sequence": len(action_items) + 1,
187 "assignee": assignee,
188 "action": action,
189 "due_date": due_date,
190 "source_line_number": line_number,
191 }
192 )
193
194 ready = bool(action_items) and not invalid_lines
195 if not action_items and not invalid_lines:
196 invalid_lines.append({"line_number": 0, "reason": "처리할 수 있는 내용이 없습니다."})
197 return {
198 "status": "completed" if ready else "needs_input",
199 "skill": skill,
200 "result": {
201 "executed": ready,
202 "format": "담당자 | 할 일 | YYYY-MM-DD",
203 "action_items": action_items,
204 "count": len(action_items),
205 "invalid_lines": invalid_lines,
206 "invalid_line_count": len(invalid_lines),
207 "ignored_header_line_count": ignored_header_line_count,
208 "ready_for_human_review": ready,
209 "reason": "" if ready else "형식이 맞지 않는 줄을 수정한 뒤 다시 요청해 주세요.",
210 },
211 "governance": governance,
212 "trace": _request_trace(request_text),
213 "disclaimer": DISCLAIMER,
214 }
