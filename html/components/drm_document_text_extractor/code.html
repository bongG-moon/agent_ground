199def _read_file_bytes(item: Any, position: int, maximum_bytes: int) -> tuple[bytes, str]:
200 filename = _filename_from_item(item, position)
201 extension = Path(filename).suffix.lower()
202 if extension not in SUPPORTED_EXTENSIONS:
203 supported = ", ".join(sorted(SUPPORTED_EXTENSIONS))
204 raise ValueError(f"지원하지 않는 파일 형식입니다: {filename}. 지원 형식: {supported}")
205
206 path = _path_from_item(item)
207 if path is not None:
208 try:
209 if not path.is_file():
210 raise FileNotFoundError
211 declared_size = int(path.stat().st_size)
212 if declared_size > maximum_bytes:
213 raise ValueError(f"파일이 개별 크기 제한을 초과했습니다: {filename}")
214 with path.open("rb") as handle:
215 content = handle.read(maximum_bytes + 1)
216 except ValueError:
217 raise
218 except (OSError, PermissionError, FileNotFoundError):
219 raise ValueError(f"업로드 파일을 읽을 수 없습니다: {filename}") from None
220 else:
221 stream = item.get("file") if isinstance(item, dict) else getattr(item, "file", item)
222 if isinstance(stream, (bytes, bytearray, memoryview)):
223 content = bytes(stream[: maximum_bytes + 1])
224 elif hasattr(stream, "read"):
225 original_position: int | None = None
226 try:
227 if hasattr(stream, "tell"):
228 original_position = int(stream.tell())
229 if hasattr(stream, "seek"):
230 stream.seek(0)
231 content = stream.read(maximum_bytes + 1)
232 except Exception:
233 raise ValueError(f"업로드 파일을 읽을 수 없습니다: {filename}") from None
234 finally:
235 if original_position is not None and hasattr(stream, "seek"):
236 try:
237 stream.seek(original_position)
238 except Exception:
239 pass
240 if not isinstance(content, (bytes, bytearray, memoryview)):
241 raise ValueError(f"업로드 파일이 바이너리 형식이 아닙니다: {filename}")
242 content = bytes(content)
243 else:
244 raise ValueError(f"지원하지 않는 파일 입력 형식입니다: {filename}")
245
246 if len(content) > maximum_bytes:
247 raise ValueError(f"파일이 개별 크기 제한을 초과했습니다: {filename}")
248 if not content:
249 raise ValueError(f"빈 파일은 처리할 수 없습니다: {filename}")
250 return content, filename

253def _validate_local_text(text: str, filename: str, maximum_chars: int) -> str:
254 normalized = text.replace("\x00", "").strip()
255 if not normalized:
256 raise LocalTextExtractionError(f"로컬에서 읽을 수 있는 텍스트가 없습니다: {filename}")

260 1 for character in normalized if ord(character) < 32 and character not in {"\n", "\r", "\t"}

264 return normalized

276def _extract_local_text(content: bytes, filename: str, maximum_chars: int) -> tuple[str, str]:
277 """현 Langflow Desktop에 포함된 parser로 일반 파일의 텍스트를 추출합니다."""
278
279 extension = Path(filename).suffix.lower()
280 if extension not in LOCAL_EXTRACTION_EXTENSIONS:
281 raise LocalTextExtractionError(
282 f"이 형식은 로컬 추출을 지원하지 않아 DRM API 처리가 필요합니다: {filename}"
283 )
284
285 try:
286 if extension == ".pdf":
287 pdf_reader = import_module("pypdf").PdfReader(io.BytesIO(content))
288 if bool(getattr(pdf_reader, "is_encrypted", False)):
289 raise LocalTextExtractionError(f"암호화되었거나 보호된 PDF입니다: {filename}")
290 text = "\n\n".join(page.extract_text() or "" for page in pdf_reader.pages)
291 parser_name = "local-pypdf"
292 elif extension == ".docx":
293 document = import_module("docx").Document(io.BytesIO(content))
294 blocks = [paragraph.text for paragraph in document.paragraphs if paragraph.text.strip()]
295 for table in document.tables:
296 for row in table.rows:
297 values = [cell.text.strip() for cell in row.cells]
298 if any(values):
299 blocks.append("\t".join(values))
300 text = "\n\n".join(blocks)
301 parser_name = "local-python-docx"
302 elif extension == ".pptx":
303 presentation = import_module("pptx").Presentation(io.BytesIO(content))
304 blocks: list[str] = []
305 for slide_index, slide in enumerate(presentation.slides, start=1):
306 slide_blocks: list[str] = []
307 for shape in slide.shapes:
308 shape_text = str(getattr(shape, "text", "") or "").strip()
309 if shape_text:
310 slide_blocks.append(shape_text)
311 if bool(getattr(shape, "has_table", False)):
312 for row in shape.table.rows:
313 values = [cell.text.strip() for cell in row.cells]
314 if any(values):
315 slide_blocks.append("\t".join(values))
316 if slide_blocks:
317 blocks.append(f"[SLIDE {slide_index}]\n" + "\n".join(slide_blocks))
318 text = "\n\n".join(blocks)
319 parser_name = "local-python-pptx"
320 elif extension == ".xlsx":
321 workbook = import_module("openpyxl").load_workbook(
322 io.BytesIO(content), read_only=True, data_only=True
323 )
324 try:
325 blocks = []
326 current_chars = 0
327 for worksheet in workbook.worksheets:
328 sheet_lines = [f"[SHEET] {worksheet.title}"]
329 for row in worksheet.iter_rows(values_only=True):
330 values = ["" if value is None else str(value) for value in row]
331 if not any(value.strip() for value in values):
332 continue
333 line = "\t".join(values).rstrip()
334 current_chars += len(line) + 1
335 if current_chars > maximum_chars:
336 raise LocalTextExtractionError(
337 f"로컬 추출 텍스트가 설정한 최대 크기를 초과했습니다: {filename}"
338 )
339 sheet_lines.append(line)
340 blocks.append("\n".join(sheet_lines))
341 text = "\n\n".join(blocks)
342 finally:
343 workbook.close()
344 parser_name = "local-openpyxl"
345 else:
346 text, parser_name = _decode_local_text(content, filename)
347 except LocalTextExtractionError:
348 raise
349 except ModuleNotFoundError:
350 raise LocalTextExtractionError(
351 f"로컬 추출에 필요한 Langflow 패키지가 설치되어 있지 않습니다: {filename}"
352 ) from None
353 except Exception:
354 raise LocalTextExtractionError(
355 f"로컬에서 파일을 해석하지 못했습니다. DRM 파일일 수 있습니다: {filename}"
356 ) from None
357
358 return _validate_local_text(text, filename, maximum_chars), parser_name

377def _parse_api_url(value: Any, allow_insecure_http: Any) -> tuple[str, bool]:
378 url = _text_value(value).strip()
379 if not url:
380 raise ValueError("DRM API 주소를 입력해 주세요.")
381 try:
382 parsed = urlsplit(url)
383 _ = parsed.port
384 except ValueError as exc:
385 raise ValueError("DRM API 주소의 포트 형식이 올바르지 않습니다.") from exc
386 scheme = parsed.scheme.lower()
387 if scheme not in {"http", "https"} or not parsed.hostname:
388 raise ValueError("DRM API 주소는 서버가 포함된 http 또는 https URL이어야 합니다.")
389 if parsed.username is not None or parsed.password is not None:
390 raise ValueError("DRM API URL 안에 사용자 이름이나 비밀번호를 넣을 수 없습니다.")
391 if parsed.fragment:
392 raise ValueError("DRM API URL에는 #fragment를 넣을 수 없습니다.")
393 allow_http = _bool_value(allow_insecure_http, "HTTP DRM API 사용 허용")
394 if scheme == "http" and not allow_http:
395 raise ValueError(
396 "문서와 인증값을 HTTP로 보낼 수 없습니다. HTTPS URL을 사용하거나, "
397 "폐쇄된 사내 테스트망에서만 'HTTP DRM API 사용 허용'을 켜 주세요."
398 )
399 return url, scheme == "https"

402def _header_value(response: Any, name: str) -> str:
403 headers = getattr(response, "headers", None)
404 if not hasattr(headers, "get"):
405 return ""
406 direct = headers.get(name)
407 if direct is not None:
408 return str(direct)
409 lowered = name.lower()
410 for key, value in getattr(headers, "items", lambda: [])():
411 if str(key).lower() == lowered:
412 return str(value)
413 return ""

416def _read_response_bytes(response: Any, maximum_bytes: int) -> bytes:
417 content_length = _header_value(response, "Content-Length").strip()
418 if content_length:
419 try:
420 declared = int(content_length)
421 except ValueError:
422 declared = 0
423 if declared > maximum_bytes:
424 raise DrmTextExtractionError("DRM API 응답이 설정한 최대 크기를 초과했습니다.")
425
426 iterator = getattr(response, "iter_content", None)
427 if callable(iterator):
428 chunks: list[bytes] = []
429 total = 0
430 for chunk in iterator(chunk_size=CHUNK_SIZE):
431 if not chunk:
432 continue
433 raw_chunk = bytes(chunk)
434 total += len(raw_chunk)
435 if total > maximum_bytes:
436 raise DrmTextExtractionError("DRM API 응답이 설정한 최대 크기를 초과했습니다.")
437 chunks.append(raw_chunk)
438 return b"".join(chunks)
439
440 raw = getattr(response, "content", b"") or b""
441 content = raw.encode("utf-8") if isinstance(raw, str) else bytes(raw)
442 if len(content) > maximum_bytes:
443 raise DrmTextExtractionError("DRM API 응답이 설정한 최대 크기를 초과했습니다.")
444 return content

447def _decode_text_response(response: Any, raw: bytes) -> tuple[str, str]:
448 content_type = _header_value(response, "Content-Type")
449 match = re.search(r"charset\s*=\s*[\"']?([^;\s\"']+)", content_type, flags=re.I)
450 candidates: list[str] = []
451 if match:
452 candidates.append(match.group(1))
453 try:
454 apparent = str(getattr(response, "apparent_encoding", "") or "").strip()
455 except Exception:
456 # requests의 stream 응답은 iter_content 이후 apparent_encoding 접근이 실패할 수 있다.
457 apparent = ""
458 if apparent:
459 candidates.append(apparent)
460 candidates.extend(["utf-8-sig", "cp949", "euc-kr"])
461
462 checked: set[str] = set()
463 for encoding in candidates:
464 key = encoding.lower()
465 if key in checked:
466 continue
467 checked.add(key)
468 try:
469 return raw.decode(encoding), encoding
470 except (LookupError, UnicodeDecodeError):
471 continue
472 return raw.decode("utf-8", errors="replace"), "utf-8-replace"

481def _post_file(
482 *,
483 client: Any,
484 url: str,
485 token: str,
486 employee_no: str,
487 filename: str,
488 content: bytes,
489 timeout_seconds: int,
490 verify_tls: bool,
491 max_response_bytes: int,
492) -> dict[str, Any]:
493 sender = getattr(client, "post", None)
494 if not callable(sender):
495 raise ValueError("HTTP 전송 객체에 post 함수가 없습니다.")
496
497 with io.BytesIO(content) as stream:
498 options = {
499 "files": {"file": (filename, stream, "application/octet-stream")},
500 "headers": {"Authorization": f"Bearer {token}"},
501 "params": {"empNo": employee_no},
502 "timeout": timeout_seconds,
503 "verify": verify_tls,
504 "allow_redirects": False,
505 "stream": True,
506 }
507 try:
508 if verify_tls:
509 response = sender(url, **options)
510 else:
511 with warnings.catch_warnings():
512 warnings.simplefilter("ignore")
513 response = sender(url, **options)
514 except Exception:
515 raise DrmTextExtractionError(f"DRM API에 연결하지 못했습니다: {filename}") from None
516
517 try:
518 status_code = int(getattr(response, "status_code", 0) or 0)
519 if status_code < 200 or status_code >= 300:
520 raise DrmTextExtractionError(
521 f"DRM API가 성공 상태를 반환하지 않았습니다: {filename} · HTTP {status_code}"
522 )
523 raw = _read_response_bytes(response, max_response_bytes)
524 if not raw:
525 raise DrmTextExtractionError(f"DRM API가 빈 텍스트를 반환했습니다: {filename}")
526 text, encoding = _decode_text_response(response, raw)
527 if not text.strip():
528 raise DrmTextExtractionError(f"DRM API가 빈 텍스트를 반환했습니다: {filename}")
529 return {
530 "file_name": filename,
531 "extension": Path(filename).suffix.lower(),
532 "source_bytes": len(content),
533 "response_bytes": len(raw),
534 "char_count": len(text),
535 "encoding": encoding,
536 "http_status": status_code,
537 "processing_path": "drm_api",
538 "drm_status": "text_extracted",
539 "text": text,
540 }
541 finally:
542 _close_response(response)

586 file_limit = _bounded_int(max_files, "최대 파일 수", 1, MAX_FILES)

588 raise ValueError("처리할 문서 또는 이미지 파일을 한 개 이상 입력해 주세요.")
589 if len(files) > file_limit:
590 raise ValueError(f"업로드 파일 수가 최대 {file_limit}개를 초과했습니다.")
591
592 timeout = _bounded_int(timeout_seconds, "제한 시간", 1, MAX_TIMEOUT_SECONDS)
593 per_file_bytes = _bounded_int(
594 max_file_size_mb, "파일당 최대 크기", 1, MAX_FILE_SIZE_MB
595 ) * 1024 * 1024
596 total_limit_bytes = _bounded_int(
597 max_total_size_mb, "전체 최대 크기", 1, MAX_TOTAL_SIZE_MB
598 ) * 1024 * 1024
599 response_limit_bytes = _bounded_int(
600 max_response_mb, "파일당 최대 응답 크기", 1, MAX_RESPONSE_MB
601 ) * 1024 * 1024
602
603 prepared: dict[int, tuple[bytes, str]] = {}
604 skipped_results: dict[int, dict[str, Any]] = {}
605 total_bytes = 0
606 for position, item in enumerate(files, start=1):
607 filename = _filename_from_item(item, position)
608 if Path(filename).suffix.lower() not in SUPPORTED_EXTENSIONS:
609 skipped_results[position - 1] = _unsupported_result(filename)
610 continue
611 content, filename = _read_file_bytes(item, position, per_file_bytes)
612 total_bytes += len(content)
613 if total_bytes > total_limit_bytes:
614 raise ValueError("업로드 파일의 전체 크기가 설정한 제한을 초과했습니다.")
615 prepared[position - 1] = (content, filename)
616
617 local_results: dict[int, dict[str, Any]] = {}
618 drm_indexes: list[int] = []
619 for index, (content, filename) in prepared.items():
620 if mode == PROCESSING_MODE_ALWAYS_DRM:
621 drm_indexes.append(index)
622 continue
623 try:
624 local_results[index] = _local_result(content, filename, response_limit_bytes)
625 except LocalTextExtractionError:
626 if mode == PROCESSING_MODE_BYPASS_DRM:
627 raise ValueError(
628 f"DRM 미사용 모드에서 로컬 텍스트 추출에 실패했습니다: {filename}"
629 ) from None
630 drm_indexes.append(index)
631
632 drm_results: dict[int, dict[str, Any]] = {}
633 if drm_indexes:
634 token = _text_value(drm_token).strip()
635 emp_no = _text_value(employee_no).strip()
636 if not token:
637 raise ValueError("DRM API 호출이 필요한 파일이 있어 DRM 토큰을 입력해야 합니다.")
638 if not emp_no:
639 raise ValueError("DRM API 호출이 필요한 파일이 있어 사번을 입력해야 합니다.")
640 if len(token) > 8192 or "\r" in token or "\n" in token:
641 raise ValueError("DRM 토큰 형식이 올바르지 않습니다.")
642 if len(emp_no) > 128 or "\r" in emp_no or "\n" in emp_no:
643 raise ValueError("사번 형식이 올바르지 않습니다.")
644
645 url, is_https = _parse_api_url(drm_api_url, allow_insecure_http)
646 tls_verification = _bool_value(verify_tls, "TLS 인증서 검증")
647 if not is_https:
648 tls_verification = False
649 try:
650 client = transport or import_module("requests")
651 except ModuleNotFoundError as exc:
652 raise ValueError("DRM API 호출에 필요한 requests 패키지가 설치되어 있지 않습니다.") from exc
653
654 for index in drm_indexes:
655 content, filename = prepared[index]
656 drm_results[index] = _post_file(
657 client=client,
658 url=url,
659 token=token,
660 employee_no=emp_no,
661 filename=filename,
662 content=content,
663 timeout_seconds=timeout,
664 verify_tls=tls_verification,
665 max_response_bytes=response_limit_bytes,
666 )
667
668 results = [
669 skipped_results.get(index) or local_results.get(index) or drm_results[index]
670 for index in range(len(files))
671 ]
672 return {
673 "success": True,
674 "data": results,
675 "errors": [],
676 "meta": {
677 "file_count": len(results),
678 "total_source_bytes": total_bytes,
679 "total_response_bytes": sum(item["response_bytes"] for item in results),
680 "total_char_count": sum(item["char_count"] for item in results),
681 "processing_mode": mode,
682 "local_file_count": sum(item["processing_path"] == "local" for item in results),
683 "drm_file_count": sum(item["processing_path"] == "drm_api" for item in results),
684 "skipped_file_count": sum(
685 item["processing_path"] == "skipped_unsupported" for item in results
686 ),
687 "network_called": bool(drm_indexes),
688 "order_preserved": True,
689 "redirects_allowed": False,
690 "response_type": "plain_text",
691 },
692 }

903 content = getattr(response, "content", response)

914 return str(content or "").strip()
