54def _validated_url(value: Any, *, allow_insecure_http: bool = False) -> str:
55 """호출 가능한 HTTP(S) URL인지 확인합니다."""
56 url = _text_value(value).strip()
57 if not url:
58 raise ValueError("H-API URL을 입력해 주세요.")
59 try:
60 parsed = urlsplit(url)
61 _ = parsed.port
62 except ValueError as exc:
63 raise ValueError("H-API URL의 포트 형식이 올바르지 않습니다.") from exc
64 if parsed.scheme.lower() not in {"http", "https"}:
65 raise ValueError("H-API URL은 http 또는 https 형식이어야 합니다.")
66 if not parsed.hostname:
67 raise ValueError("H-API URL에서 서버 주소를 확인할 수 없습니다.")
68 if parsed.username is not None or parsed.password is not None:
69 raise ValueError("URL 안에 사용자 이름이나 비밀번호를 넣을 수 없습니다.")
70 if parsed.fragment:
71 raise ValueError("H-API URL에는 #fragment를 넣을 수 없습니다.")
72 if parsed.scheme.lower() == "http" and not allow_insecure_http:
73 raise ValueError(
74 "H-API 인증 토큰을 HTTP로 보낼 수 없습니다. HTTPS URL을 사용하거나, "
75 "폐쇄된 테스트망에서만 'HTTP API 사용 허용'을 명시적으로 켜 주세요."
76 )
77 return url

82 parsed = urlsplit(url)

103def _extract_path(payload: Any, response_path: str) -> Any:
104 """점으로 구분한 응답 경로를 엄격하게 따라가 실제 데이터만 꺼냅니다."""
105 path = str(response_path or "").strip()
106 if not path:
107 return payload
108
109 current = payload
110 for token in [part.strip() for part in path.split(".") if part.strip()]:
111 if isinstance(current, dict):
112 if token not in current:
113 raise ValueError(f"응답 경로 '{path}'에서 '{token}' 키를 찾을 수 없습니다.")
114 current = current[token]
115 elif isinstance(current, list) and token.isdigit():
116 index = int(token)
117 if index >= len(current):
118 raise ValueError(f"응답 경로 '{path}'의 배열 위치 {index}가 범위를 벗어났습니다.")
119 current = current[index]
120 else:
121 raise ValueError(f"응답 경로 '{path}'를 끝까지 따라갈 수 없습니다.")
122 return current

132def _rows_from_json(value: Any) -> list[dict[str, Any]]:
133 """JSON 값 하나를 DataFrame에 바로 넣을 수 있는 행 목록으로 변환합니다."""
134 if value is None:
135 return []
136 items = value if isinstance(value, list) else [value]
137 rows: list[dict[str, Any]] = []
138 for item in items:
139 if isinstance(item, dict):
140 rows.append({str(key): _table_cell(cell) for key, cell in item.items()})
141 else:
142 rows.append({"value": _table_cell(item)})
143 return rows

146def _header_value(response: Any, name: str) -> str:
147 """응답 헤더를 대소문자 차이와 관계없이 읽습니다."""
148 headers = getattr(response, "headers", None)
149 if not hasattr(headers, "get"):
150 return ""
151 direct = headers.get(name)
152 if direct is not None:
153 return str(direct)
154 lowered = name.lower()
155 for key, value in getattr(headers, "items", lambda: [])():
156 if str(key).lower() == lowered:
157 return str(value)
158 return ""

161def _read_response_bytes(response: Any, maximum_bytes: int) -> bytes:
162 """응답을 제한 크기까지만 읽어 과도한 메모리 사용을 막습니다."""
163 content_length = _header_value(response, "Content-Length").strip()
164 if content_length:
165 try:
166 declared_size = int(content_length)
167 except ValueError:
168 declared_size = 0
169 if declared_size > maximum_bytes:
170 raise ValueError(f"H-API 응답이 허용 크기 {maximum_bytes:,}바이트를 초과합니다.")
171
172 iterator = getattr(response, "iter_content", None)
173 if callable(iterator):
174 chunks: list[bytes] = []
175 total = 0
176 for chunk in iterator(chunk_size=64 * 1024):
177 if not chunk:
178 continue
179 chunk_bytes = bytes(chunk)
180 total += len(chunk_bytes)
181 if total > maximum_bytes:
182 raise ValueError(f"H-API 응답이 허용 크기 {maximum_bytes:,}바이트를 초과합니다.")
183 chunks.append(chunk_bytes)
184 return b"".join(chunks)
185
186 raw = getattr(response, "content", b"") or b""
187 raw_bytes = raw.encode("utf-8") if isinstance(raw, str) else bytes(raw)
188 if len(raw_bytes) > maximum_bytes:
189 raise ValueError(f"H-API 응답이 허용 크기 {maximum_bytes:,}바이트를 초과합니다.")
190 return raw_bytes

193def _json_response(response: Any) -> Any:
194 """HTTP 응답 본문을 JSON으로만 해석합니다."""
195 raw = _read_response_bytes(response, MAX_RESPONSE_BYTES)
196 if not raw:
197 return None
198 content_type = _header_value(response, "Content-Type").lower()
199 if content_type and "json" not in content_type:
200 raise ValueError("H-API 응답의 Content-Type이 JSON이 아닙니다.")
201 try:
202 return json.loads(raw)
203 except (json.JSONDecodeError, UnicodeDecodeError) as exc:
204 raise ValueError("H-API 응답 본문이 올바른 JSON 형식이 아닙니다.") from exc

214def request_h_api_table(
215 api_url: Any,
216 h_api_token: Any,
217 bind_params_json: Any,
218 response_path: Any = "data.row",
219 timeout_seconds: Any = 30,
220 max_rows: Any = 5000,
221 allow_insecure_http: Any = False,
222 transport: Any | None = None,
223) -> list[dict[str, Any]]:
224 """H-API를 한 번 호출하고 응답 데이터 행만 반환합니다.
225
226 transport는 테스트에서 가짜 HTTP 전송 객체를 주입하기 위한 선택 인자입니다.
227 실제 Component 실행에서는 requests 모듈을 사용합니다.
228 """
229 allow_http = _bool_value(allow_insecure_http, "HTTP API 사용 허용")
230 url = _validated_url(api_url, allow_insecure_http=allow_http)
231 token = _text_value(h_api_token).strip()
232 if not token:
233 raise ValueError("H-API Token을 입력해 주세요.")
234 bind_params = _parse_bind_params(bind_params_json)
235 timeout = _positive_int(timeout_seconds, "제한 시간", MAX_TIMEOUT_SECONDS)
236 row_limit = _positive_int(max_rows, "최대 반환 행 수", MAX_ROWS)
237
238 try:
239 client = transport or import_module("requests")
240 except ModuleNotFoundError as exc:
241 raise ValueError("H-API 호출에 필요한 requests 패키지가 설치되어 있지 않습니다.") from exc
242 sender = getattr(client, "request", None)
243 if not callable(sender):
244 raise ValueError("HTTP 전송 객체에 request 함수가 없습니다.")
245
246 try:
247 response = sender(
248 "POST",
249 url,
250 headers={
251 "h-api-token": token,
252 "Accept": "application/json",
253 "Content-Type": "application/json",
254 },
255 json={"bindParams": bind_params},
256 timeout=timeout,
257 verify=True,
258 allow_redirects=False,
259 stream=True,
260 )
261 except Exception:
262 raise ValueError("H-API 서버에 연결하지 못했습니다.") from None
263
264 try:
265 status_code = int(getattr(response, "status_code", 0) or 0)
266 if status_code < 200 or status_code >= 300:
267 if 300 <= status_code < 400:
268 raise ValueError("H-API가 다른 주소로 이동을 요청했습니다. 고정된 최종 URL을 입력해 주세요.")
269 raise ValueError(f"H-API가 성공 상태를 반환하지 않았습니다: HTTP {status_code}")
270
271 final_url = str(getattr(response, "url", url) or url)
272 if _url_origin(final_url) != _url_origin(url):
273 raise ValueError("H-API 응답의 최종 origin(서버·scheme·port)이 입력 주소와 다릅니다.")
274
275 payload = _json_response(response)
276 selected = _extract_path(payload, _text_value(response_path))
277 return _rows_from_json(selected)[:row_limit]
278 finally:
279 _close_response(response)
