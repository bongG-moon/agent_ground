192def _normalize_theme(value: Any, warnings: list[dict[str, Any]]) -> dict[str, str]:
193 raw = _dict(value)
194 colors = _dict(raw.get("colors"))
195 if colors:
196 # Flow의 design_system.colors와 독립 Component의 평면 theme 계약을 모두 받습니다.
197 raw = {**raw, **colors}
198 defaults = {
199 "background": "#0f172a",
200 "surface": "#ffffff",
201 "text": "#0f172a",
202 "muted": "#475569",
203 "primary": "#2563eb",
204 "accent": "#0f766e",
205 "danger": "#dc2626",
206 "chart_1": "#2563eb",
207 "chart_2": "#0f766e",
208 "chart_3": "#f59e0b",
209 "chart_4": "#7c3aed",
210 }
211 aliases = {
212 "background": ("background", "background_color"),
213 "surface": ("surface", "surface_color"),
214 "text": ("text", "text_color"),
215 "muted": ("muted", "muted_text", "muted_text_color"),
216 "primary": ("primary", "primary_color"),
217 "accent": ("accent", "accent_color"),
218 "danger": ("danger", "danger_color"),
219 "chart_1": ("chart_1",),
220 "chart_2": ("chart_2",),
221 "chart_3": ("chart_3",),
222 "chart_4": ("chart_4",),
223 }
224 result: dict[str, str] = {}
225 for key, keys in aliases.items():
226 candidate = next((raw.get(alias) for alias in keys if raw.get(alias) not in (None, "")), None)
227 if candidate is None:
228 result[key] = defaults[key]
229 continue
230 text = str(candidate).strip()
231 if _COLOR_RE.fullmatch(text):
232 result[key] = text.lower()
233 else:
234 result[key] = defaults[key]
235 warnings.append(
236 {
237 "code": "invalid_theme_color_replaced",
238 "message": "안전하지 않거나 지원하지 않는 색상을 기본값으로 교체했습니다.",
239 "field": key,
240 }
241 )
242
243 typography = _dict(raw.get("typography"))
244 font_value = str(
245 raw.get("font_family")
246 or raw.get("font")
247 or typography.get("body_family")
248 or "sans"
249 ).strip().lower()
250 if "mono" in font_value or "consol" in font_value:
251 font_key = "mono"
252 elif "serif" in font_value and "sans" not in font_value:
253 font_key = "serif"
254 else:
255 font_key = "sans"
256 result["font_stack"] = _FONT_STACKS.get(font_key, _FONT_STACKS["sans"])
257 known_font_tokens = (
258 "sans",
259 "serif",
260 "mono",
261 "arial",
262 "pretendard",
263 "noto",
264 "segoe",
265 "georgia",
266 "times",
267 "consolas",
268 "cascadia",
269 )
270 if not any(token in font_value for token in known_font_tokens):
271 warnings.append(
272 {
273 "code": "unsupported_font_replaced",
274 "message": "외부 글꼴을 불러오지 않고 시스템 기본 글꼴을 사용합니다.",
275 }
276 )
277 return result

303def _normalize_slide(
304 slide: dict[str, Any],
305 index: int,
306 used_ids: set[str],
307 warnings: list[dict[str, Any]],
308) -> dict[str, Any]:
309 raw_id = _safe_id(slide.get("slide_id") or slide.get("id") or f"slide-{index + 1}")
310 slide_id = raw_id or f"slide-{index + 1}"
311 if slide_id in used_ids:
312 slide_id = f"{slide_id}-{index + 1}"
313 used_ids.add(slide_id)
314
315 layout = str(slide.get("layout") or "content").strip().lower().replace("-", "_")
316 layout = _LAYOUT_ALIASES.get(layout, layout)
317 if layout not in _LAYOUTS:
318 warnings.append(
319 {
320 "code": "unsupported_layout_replaced",
321 "message": "지원하지 않는 슬라이드 레이아웃을 content로 교체했습니다.",
322 "position": index + 1,
323 "layout": layout,
324 }
325 )
326 layout = "content"
327
328 elements = _slide_elements(slide)
329 if len(elements) > MAX_ELEMENTS_PER_SLIDE:
330 warnings.append(
331 {
332 "code": "element_count_truncated",
333 "message": f"한 슬라이드의 요소를 {MAX_ELEMENTS_PER_SLIDE}개로 제한했습니다.",
334 "position": index + 1,
335 }
336 )
337 elements = elements[:MAX_ELEMENTS_PER_SLIDE]
338
339 return {
340 "slide_id": slide_id,
341 "layout": layout,
342 "design_role": _safe_id(slide.get("design_role")) or "framing",
343 "visual_weight": _safe_id(slide.get("visual_weight")) or "balanced",
344 "key_message": _text(slide.get("key_message"), "", 800),
345 "eyebrow": _text(slide.get("eyebrow") or slide.get("section"), "", 120),
346 "title": _text(slide.get("title"), f"슬라이드 {index + 1}", 240),
347 "subtitle": _text(slide.get("subtitle"), "", 480),
348 "footer": _text(slide.get("footer"), "", 180),
349 "elements": elements,
350 }

353def _slide_elements(slide: dict[str, Any]) -> list[dict[str, Any]]:
354 elements = slide.get("elements")
355 if isinstance(elements, list):
356 return [deepcopy(item) for item in elements if isinstance(item, dict)]
357
358 inferred: list[dict[str, Any]] = []
359 content = slide.get("content") or slide.get("body")
360 if isinstance(content, str) and content.strip():
361 inferred.append({"type": "text", "text": content})
362 elif isinstance(content, list):
363 inferred.append({"type": "bullets", "items": content})
364 if isinstance(slide.get("bullets"), list):
365 inferred.append({"type": "bullets", "items": slide["bullets"]})
366 if isinstance(slide.get("kpis"), list):
367 inferred.append({"type": "kpi_grid", "items": slide["kpis"]})
368 if isinstance(slide.get("table"), dict):
369 inferred.append({"type": "table", **deepcopy(slide["table"])})
370 if isinstance(slide.get("chart"), dict):
371 inferred.append(deepcopy(slide["chart"]))
372 return inferred

707def _render_element(
708 element: dict[str, Any],
709 *,
710 slide_index: int,
711 element_index: int,
712 theme: dict[str, str],
713 warnings: list[dict[str, Any]],
714) -> str:
715 element_type = str(element.get("type") or element.get("element_type") or "text").strip().lower().replace("-", "_")
716 width = _WIDTHS.get(str(element.get("width") or "full").strip().lower(), _WIDTHS["full"])
717 title = _text(element.get("title"), "", 180)
718 title_markup = f'<h2 class="element-title">{html.escape(title)}</h2>' if title else ""
719 card = bool(element.get("card", element_type not in {"text", "bullets", "bullet_list", "quote", "kpi", "kpis", "kpi_grid", "metrics"}))
720 classes = f"slide-element {width}" + (" element-card" if card else "")

723 if element_type in {"text", "paragraph"}:

728 f"<li>{html.escape(_item_text(item, 900))}</li>" for item in items

730 elif element_type in {"kpi", "kpis", "kpi_grid", "metrics"}:
731 body = _render_kpis(element)
732 elif element_type == "table":
733 body = _render_table(element)
734 elif element_type in {"bar", "bar_chart", "column_chart"}:
735 body = _render_bar_chart(element, slide_index, element_index, theme)
736 elif element_type in {"line", "line_chart", "trend_line_chart"}:
737 body = _render_line_chart(element, slide_index, element_index)
738 elif element_type in {"scatter", "scatter_chart", "scatter_plot"}:
739 body = _render_scatter_chart(element, slide_index, element_index)
740 elif element_type in {"callout", "insight"}:
741 body = f'<div class="callout">{html.escape(_text(element.get("text") or element.get("content"), "", 3000))}</div>'
742 elif element_type == "quote":
743 body = f'<blockquote class="quote">{html.escape(_text(element.get("text") or element.get("content"), "", 3000))}</blockquote>'
744 elif element_type == "image":
745 body = _render_image(element, warnings, slide_index + 1)

749 "code": "unsupported_element_skipped",

757 return ""
