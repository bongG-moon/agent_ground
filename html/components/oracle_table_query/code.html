60def _sql_guard_text(sql: str) -> str:
61 """주석과 문자열 literal을 공백으로 가린 SQL을 만들어 안전성 검사에 사용합니다.
62
63 반환 문자열의 길이를 원문과 같게 유지하므로 허용 가능한 마지막 세미콜론의
64 위치를 원문에서도 동일하게 찾을 수 있습니다.
65 """
66
67 result: list[str] = []
68 index = 0
69 state = "normal"
70 while index < len(sql):
71 char = sql[index]
72 next_char = sql[index + 1] if index + 1 < len(sql) else ""
73
74 if state == "normal":
75 if char == "-" and next_char == "-":
76 result.extend((" ", " "))
77 index += 2
78 state = "line_comment"
79 continue
80 if char == "/" and next_char == "*":
81 result.extend((" ", " "))
82 index += 2
83 state = "block_comment"
84 continue
85 if char == "'":
86 result.append(" ")
87 index += 1
88 state = "single_quote"
89 continue
90 if char == '"':
91 result.append(" ")
92 index += 1
93 state = "double_quote"
94 continue
95 result.append(char)
96 index += 1
97 continue
98
99 if state == "line_comment":
100 if char in "\r\n":
101 result.append(char)
102 state = "normal"
103 else:
104 result.append(" ")
105 index += 1
106 continue
107
108 if state == "block_comment":
109 if char == "*" and next_char == "/":
110 result.extend((" ", " "))
111 index += 2
112 state = "normal"
113 else:
114 result.append("\n" if char == "\n" else " ")
115 index += 1
116 continue
117
118 quote = "'" if state == "single_quote" else '"'
119 if char == quote and next_char == quote:
120 result.extend((" ", " "))
121 index += 2
122 continue
123 result.append("\n" if char == "\n" else " ")
124 index += 1
125 if char == quote:
126 state = "normal"
127
128 if state in {"single_quote", "double_quote", "block_comment"}:
129 raise ValueError("조회 SQL의 따옴표 또는 블록 주석이 닫히지 않았습니다.")
130 return "".join(result)

133def validate_read_only_sql(sql: Any) -> str:
134 """단일 SELECT/WITH 조회문인지 확인하고 driver에 전달할 SQL을 반환합니다."""
135
136 text = str(sql or "").strip()
137 if not text:
138 raise ValueError("조회 SQL을 입력해 주세요.")
139
140 guard_text = _sql_guard_text(text)
141 semicolon_indexes = [index for index, char in enumerate(guard_text) if char == ";"]
142 if len(semicolon_indexes) > 1:
143 raise ValueError("한 번에 하나의 조회 SQL만 실행할 수 있습니다.")
144 if semicolon_indexes:
145 semicolon_index = semicolon_indexes[0]
146 if guard_text[semicolon_index + 1 :].strip():
147 raise ValueError("조회 SQL 뒤에 다른 문장을 함께 실행할 수 없습니다.")
148 # Oracle Python driver에는 마지막 세미콜론을 제외한 SQL을 전달합니다.
149 text = (text[:semicolon_index] + text[semicolon_index + 1 :]).strip()
150 guard_text = _sql_guard_text(text)
151
152 tokens = re.findall(r"[A-Za-z_][A-Za-z0-9_$#]*", guard_text.upper())
153 if not tokens or tokens[0] not in {"SELECT", "WITH"}:
154 raise ValueError("조회 전용 Component에서는 SELECT 또는 WITH로 시작하는 SQL만 허용합니다.")
155
156 forbidden = sorted(set(tokens).intersection(FORBIDDEN_SQL_KEYWORDS))
157 if forbidden:
158 raise ValueError(f"조회 SQL에 허용되지 않는 명령이 포함되어 있습니다: {', '.join(forbidden)}")
159 return text

162def normalize_bind_parameters(value: Any) -> dict[str, Any]:
163 """DictInput 또는 Langflow Data에 담긴 Oracle bind 변수 map을 정리합니다."""
164
165 if value in (None, ""):
166 return {}
167 if isinstance(value, Mapping):
168 source = value
169 else:
170 data = getattr(value, "data", None)
171 if not isinstance(data, Mapping):
172 raise ValueError("바인드 변수는 JSON 객체 형태로 입력해 주세요.")
173 source = data
174
175 result: dict[str, Any] = {}
176 for key, item in source.items():
177 name = str(key).strip()
178 if not name:
179 raise ValueError("바인드 변수 이름은 비어 있을 수 없습니다.")
180 if name.startswith(":"):
181 raise ValueError(f"바인드 변수 객체의 키에는 콜론을 붙이지 마세요: {name}")
182 result[name] = item
183 return result

223def _column_names(description: Any) -> list[str]:
224 """cursor description에서 중복되지 않는 컬럼명을 생성합니다."""
225
226 columns: list[str] = []
227 counts: dict[str, int] = {}
228 for index, column in enumerate(description or [], 1):
229 raw_name = getattr(column, "name", None)
230 if raw_name is None:
231 try:
232 raw_name = column[0]
233 except (TypeError, IndexError):
234 raw_name = f"COLUMN_{index}"
235 base_name = str(raw_name or f"COLUMN_{index}")
236 counts[base_name] = counts.get(base_name, 0) + 1
237 suffix = counts[base_name]
238 columns.append(base_name if suffix == 1 else f"{base_name}_{suffix}")
239 return columns

242def _table_value(value: Any) -> Any:
243 """Oracle 결과 값을 DataFrame과 JSON 화면에서 확인 가능한 값으로 바꿉니다."""
244
245 if value is None or isinstance(value, (int, float, bool)):
246 return value
247 if isinstance(value, str):
248 if len(value.encode("utf-8")) > MAX_CELL_BYTES:
249 raise ValueError(f"Oracle 셀 값이 허용 크기 {MAX_CELL_BYTES:,}바이트를 초과합니다.")
250 return value
251 if isinstance(value, (datetime, date)):
252 return value.isoformat()
253 if isinstance(value, Decimal):
254 return float(value)
255 if isinstance(value, (bytes, bytearray, memoryview)):
256 raw = bytes(value)
257 if len(raw) > MAX_CELL_BYTES:
258 raise ValueError(f"Oracle 셀 값이 허용 크기 {MAX_CELL_BYTES:,}바이트를 초과합니다.")
259 return base64.b64encode(raw).decode("ascii")
260
261 # CLOB/BLOB 같은 Oracle LOB 객체는 실제 값만 읽어서 테이블에 넣습니다.
262 reader = getattr(value, "read", None)
263 if callable(reader):
264 try:
265 loaded = reader(1, MAX_CELL_BYTES + 1)
266 except Exception:
267 raise ValueError("Oracle LOB 값을 제한 크기로 읽지 못했습니다.") from None
268 return _table_value(loaded)
269 return str(value)

272def _rows_to_records(rows: list[Any], columns: list[str]) -> list[dict[str, Any]]:
273 """cursor row를 컬럼명이 있는 record 목록으로 변환합니다."""
274
275 records: list[dict[str, Any]] = []
276 for row in rows:
277 if isinstance(row, Mapping):
278 record = {str(key): _table_value(item) for key, item in row.items()}
279 else:
280 record = {column: _table_value(item) for column, item in zip(columns, row, strict=False)}
281 records.append(record)
282 return records

296def execute_oracle_query(
297 *,
298 dsn: str,
299 username: str,
300 password: str,
301 sql: str,
302 bind_parameters: Mapping[str, Any] | None,
303 max_rows: int,
304 query_timeout_seconds: int,
305 connect_fn: Callable[..., Any] | None = None,
306) -> tuple[list[dict[str, Any]], list[str], bool]:
307 """Oracle 조회를 실행합니다.
308
309 ``connect_fn``을 주입할 수 있어 실제 DB 없이 connection/cursor 가짜 객체로
310 조회 계약과 리소스 종료를 테스트할 수 있습니다.
311 """
312
313 connection_factory = connect_fn or _load_oracle_connect()
314 connection = None
315 cursor = None
316 try:
317 connect_kwargs: dict[str, Any] = {"dsn": dsn}
318 if username and password:
319 connect_kwargs.update({"user": username, "password": password})
320 connection = connection_factory(**connect_kwargs)
321 # oracledb call_timeout은 각 DB 왕복 호출의 최대 시간이며 millisecond 단위입니다.
322 connection.call_timeout = query_timeout_seconds * 1000
323 cursor = connection.cursor()
324 parameters = dict(bind_parameters or {})
325 if parameters:
326 cursor.execute(sql, parameters)
327 else:
328 cursor.execute(sql)
329
330 description = getattr(cursor, "description", None)
331 if description is None:
332 raise RuntimeError("Oracle 조회 결과에 컬럼 정보가 없습니다.")
333 columns = _column_names(description)
334 fetched = list(cursor.fetchmany(max_rows + 1))
335 truncated = len(fetched) > max_rows
336 return _rows_to_records(fetched[:max_rows], columns), columns, truncated
337 finally:
338 _close_quietly(cursor)
339 _close_quietly(connection)

342def _safe_error_text(error: Exception, sensitive_values: list[str]) -> str:
343 """접속 정보가 노출되지 않도록 driver 오류 문자열을 짧게 정리합니다."""
344
345 if error.__class__.__module__ not in {"builtins", __name__}:
346 return f"외부 Oracle driver 오류가 발생했습니다 ({error.__class__.__name__})."
347 text = str(error).strip() or error.__class__.__name__
348 for value in sensitive_values:
349 if value:
350 text = text.replace(value, "[보호됨]")
351 return text[:500]

427 def build_data_table(self) -> DataFrame:
428 """입력값을 검증하고 Oracle 조회 결과만 DataFrame으로 반환합니다."""
429
430 dsn = str(getattr(self, "dsn", "") or "").strip()
431 if not dsn:
432 raise ValueError("Oracle DSN/TNS를 입력해 주세요.")
433
434 username = str(getattr(self, "username", "") or "").strip()
435 password = _secret_text(getattr(self, "password", ""))
436 if bool(username) != bool(password):
437 raise ValueError("사용자 ID와 비밀번호는 둘 다 입력하거나 둘 다 비워야 합니다.")
438
439 sql = validate_read_only_sql(getattr(self, "sql_query", ""))
440 parameters = normalize_bind_parameters(getattr(self, "bind_parameters", {}))
441 max_rows = normalize_max_rows(getattr(self, "max_rows", DEFAULT_MAX_ROWS))
442 query_timeout = normalize_query_timeout(
443 getattr(self, "query_timeout_seconds", DEFAULT_QUERY_TIMEOUT_SECONDS)
444 )
445
446 try:
447 rows, columns, truncated = execute_oracle_query(
448 dsn=dsn,
449 username=username,
450 password=password,
451 sql=sql,
452 bind_parameters=parameters,
453 max_rows=max_rows,
454 query_timeout_seconds=query_timeout,
455 )
456 except Exception as exc:
457 message = _safe_error_text(exc, [dsn, username, password, sql, *map(str, parameters.values())])
458 raise RuntimeError(f"Oracle 데이터 조회에 실패했습니다: {message}") from None
459
460 table = build_dataframe(rows, columns)
461 self.status = {
462 "조회 행 수": len(rows),
463 "컬럼 수": len(columns),
464 "최대 행 제한 적용": truncated,
465 }
