61def _parsed_url(value: Any, field_name: str = "API 주소"):
62 """호출 가능한 HTTP(S) URL을 검증하고 파싱 결과를 함께 반환합니다."""
63 url = _text_value(value).strip()
64 if not url:
65 raise ValueError(f"{field_name}을(를) 입력해 주세요.")
66 try:
67 parsed = urlsplit(url)
68 _ = parsed.port
69 except ValueError as exc:
70 raise ValueError(f"{field_name}의 포트 형식이 올바르지 않습니다.") from exc
71 if parsed.scheme.lower() not in {"http", "https"}:
72 raise ValueError(f"{field_name}은(는) http 또는 https 형식이어야 합니다.")
73 if not parsed.hostname:
74 raise ValueError(f"{field_name}에서 서버 주소를 확인할 수 없습니다.")
75 if parsed.username is not None or parsed.password is not None:
76 raise ValueError("URL 안에 사용자 이름이나 비밀번호를 넣을 수 없습니다.")
77 if parsed.fragment:
78 raise ValueError("API URL에는 #fragment를 넣을 수 없습니다.")
79 return url, parsed

83 """화면 문자열을 JSON으로 파싱하고 필요한 최상위 자료형을 확인합니다."""

97def _safe_headers(value: Any) -> dict[str, str]:
98 """헤더 JSON을 requests에 전달 가능한 안전한 문자열 사전으로 변환합니다."""
99 parsed = _parse_json(value, "요청 헤더 JSON", dict, blank_value={})
100 result: dict[str, str] = {}
101 for raw_name, raw_value in parsed.items():
102 name = str(raw_name).strip()
103 lowered = name.lower()
104 if not HEADER_NAME_PATTERN.fullmatch(name):
105 raise ValueError(f"요청 헤더 이름 '{name}'의 형식이 올바르지 않습니다.")
106 if lowered in BLOCKED_HEADERS:
107 raise ValueError(f"안전을 위해 '{name}' 헤더는 직접 지정할 수 없습니다.")
108 if raw_value is None or isinstance(raw_value, (dict, list)):
109 raise ValueError(f"요청 헤더 '{name}'의 값은 문자열이나 숫자여야 합니다.")
110 text = str(raw_value)
111 if "\r" in text or "\n" in text:
112 raise ValueError(f"요청 헤더 '{name}'의 값에는 줄바꿈을 넣을 수 없습니다.")
113 result[name] = text
114 return result

117def _extract_path(payload: Any, response_path: str) -> Any:
118 """점으로 구분한 응답 경로를 엄격하게 따라가 실제 데이터만 꺼냅니다."""
119 path = str(response_path or "").strip()
120 if not path:
121 return payload
122
123 current = payload
124 for token in [part.strip() for part in path.split(".") if part.strip()]:
125 if isinstance(current, dict):
126 if token not in current:
127 raise ValueError(f"응답 경로 '{path}'에서 '{token}' 키를 찾을 수 없습니다.")
128 current = current[token]
129 elif isinstance(current, list) and token.isdigit():
130 index = int(token)
131 if index >= len(current):
132 raise ValueError(f"응답 경로 '{path}'의 배열 위치 {index}가 범위를 벗어났습니다.")
133 current = current[index]
134 else:
135 raise ValueError(f"응답 경로 '{path}'를 끝까지 따라갈 수 없습니다.")
136 return current

146def _rows_from_json(value: Any) -> list[dict[str, Any]]:
147 """JSON 값 하나를 DataFrame에 바로 넣을 수 있는 행 목록으로 변환합니다."""
148 if value is None:
149 return []
150 items = value if isinstance(value, list) else [value]
151 rows: list[dict[str, Any]] = []
152 for item in items:
153 if isinstance(item, dict):

156 rows.append({"value": _table_cell(item)})

160def _header_value(response: Any, name: str) -> str:
161 """응답 헤더를 대소문자 차이와 관계없이 읽습니다."""
162 headers = getattr(response, "headers", None)
163 if not hasattr(headers, "get"):
164 return ""
165 direct = headers.get(name)
166 if direct is not None:
167 return str(direct)
168 lowered = name.lower()
169 for key, value in getattr(headers, "items", lambda: [])():
170 if str(key).lower() == lowered:
171 return str(value)
172 return ""

175def _read_response_bytes(response: Any, maximum_bytes: int) -> bytes:
176 """응답을 사용자가 지정한 제한 크기까지만 읽습니다."""
177 content_length = _header_value(response, "Content-Length").strip()
178 if content_length:
179 try:
180 declared_size = int(content_length)
181 except ValueError:
182 declared_size = 0
183 if declared_size > maximum_bytes:
184 raise ValueError(f"API 응답이 설정한 최대 크기 {maximum_bytes:,}바이트를 초과합니다.")
185
186 iterator = getattr(response, "iter_content", None)
187 if callable(iterator):
188 chunks: list[bytes] = []
189 total = 0
190 for chunk in iterator(chunk_size=64 * 1024):
191 if not chunk:
192 continue
193 chunk_bytes = bytes(chunk)
194 total += len(chunk_bytes)
195 if total > maximum_bytes:
196 raise ValueError(f"API 응답이 설정한 최대 크기 {maximum_bytes:,}바이트를 초과합니다.")
197 chunks.append(chunk_bytes)
198 return b"".join(chunks)
199
200 raw = getattr(response, "content", b"") or b""
201 raw_bytes = raw.encode("utf-8") if isinstance(raw, str) else bytes(raw)
202 if len(raw_bytes) > maximum_bytes:
203 raise ValueError(f"API 응답이 설정한 최대 크기 {maximum_bytes:,}바이트를 초과합니다.")
204 return raw_bytes

236def _validate_redirect_target(current_url: str, target_url: str, original_origin: tuple[str, str, int]) -> str:
237 """리다이렉트가 같은 서버 안에서 안전한 방향으로만 이동하는지 확인합니다."""
238 joined = urljoin(current_url, target_url)
239 validated, parsed = _parsed_url(joined, "리다이렉트 URL")
240 current_parsed = urlsplit(current_url)
241 if current_parsed.scheme.lower() == "https" and parsed.scheme.lower() != "https":
242 raise ValueError("HTTPS에서 HTTP로 낮아지는 API 리다이렉트는 허용하지 않습니다.")
243 if _origin(parsed) != original_origin:
244 raise ValueError("다른 origin(서버·scheme·port)으로 이동하는 API 리다이렉트는 허용하지 않습니다.")
245 return validated

248def _request_without_unsafe_redirects(
249 client: Any,
250 method: str,
251 url: str,
252 headers: dict[str, str],
253 query_params: dict[str, Any],
254 body: Any,
255 timeout_seconds: int,
256) -> Any:
257 """자동 리다이렉트를 끄고 같은 host의 GET 리다이렉트만 제한적으로 따라갑니다."""
258 sender = getattr(client, "request", None)
259 if not callable(sender):
260 raise ValueError("HTTP 전송 객체에 request 함수가 없습니다.")
261
262 original_origin = _origin(urlsplit(url))
263 current_url = url
264 current_query: dict[str, Any] | None = query_params
265
266 for redirect_count in range(MAX_REDIRECTS + 1):
267 request_options: dict[str, Any] = {
268 "headers": headers,
269 "params": current_query,
270 "timeout": timeout_seconds,
271 "verify": True,
272 "allow_redirects": False,
273 "stream": True,
274 }
275 if body is not _MISSING:
276 request_options["json"] = body
277
278 try:
279 response = sender(method, current_url, **request_options)
280 except Exception:
281 raise ValueError("API 서버에 연결하지 못했습니다.") from None
282
283 status_code = int(getattr(response, "status_code", 0) or 0)
284 actual_url = str(getattr(response, "url", current_url) or current_url)
285 try:
286 _validate_redirect_target(current_url, actual_url, original_origin)
287 except Exception:
288 _close_response(response)
289 raise
290
291 if status_code not in REDIRECT_STATUS_CODES:
292 return response
293
294 if method != "GET":
295 _close_response(response)
296 raise ValueError("POST 요청의 리다이렉트는 중복 호출 위험 때문에 자동 실행하지 않습니다.")
297 if redirect_count >= MAX_REDIRECTS:
298 _close_response(response)
299 raise ValueError(f"API 리다이렉트가 허용 횟수 {MAX_REDIRECTS}회를 초과했습니다.")
300 location = _header_value(response, "Location").strip()
301 if not location:
302 _close_response(response)
303 raise ValueError("API 리다이렉트 응답에 Location 헤더가 없습니다.")
304 try:
305 next_url = _validate_redirect_target(current_url, location, original_origin)
306 finally:
307 _close_response(response)
308 current_url = next_url
309 current_query = None
310
311 raise ValueError("API 리다이렉트를 처리하지 못했습니다.")

314def request_api_table(
315 api_url: Any,
316 http_method: Any = "GET",
317 headers_json: Any = "{}",
318 query_params_json: Any = "{}",
319 body_json: Any = "",
320 response_path: Any = "",
321 timeout_seconds: Any = 30,
322 max_response_bytes: Any = 10 * 1024 * 1024,
323 max_rows: Any = 5000,
324 allow_insecure_http: Any = False,
325 transport: Any | None = None,
326) -> list[dict[str, Any]]:
327 """API를 한 번 호출하고 JSON 응답의 데이터 행만 반환합니다.
328
329 transport는 테스트에서 가짜 HTTP 전송 객체를 주입하기 위한 선택 인자입니다.
330 실제 Component 실행에서는 requests 모듈을 사용합니다.
331 """
332 url, parsed_url = _parsed_url(api_url)
333 allow_http = _bool_value(allow_insecure_http, "HTTP API 사용 허용")
334 if parsed_url.scheme.lower() == "http" and not allow_http:
335 raise ValueError(
336 "요청 헤더나 본문을 HTTP로 보낼 수 없습니다. HTTPS URL을 사용하거나, "
337 "폐쇄된 테스트망에서만 'HTTP API 사용 허용'을 명시적으로 켜 주세요."
338 )
339 method = _text_value(http_method).strip().upper()
340 if method not in ALLOWED_METHODS:
341 raise ValueError(f"지원하지 않는 HTTP 방식입니다: {method or '(비어 있음)'}")
342
343 headers = _safe_headers(headers_json)
344 query_params = _parse_json(query_params_json, "URL 쿼리 파라미터 JSON", dict, blank_value={})
345 body_text = _text_value(body_json).strip()
346 body = _MISSING if not body_text else _parse_json(body_text, "요청 본문 JSON")
347 if method == "GET" and body is not _MISSING:
348 raise ValueError("GET 요청에는 요청 본문을 넣을 수 없습니다. URL 쿼리 파라미터를 사용해 주세요.")
349
350 timeout = _positive_int(timeout_seconds, "제한 시간", MAX_TIMEOUT_SECONDS)
351 response_limit = _positive_int(max_response_bytes, "최대 응답 크기", MAX_RESPONSE_BYTES)
352 row_limit = _positive_int(max_rows, "최대 반환 행 수", MAX_ROWS)
353
354 try:
355 client = transport or import_module("requests")
356 except ModuleNotFoundError as exc:
357 raise ValueError("API 호출에 필요한 requests 패키지가 설치되어 있지 않습니다.") from exc
358 response = _request_without_unsafe_redirects(
359 client=client,
360 method=method,
361 url=url,
362 headers=headers,
363 query_params=query_params,
364 body=body,
365 timeout_seconds=timeout,
366 )
367 try:
368 status_code = int(getattr(response, "status_code", 0) or 0)
369 if status_code < 200 or status_code >= 300:
370 raise ValueError(f"API가 성공 상태를 반환하지 않았습니다: HTTP {status_code}")
371 payload = _json_response(response, response_limit)
372 selected = _extract_path(payload, _text_value(response_path))
373 return _rows_from_json(selected)[:row_limit]
374 finally:
375 _close_response(response)
