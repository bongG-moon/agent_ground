478 height = 260
479 pad_x = 36
480 pad_top = 38

482 values = [value for _, value in points]

484 max_value = max(values)

486 coords = []
487 for index, (_, value) in enumerate(points):
488 px = pad_x + (width - pad_x * 2) * index / max(len(points) - 1, 1)
489 py = height - pad_bottom - (height - pad_top - pad_bottom) * (value - min_value) / span
490 coords.append((px, py))
491 polyline = " ".join(f"{xv:.1f},{yv:.1f}" for xv, yv in coords)
492 point_marks = []
493 for (label, value), (xv, yv) in zip(points, coords):
494 label_y = yv - 12 if yv > pad_top + 18 else yv + 22
495 anchor = "middle"
496 if xv < pad_x + 18:
497 anchor = "start"
498 elif xv > width - pad_x - 18:
499 anchor = "end"
500 point_marks.append(
501 f"""

950 return actual_number < expected_number
951 if operator == "lte":
952 return actual_number <= expected_number
953 actual_text = str(actual or "")
954 if operator in {"in", "not_in"}:
955 expected_values = expected if isinstance(expected, list) else [expected]
956 expected_texts = {str(item or "") for item in expected_values}
957 matched = actual_text in expected_texts
958 return not matched if operator == "not_in" else matched
959 expected_text = str(expected or "")
960 if operator == "contains":
961 return expected_text.lower() in actual_text.lower()
962 if operator == "ne":
963 return actual_text != expected_text
964 return actual_text == expected_text
